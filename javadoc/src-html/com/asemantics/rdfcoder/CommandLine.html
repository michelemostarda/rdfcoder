121    private String rewriteActions(String in) {
122        StringBuilder sb = new StringBuilder();
123        if(in.indexOf("read")    != -1) { sb.append("r"); }
124        if(in.indexOf("write")   != -1) { sb.append("w"); }
125        if(in.indexOf("execute") != -1) { sb.append("e"); }
126        if(in.indexOf("delete")  != -1) { sb.append("d"); }
127        return sb.toString();
128    }

130    private void lsDir(File target) {
131        if( ! target.exists() ) {
132            System.err.println( String.format("Cannot find dir '%s'", target.getAbsolutePath()) );
133            return;
134        }
135        if( target.isFile() ) {
136            System.out.println( target.getAbsolutePath() );
137            System.out.println();
138            return;
139        }
140        File[] content = target.listFiles();
141        FilePermission fp;
142        System.out.println( target.getAbsolutePath() );
143        System.out.println();
144        for(File f : content) {
145            fp = new FilePermission(f.getAbsolutePath(), "read,write,execute,delete");
146            System.out.printf(
147                    "%s\t%s\t\t\t%s\t%d\n",
148                    ( f.isDirectory() ? "d" : "-"),
149                    f.getName(),
150                    rewriteActions( fp.getActions() ),
151                    f.length()
152            );
153        }
154        System.out.println();
155    }

304        System.out.println("Model saved with parameters " + parameters.toString());
305    }

362    /**
363     * Command to obtain help on other command.
364     *
365     * @param args
366     * @throws IllegalAccessException
367     * @throws InvocationTargetException
368     */
369    public void command_help(String[] args) throws IllegalAccessException, InvocationTargetException {
370        if(args.length == 0) {
371            printUsage(System.out);
372            return;
373        }
374        if(args.length == 1) {
375            String longHelp = getLongCommandDescription(args[0]);
376            if(longHelp == null) {
377                System.out.println("cannot find help for command '" + args[0] + "'");
378            } else {
379                System.out.println(longHelp);
380            }
381            return;
382        }
383        System.out.println("invalid command");
384        System.out.println( __command_help() );
385    }

527    protected void configureCommandCompletors(ConsoleReader cr) {
528        CandidateListCompletionHandler completionHandler = new CandidateListCompletionHandler();
529        cr.setCompletionHandler(completionHandler);
530
531        // help completor.
532        final ArgumentCompletor helpCompletor = new ArgumentCompletor(
533                new Completor[]{
534                        new SimpleCompletor(new String[]{"help"}),
535                        new SimpleCompletor(getCommandNames()),
536                        new NullCompletor()
537                }
538        );
539
540        // loadclasspath completor.
541        final ArgumentCompletor loadClasspathCompletor = new ArgumentCompletor(
542                new Completor[]{
543                        new SimpleCompletor(new String[]{"loadclasspath"}),
544                        new SimpleCompletor("<modelname>"),
545                        new ArgumentCompletor(
546                            new FileNameCompletor()
547                        )
548                }
549        );
550
551        // Generic completor.
552        final ArgumentCompletor completor = new ArgumentCompletor(
553                new Completor[]{
554                        new SimpleCompletor(getCommandNames()),
555                        new ArgumentCompletor(
556                                new FileNameCompletor()
557                        )
558                }
559        );
560
561        final MultiCompletor multiCompletor = new MultiCompletor(
562                new Completor[]{
563                        helpCompletor,
564                        loadClasspathCompletor,
565                        completor
566                }
567        );
568        cr.addCompletor(multiCompletor);
569    }
