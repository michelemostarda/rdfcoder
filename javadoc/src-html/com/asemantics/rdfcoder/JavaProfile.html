112        /**
113         * Returns the <i>JRE</i> location on the basis of the Operative System.
114         *
115         * @return the file addressing the JRE root dir.
116         */
117        public static File getJRELocation() throws ProfileException {
118            final String javaHome = System.getProperty("java.home");
119            if(javaHome == null) {
120                throw new ProfileException("Error while retrieving the Java Home.");
121            }
122            return new File(javaHome);
123        }

168        /**
169         * Checks that in the current repository a <i>Java Runtime Environment</i>
170         * model already exists.
171         *
172         * @param pathToJRE
173         * @return <code>true</code> if the resource is already defined.
174         */
175        public boolean checkJREInit(File pathToJRE) {
176            if( pathToJRE == null ) {
177                throw new RDFCoderException("pathToJRE cannot be null.");
178            }
179            if( ! pathToJRE.exists() ) {
180                throw new RDFCoderException("specified JRE path: '" + pathToJRE.getAbsolutePath() + "' doesn't exist.");
181            }
182    
183            String jreResourceName = getJREModelResourceName( pathToJRE.getName() );
184            return repository.containsResource(jreResourceName);
185        }

187        public JREReport initJRE(File pathToJRE) {
188            if( pathToJRE == null ) {
189                throw new RDFCoderException("pathToJRE cannot be null.");
190            }
191            if( ! pathToJRE.exists() ) {
192                throw new RDFCoderException("specified JRE path: '" + pathToJRE.getAbsolutePath() + "' doesn't exist.");
193            }
194    
195            // Prepares jars list.
196            File[] libJars = new File( pathToJRE, JRE_LIB_DIR).listFiles( new CoderUtils.JavaJarFilter() );
197            File[] libExts = new File( pathToJRE, JRE_EXT_DIR).listFiles( new CoderUtils.JavaJarFilter() );
198            File[] libSecs = new File( pathToJRE, JRE_SEC_DIR).listFiles( new CoderUtils.JavaJarFilter() );
199            List<File> files = new ArrayList<File>(libJars.length + libExts.length + libSecs.length);
200            files.addAll( Arrays.asList(libJars) );
201            files.addAll( Arrays.asList(libExts) );
202            files.addAll( Arrays.asList(libSecs) );
203    
204            // Creates structures.
205            JStatistics statistics    = new JStatistics();
206            ObjectsTable objectsTable = model.getObjectsTable();
207            CodeModelBase cmb         = model.getCodeModelBase();
208            JavaCodeHandler javaCodeHandler = model.getCoderFactory().createHandlerOnModel(cmb);
209    
210            CodeStorage cs = model.getCoderFactory().createCodeStorage();
211            if( ! cs.supportsFile() ) {
212                throw new IllegalStateException();
213            }
214    
215            // Initializes structures.
216            final String jreName = pathToJRE.getName();
217            JavaBytecodeJarParser parser = new JavaBytecodeJarParser();
218            JavaCodeHandler statCH = statistics.createStatisticsCodeHandler(javaCodeHandler);
219            parser.initialize(statCH, objectsTable);
220            javaCodeHandler.startParsing( jreName, pathToJRE.getAbsolutePath() );
221    
222            // Does jar parsing.
223            for(File f : files) {
224                try {
225                    parser.parseFile(f);
226                } catch (Exception e) {
227                    logger.error( String.format("Error while parsing file [%s]", f.getAbsolutePath() ), e );
228                }
229            }
230    
231            // Disposes parsing.
232            javaCodeHandler.endParsing();
233    
234            // Stores the objects table content.
235            try {
236                serializeJREObjectsTable(jreName, objectsTable);
237            } catch (Exception e) {
238                throw new RDFCoderException("Cannot serialize Object Table in repository.", e);   
239            }
240    
241            objectsTable = null;
242            parser.dispose();
243            parser = null;
244    
245            // Saves model in repository.
246            OutputStream os;
247            Repository.Resource resource = null;
248            try {
249                resource = repository.createResource(
250                        getJREModelResourceName(pathToJRE.getName()),
251                        Repository.ResourceType.XML
252                );
253                os  = resource.getOutputStream();
254                codeStorage.saveModel(cmb, os);
255                os.close();
256            } catch (Exception e) {
257                resource.delete();
258                throw new RDFCoderException("Cannot store model in repository.", e);
259            }
260    
261            // Disposes model.
262            cmb.clearAll();
263            cmb = null;
264    
265            return new JREReport(pathToJRE ,statistics);
266        }

268        /**
269         * Loads an existing JRE data.
270         *
271         * @param pathToJRE
272         */
273        public void loadJRE(File pathToJRE) {
274            try {
275                model.getObjectsTable().load( deserializeJREObjectsTable( pathToJRE.getName() ) );
276            } catch (RepositoryException re) {
277                logger.info("Cannot retrieve Objects Table.", re);
278            } catch (InvalidClassException ice) {
279                logger.info("Obsolete Objects Table, removing it.", ice);
280                removeObjectsTable( pathToJRE.getName() );
281            } catch (Exception e) {
282                throw new RDFCoderException("A generic error occurred while loading the Objects Table.", e);
283            }
284        }

286        /**
287         * This utility method initializes or loads the given <i>JRE</i> profile depending if it exists or not.
288         *
289         * @param pathToJRE the location of the JRE.
290         * @return the initialization report if the init happened, <code>null</code> otherwise.
291         * @throws ProfileException
292         */
293        public JREReport initOrLoadJRE(File pathToJRE) throws ProfileException {
294            if ( ! checkJREInit(pathToJRE)) {
295                try {
296                    return initJRE(pathToJRE);
297                } catch (Throwable t) {
298                    throw new ProfileException("An error occurred while init the JRE profile.", t);
299                }
300            } else {
301                try {
302                    loadJRE(pathToJRE);
303                    return null;
304                } catch (Throwable t) {
305                    throw new ProfileException("An error occurred while loading the JRE profile.", t);
306                }
307            }
308        }

328        public JStatistics loadJar(String libName, String pathToJar) throws IOException, ParserException {
329            JStatistics statistics = new JStatistics();
330            JavaCodeHandler javaCodeHandler = model.getCoderFactory().createHandlerOnModel( model.getCodeModelBase() );
331    
332            JavaBytecodeJarParser parser = new JavaBytecodeJarParser();
333            JavaCodeHandler statCH = statistics.createStatisticsCodeHandler(javaCodeHandler);
334            parser.initialize( statCH, model.getObjectsTable() );
335    
336            File lib = new File(pathToJar);
337    
338            javaCodeHandler.startParsing(libName, lib.getAbsolutePath());
339            parser.parseFile( new File(pathToJar) );
340            javaCodeHandler.endParsing();
341    
342            statistics.detachHandlers();
343            parser.dispose();
344            parser = null;
345    
346            return statistics;
347        }

349        /**
350         * Loads Java resources.
351         * 
352         * @param libName
353         * @param path
354         * @param fileParser
355         * @return
356         */
357        private JStatistics loadJava(String libName, String path, FileParser fileParser, FilenameFilter filenameFilter) {
358            JavaCodeHandler ch = model.getCoderFactory().createHandlerOnModel( model.getCodeModelBase() );
359    
360            DirectoryParser directoryParser = new DirectoryParser( fileParser, filenameFilter );
361            JStatistics statistics = new JStatistics();
362            JavaCodeHandler sch = statistics.createStatisticsCodeHandler(ch);
363    
364            directoryParser.initialize(sch, model.getObjectsTable() );
365            directoryParser.parseDirectory(libName, new File(path) );
366    
367            directoryParser.dispose();
368            ch = null;
369    
370            return statistics;
371        }

373        /**
374         * Serializes the content of the JRE {@link com.asemantics.rdfcoder.parser.ObjectsTable}
375         * into the repository.
376         *
377         * @param objectsTable
378         */
379        private void serializeJREObjectsTable(String jreName, ObjectsTable objectsTable)
380        throws RepositoryException, IOException {
381            Repository.Resource resource = repository.createResource(
382                    getJREObjectTableResourceName( jreName ),
383                    Repository.ResourceType.BINARY
384            );
385            OutputStream os = resource.getOutputStream();
386            ObjectOutputStream oos = null;
387            try {
388                oos = new ObjectOutputStream(os);
389                oos.writeObject( objectsTable );
390            } finally {
391                if(oos != null) { oos.close(); }
392            }
393        }
394    
395        /**
396         * Deserializes the content of the <i>JRE</i> {@link com.asemantics.rdfcoder.parser.ObjectsTable}
397         * from the repository.
398         * 
399         * @param jreName
400         * @return
401         * @throws RepositoryException
402         * @throws IOException
403         * @throws ClassNotFoundException
404         */
405        private ObjectsTable deserializeJREObjectsTable(String jreName)
406        throws RepositoryException, IOException, ClassNotFoundException {
407            Repository.Resource resource = repository.getResource( getJREObjectTableResourceName( jreName ) );
408            InputStream is = resource.getInputStream();
409            ObjectInputStream ois = null;
410            try {
411                ois = new ObjectInputStream(is);
412                ObjectsTable ot = (ObjectsTable) ois.readObject();
413                return ot;
414            } finally {
415                if(ois != null) { ois.close(); }
416            }
417        }

419        /**
420         * Removes the object table associated to the given <i>JRE</i>.
421         *
422         * @param jreName
423         */
424        private void removeObjectsTable(String jreName) {
425            try {
426                repository.removeResource( getJREObjectTableResourceName( jreName ) );
427            } catch (RepositoryException re) {
428                logger.error("Error while removing object table.", re);
429            }
430        }
