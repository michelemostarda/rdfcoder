019    package com.asemantics.rdfcoder.model.ontology;
020    
021    import static com.asemantics.rdfcoder.model.CodeModelBase.ASSET;
022    import static com.asemantics.rdfcoder.model.CodeModelBase.ASSET_PREFIX;
023    import static com.asemantics.rdfcoder.model.CodeModelBase.CONTAINS_LIBRARY;
024    import static com.asemantics.rdfcoder.model.CodeModelBase.LIBRARY_DATETIME;
025    import static com.asemantics.rdfcoder.model.CodeModelBase.LIBRARY_LOCATION;
026    
027    import java.net.URL;
028    
029    
030    /**
031     * Defines ontology predicates common to every <i>Ontology</i>.
032     */
033    public abstract class BaseOntology extends DefaultOntology {
034    
035        protected BaseOntology() {
036            try {
037                defineRelation(ASSET, new URL(CONTAINS_LIBRARY), ASSET_PREFIX);
038                defineRelation(ASSET, new URL(LIBRARY_LOCATION));
039                defineRelation(ASSET, new URL(LIBRARY_DATETIME));
040            } catch (Exception e) {
041                 throw new RuntimeException( "Error while defining Base Ontology.", e);
042            }
043        }
044    
045    }
