083    public void undefineRelation(String subjectPrefix, URL predicate) throws OntologyException {
084        validatePrefix(subjectPrefix);
085
086        final String predicateStr = predicate.toString();
087        final List<PropertyObjectBase> properties = getPropertiesOrException(predicateStr, "Cannot undefine relation");
088        for(PropertyObjectBase base : properties) {
089            if( ((LiteralObjectProperty) base).subPrefixStr.equals( subjectPrefix ) ) {
090                properties.remove(base);
091            }
092        }
093        updateSession();
094    }
095
096    public void undefineRelationList(String subjectPrefix, URL predicate) throws OntologyException {
097        undefineRelation(subjectPrefix, predicate);
098    }
099
100    public void undefineRelation(String subjectPrefix, URL predicate, String objectPrefix) throws OntologyException {
101        validatePrefix(subjectPrefix);
102
103        final String predicateStr = predicate.toString();
104        final List<PropertyObjectBase> properties = getPropertiesOrException(predicateStr, "Cannot undefine relation");
105        for(PropertyObjectBase base : properties) {
106            URIObjectProperty property = (URIObjectProperty) base;
107            if(
108                    property.subPrefixStr.equals( subjectPrefix )
109                        &&
110                    property.objPrefixStr.equals( objectPrefix  )
111            ) {
112                properties.remove(property);
113            }
114        }
115        updateSession();
116    }
117
118    protected void validatePrefix(String prefix) throws OntologyException {
119        if( prefix == null ) {
120            throw new OntologyException( String.format("invalid prefix: '%s'", prefix) );
121        }
122    }
123
124    protected void updateSession() {
125        session++;
126    }
127
128    protected List<PropertyObjectBase> getProperties(String predicate) {
129        List<PropertyObjectBase> properties = predicateIndex.get(predicate);
130        if(properties == null) {
131            properties = new ArrayList<PropertyObjectBase>();
132            predicateIndex.put(predicate, properties);
133        }
134        return properties;
135    }
136
137
138    protected List<PropertyObjectBase> getPropertiesOrException(String predicate, String msgPrefix)
139    throws OntologyException {
140        List<PropertyObjectBase> properties = predicateIndex.get(predicate);
141        if (properties == null) {
142            throw new OntologyException(
143                    String.format("%s '%s' because is undefined.", msgPrefix, predicate)
144            );
145        }
146        return properties;
147    }
148
149    protected void addPropertyInList( List<PropertyObjectBase> properties, PropertyObjectBase property)
150    throws OntologyException {
151        if(properties.contains(property)) {
152            throw new OntologyException("Property rule redefinition");
153        }
154        properties.add(property);
155        updateSession();
156    }

158    /**
159     * Validates a triple over the predicates list.
160     *
161     * @param subject
162     * @param predicate
163     * @param object
164     * @param literal
165     * @return the property base applicable on validation.
166     * @throws OntologyException
167     */
168    protected PropertyObjectBase validateTerms(String subject, String predicate, Object object, boolean literal)
169    throws OntologyException {
170        List<PropertyObjectBase> properties = predicateIndex.get(predicate);
171
172        if(properties == null) {
173            throw new OntologyException( String.format("predicate '%s' is not defined.", predicate) );
174        }
175
176        List<OntologyException> causes = new ArrayList<OntologyException>();
177        for(PropertyObjectBase property : properties) {
178            try {
179                if( literal == property.isLiteral() ) {
180                    property.validate(subject, predicate, object);                    
181                    return property;
182                }
183            } catch (OntologyException oe) {
184                causes.add(oe);
185            }
186        }
187        throw new OntologyException(
188                String.format("No property matches specified terms for predicate: '%s'", predicate)
189        );
190    }
191
192    public void validateTriple(String subject, String predicate, Object object) throws OntologyException {
193        validateTerms(subject, predicate,  object, false);
194    }
195
196    public void validateTripleLiteral(String subject, String predicate) throws OntologyException {
197        validateTerms(subject, predicate,  null, true);
198    }

203    private long session = 0L;

205    /**
206     * Session in which properties has been defined last time.
207     */
208    private long propertiesSession;

210    /**
211     * List of ordered properties.
212     */
213    private PropertyObjectBase[] properties;
214
215    /**
216     * Populates the properties array.
217     */
218    private void orderRelations() {
219        List<PropertyObjectBase> relationsList = new ArrayList<PropertyObjectBase>();
220        for(List<PropertyObjectBase> propertyList :  predicateIndex.values() ) {
221            relationsList.addAll( propertyList );
222        }
223        Collections.sort( relationsList );
224        properties = relationsList.toArray( new PropertyObjectBase[ relationsList.size() ] );
225        propertiesSession = session;
226    }
227
228    public int getRelationsCount() {
229        if(properties == null || propertiesSession != session) {
230            orderRelations();
231        }
232        return properties.length;
233    }

267    /**
268     * Defines a base property.
269     */
270    abstract class PropertyObjectBase implements Comparable {
271
272        /**
273         * Returns the prefix of a term.
274         *
275         * @param term
276         * @return
277         * @throws OntologyException
278         */
279        String getPrefix(String term) throws OntologyException {
280            try {
281                return IdentifierReader.readIdentifier(term).getStrongestQualifier() + CodeModel.PREFIX_SEPARATOR;
282            } catch (Exception e) {
283                throw new OntologyException("Invalid term.", e);
284            }
285        }
286
287        /**
288         * Validates a triple over this property.
289         *
290         * @param subject
291         * @param predicate
292         * @param object
293         * @throws OntologyException
294         */
295        abstract void validate(String subject, String predicate, Object object) throws OntologyException;
296
297        /**
298         * Returns <code>true</code> if this is a literal property, <code>false</code>
299         * otherwise.
300         *
301         * @return
302         */
303        abstract boolean isLiteral();
304
305        /**
306         * Prints the current property as a human readable string on the given print stream.
307         *
308         * @param ps
309         */
310        abstract void print(PrintStream ps);
311
312    }

314    /**
315     * Represents a literal property.
316     */
317    class LiteralObjectProperty extends PropertyObjectBase {
318
319        /**
320         * The subject expected prefix.
321         */
322        final String subPrefixStr;
323
324        /**
325         * predicate string.
326         */
327        final String predicateStr;
328
329        /**
330         * Constructor.
331         *
332         * @param subjectPrefix
333         * @param predicate
334         */
335        LiteralObjectProperty(String subjectPrefix, String predicate) {
336            subPrefixStr = subjectPrefix;
337            predicateStr = predicate;
338        }
339
340        public String getSubjectPrefix() {
341            return subPrefixStr;
342        }
343
344        public String getPredicate() {
345            return predicateStr;
346        }
347
348        public void validate(String subject, String predicate, Object object) throws OntologyException {
349            if( subPrefixStr == null ) {
350                return;
351            }
352
353            String subPrefix = getPrefix(subject);
354            if(
355                ! subPrefixStr.equals( subPrefix )
356            ) {
357                throw new OntologyException(
358                    String.format("Invalid subject prefix: '%s' for predicate: '%s'", subPrefix, predicateStr)
359                );
360            }
361        }
362
363        boolean isLiteral() {
364            return true;
365        }
366
367        void print(PrintStream ps) {
368            ps.printf("[%s] --[%s]--> 'LITERAL'", subPrefixStr, predicateStr);
369            ps.println();
370        }
371
372        public boolean equals(Object obj) {
373            if(obj == null) {
374                return false;
375            }
376            if(obj == this) {
377                return true;
378            }
379            if(obj instanceof LiteralObjectProperty) {
380                LiteralObjectProperty other = (LiteralObjectProperty) obj;
381                return
382                        subPrefixStr.equals(other.subPrefixStr)
383                                &&
384                        predicateStr.equals(other.predicateStr);
385            }
386            return false;
387        }
388
389        public int hashCode() {
390            return subPrefixStr.hashCode() * predicateStr.hashCode();
391        }
392
393        /**
394         * Ordered over predicate strings.
395         *
396         * @param obj
397         * @return
398         */
399        public int compareTo(Object obj) {
400            LiteralObjectProperty literalObjectProperty = (LiteralObjectProperty) obj;
401            int cmp = predicateStr.compareTo(literalObjectProperty.predicateStr);
402            return cmp != 0 ? cmp : subPrefixStr.compareTo(literalObjectProperty.subPrefixStr);
403        }
404    }

406    /**
407     * Represents a property.
408     */
409    class URIObjectProperty extends LiteralObjectProperty {
410
411        /**
412         * Object prefix string.
413         */
414        String objPrefixStr;
415
416        /**
417         * Constructor.
418         *
419         * @param subjectPrefix
420         * @param predicate
421         * @param objectPrefix
422         */
423        URIObjectProperty(String subjectPrefix, String predicate, String objectPrefix) {
424            super(subjectPrefix, predicate);
425            objPrefixStr = objectPrefix;
426        }
427
428        public String getObjectPrefix() {
429            return objPrefixStr;
430        }
431
432        public void validate(String subject, String predicate, Object object) throws OntologyException {
433            if( ! (object instanceof String) ) {
434                throw new IllegalArgumentException("object must be a string.");
435            }
436
437            final String objectStr = (String) object;
438
439            super.validate(subject, predicate, objectStr);
440
441            if(objPrefixStr == null) {
442                return;
443            }
444
445            String objPrefix = getPrefix(objectStr);
446            if(
447                ! objPrefix.equals( objPrefixStr )
448            ) {
449                throw new OntologyException(
450                    String.format("Invalid object prefix: '%s' for predicate: '%s'", objPrefixStr, predicate)
451                );
452            }
453        }
454
455        boolean isLiteral() {
456            return false;
457        }
458
459        void print(PrintStream ps) {
460            ps.printf("[%s] --[%s]--> [%s]", subPrefixStr, predicateStr, objPrefixStr);
461            ps.println();
462        }
463
464        public boolean equals(Object obj) {
465            if(obj == null) {
466                return false;
467            }
468            if(obj == this) {
469                return true;
470            }
471
472            if( ! super.equals(obj)) {
473                return false;
474            }
475
476            if(obj instanceof URIObjectProperty) {
477                URIObjectProperty other = (URIObjectProperty) obj;
478                return objPrefixStr.equals(other.objPrefixStr);
479            }
480            return false;
481        }
482
483        public int hashCode() {
484            return super.hashCode() * objPrefixStr.hashCode();
485        }
486    }

488    class ListObjectProperty extends LiteralObjectProperty {
489
490        private ListBounds listBounds;
491
492        ListObjectProperty(String subjectPrefix, String predicate, ListBounds listBounds) {
493            super(subjectPrefix, predicate);
494            this.listBounds = listBounds;
495        }
496
497        public void validate(String subject, String predicate, Object object) throws OntologyException {
498            if( ! (object.getClass().isArray() && object.getClass().getComponentType().equals(String.class)) ) {
499                throw new IllegalArgumentException("object must be an String array.");
500            }
501
502            super.validate(subject, predicate, object);
503
504            if( ! listBounds.inBounds( (String[]) object )) {
505                throw new OntologyException("Invalid object size.");
506            }
507        }
508
509        @Override
510        boolean isLiteral() {
511            return false;
512        }
513
514        @Override
515        void print(PrintStream ps) {
516            ps.printf("[%s] --[%s]--> %s", subPrefixStr, predicateStr, listBounds);
517            ps.println();
518        }
519
520        @Override
521        public boolean equals(Object obj) {
522            if(obj == null) {
523                return false;
524            }
525            if(obj == this) {
526                return true;
527            }
528            if(obj instanceof ListObjectProperty) {
529                final ListObjectProperty other = (ListObjectProperty) obj;
530                return super.equals(other) && listBounds.equals(other.listBounds);
531            }
532            return false;
533        }
534
535        @Override
536        public int hashCode() {
537            return super.hashCode() * listBounds.hashCode();
538        }
539    }
