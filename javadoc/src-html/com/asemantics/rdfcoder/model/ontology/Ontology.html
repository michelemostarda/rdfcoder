019    package com.asemantics.rdfcoder.model.ontology;
020    
021    import java.io.OutputStream;
022    import java.io.PrintStream;
023    import java.net.URL;
024    
025    /**
026     * Defines the ontology generated by a {@link com.asemantics.rdfcoder.model.java.JavaCodeHandler}
027     */
028    public interface Ontology {
029    
030        /**
031         * Defines a relation <i>Subject</i>, <i>predicate</i>,
032         * <i>object</i> in the ontology.
033         *
034         * @param subjectPrefix
035         * @param predicate
036         * @param objectPrefix
037         */
038        void defineRelation(String subjectPrefix, URL predicate, String objectPrefix) throws OntologyException;
039    
040        /**
041         * Defines a relation <i>Subject</i>, <i>predicate</i>,
042         * <i>literal object</i> in the ontology.
043         *
044         * @param subjectPrefix
045         * @param predicate
046         */
047        void defineRelation(String subjectPrefix, URL predicate) throws OntologyException;
048    
049        /**
050         * Defines a relation *  , <i>predicate</i> , * in the ontology. 
051         *
052         * @param predicate
053         */
054        void defineRelation(URL predicate) throws OntologyException;
055    
056        /**
057         * Removes an already defined relation from the ontology.
058         *
059         * @param subjectPrefix
060         * @param predicate
061         * @param objectPrefix
062         */
063        void undefineRelation(String subjectPrefix, URL predicate, String objectPrefix) throws OntologyException;
064    
065        /**
066         *  Removes an already defined relation from the ontology.
067         *
068         * @param subjectPrefix
069         * @param predicate
070         */
071        void undefineRelation(String subjectPrefix, URL predicate) throws OntologyException;
072    
073        /**
074         * Validates a triple over the current ontology.
075         *
076         * @param subject the triple subject.
077         * @param predicate the triple predicate.
078         * @param object the triple object.
079         */
080        void validateTriple(String subject, String predicate, String object) throws OntologyException;
081    
082        /**
083         * Validates a triple literal over the current ontology.
084         *
085         * @param subject the triple subject.
086         * @param predicate the triple object.
087         */
088        void validateTripleLiteral(String subject, String predicate) throws OntologyException;
089    
090        /**
091         * Returns the number of relations defined in the ontology.
092         *
093         * @return a positive number.
094         */
095        int getRelationsCount();
096    
097        /**
098         * Returns <code>true</code> if the i-th <i>object</i> is a literal,
099         * <code>false</code> otherwise.
100         * 
101         * @param i
102         * @return <code>true<code> if literal.
103         */
104        boolean isLiteralRelation(int i);
105    
106        /**
107         * Returns the relation i-th subject prefix.
108         *
109         * @param i
110         * @return the subject prefix string.
111         */
112        String getRelationSubjectPrefix(int i);
113    
114         /**
115         * Returns the relation i-th predicate.
116          *
117         * @param i
118         * @return the predicate.
119         */
120        URL getRelationPredicate(int i);
121    
122        /**
123         * Returns the relation i-th object prefix.
124         *
125         * @param i index discriminative of the object.
126         * @return the prefix.
127         */
128        String getRelationObjectPrefix(int i);
129    
130        /**
131         * Prints the current ontology in a readable form.
132         *
133         * @param ps
134         */
135        void printOntology(PrintStream ps);
136    
137        /**
138         * Prints the current ontology in OWL format.
139         *
140         * @param os
141         */
142        void toOWL(OutputStream os);
143    
144    }
