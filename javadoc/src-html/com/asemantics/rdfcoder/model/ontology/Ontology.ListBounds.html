019package com.asemantics.rdfcoder.model.ontology;
020
021import java.io.OutputStream;
022import java.io.PrintStream;
023import java.net.URL;
024
025/**
026 * Defines the ontology generated by a {@link com.asemantics.rdfcoder.model.java.JavaCodeHandler}
027 */
028public interface Ontology {
029
030    /**
031     * Defines a relation <i>Subject</i>, <i>predicate</i>,
032     * <i>object</i> in the ontology.
033     *
034     * @param subjectPrefix
035     * @param predicate
036     * @param objectPrefix
037     */
038    void defineRelation(String subjectPrefix, URL predicate, String objectPrefix) throws OntologyException;
039
040    /**
041     * Defines a relation <i>Subject</i>, <i>predicate</i>,
042     * <i>list</i> in the ontology.
043     *
044     * @param subjectPrefix
045     * @param predicate
046     * @param listBounds
047     * @throws OntologyException
048     */
049    void defineRelation(String subjectPrefix, URL predicate, ListBounds listBounds) throws OntologyException;
050
051    /**
052     * Defines a relation <i>Subject</i>, <i>predicate</i>,
053     * <i>literal object</i> in the ontology.
054     *
055     * @param subjectPrefix
056     * @param predicate
057     */
058    void defineRelation(String subjectPrefix, URL predicate) throws OntologyException;
059
060    /**
061     * Defines a relation *  , <i>predicate</i> , * in the ontology. 
062     *
063     * @param predicate
064     */
065    void defineRelation(URL predicate) throws OntologyException;
066
067    /**
068     * Removes an already defined relation from the ontology.
069     *
070     * @param subjectPrefix
071     * @param predicate
072     * @param objectPrefix
073     */
074    void undefineRelation(String subjectPrefix, URL predicate, String objectPrefix) throws OntologyException;
075
076    /**
077     *  Removes an already defined relation from the ontology.
078     *
079     * @param subjectPrefix
080     * @param predicate
081     */
082    void undefineRelation(String subjectPrefix, URL predicate) throws OntologyException;
083
084    /**
085     * Removes an already defined relation wich object is a list.
086     *
087     * @param subjectPrefix
088     * @param predicate
089     */
090    void undefineRelationList(String subjectPrefix, URL predicate) throws OntologyException;
091
092    /**
093     * Validates a triple over the current ontology.
094     *
095     * @param subject the triple subject.
096     * @param predicate the triple predicate.
097     * @param object the triple object.
098     */
099    void validateTriple(String subject, String predicate, Object object) throws OntologyException;
100
101    /**
102     * Validates a triple literal over the current ontology.
103     *
104     * @param subject the triple subject.
105     * @param predicate the triple object.
106     */
107    void validateTripleLiteral(String subject, String predicate) throws OntologyException;
108
109    /**
110     * Returns the number of relations defined in the ontology.
111     *
112     * @return a positive number.
113     */
114    int getRelationsCount();
115
116    /**
117     * Returns <code>true</code> if the i-th <i>object</i> is a literal,
118     * <code>false</code> otherwise.
119     * 
120     * @param i
121     * @return <code>true</code> if literal.
122     */
123    boolean isLiteralRelation(int i);
124
125    /**
126     * Returns the relation i-th subject prefix.
127     *
128     * @param i
129     * @return the subject prefix string.
130     */
131    String getRelationSubjectPrefix(int i);
132
133     /**
134     * Returns the relation i-th predicate.
135      *
136     * @param i
137     * @return the predicate.
138     */
139    URL getRelationPredicate(int i);
140
141    /**
142     * Returns the relation i-th object prefix.
143     *
144     * @param i index discriminative of the object.
145     * @return the prefix.
146     */
147    String getRelationObjectPrefix(int i);
148
149    /**
150     * Prints the current ontology in a readable form.
151     *
152     * @param ps
153     */
154    void printOntology(PrintStream ps);
155
156    /**
157     * Prints the current ontology in OWL format.
158     *
159     * @param os
160     */
161    void toOWL(OutputStream os);
162
163    /**
164     * Declares the restriction bounds for an array.
165     */
166    class ListBounds {
167
168        int minSize;
169        int maxSize;
170
171        /**
172         * Constructor for bounds in interval <i>[minSize, maxSize]</i>.
173         *
174         * @param minSize minimum accepted size.
175         * @param maxSize maximum expected size.
176         */
177        public ListBounds(int minSize, int maxSize) {
178            this.minSize = minSize;
179            this.maxSize = maxSize;
180        }
181
182        public ListBounds() {
183            this(0, Integer.MAX_VALUE);
184        }
185
186        boolean inBounds(String[] values) {
187            return values.length >= minSize && values.length <= maxSize;
188        }
189
190        @Override
191        public String toString() {
192            return String.format("%s %d %d", ListBounds.this.getClass().getSimpleName(), minSize, maxSize);
193        }
194
195        @Override
196        public boolean equals(Object obj) {
197            if(obj == null) {
198                return false;
199            }
200            if(obj == this) {
201                return true;
202            }
203            if(obj instanceof ListBounds) {
204                final ListBounds other = (ListBounds) obj;
205                return minSize == other.minSize && maxSize == other.maxSize;
206            }
207            return false;
208        }
209
210        @Override
211        public int hashCode() {
212            return minSize * maxSize;
213        }
214    }
215}
