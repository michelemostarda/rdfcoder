019    package com.asemantics.rdfcoder.model;
020    
021    /**
022     * Defines the <i>Code Model</i>, that is a graph representing
023     * the <i>Kabbalah</i> model. 
024     *
025     * @author Michele Mostarda
026     */
027    public interface CodeModel {
028        
029        /**
030         * RDF schema prefix URI.
031         */
032        static final String RDF_SCHEMA_URI = "http://www.w3.org/2000/01/rdf-schema#";
033    
034        /**
035         * Defines the relation subclass of.
036         */
037        static final String SUBCLASSOF = RDF_SCHEMA_URI + "subClassOf";
038    
039        /**
040         * The separator used for URI prefixes.
041         */
042        static final String URI_PREFIX_SEPARATOR = "#";
043    
044        /**
045         * Coder base prefix URI.
046         */
047        static final String CODER_URI_BASE = "http://www.rdfcoder.org/2007/1.0";
048    
049        /**
050         * Coder prefix URI with prefix separator.
051         */
052        static final String CODER_URI = CODER_URI_BASE + URI_PREFIX_SEPARATOR;
053    
054        /**
055         * Prefix separator definition.
056         */
057        static final String PREFIX_SEPARATOR = ":";
058        
059        /**
060         * The temporary Type ID prefix.
061         */
062        static final String TEMPORARY_TYPE_ID_PREFIX = "tmpTID_";
063    
064        /**
065         * No element specified.
066         */
067        static final String ALL_MATCH = null;
068    
069        /**
070         * Allows to perform a triple search on the org.asemantics.model.
071         * @param subject
072         * @param predicate
073         * @param object
074         * @return the iterator to access the search result.
075         */
076        TripleIterator searchTriples(String subject, String predicate, String object);
077    
078        /**
079         * Allows to add a triple into the model.
080         * NOTE: this method accepts only resource objects.
081         *
082         * @param subject a subject resource
083         * @param predicate a predicate
084         * @param object a resource object
085         */
086        void addTriple(String subject, String predicate, String object);
087    
088        /**
089         * Allows to remove a triple from the model.
090         * NOTE: this method accepts only resource objects.
091         *
092         * @param subject
093         * @param predicate
094         * @param object
095         * @see #addTriple(String, String, String)
096         */
097        void removeTriple(String subject, String predicate, String object);
098    
099        /**
100         * Allows to add a triple into the model.
101         * NOTE: this method accepts only literal object.
102         *
103         * @param subject a subject resource
104         * @param predicate a predicate
105         * @param literal a literal object
106         */
107        void addTripleLiteral(String subject, String predicate, String literal);
108    
109        /**
110         * Allows to remove a triple from the model.
111         * NOTE: this method accepts only literal object.
112         *
113         * @param subject the triple subject.
114         * @param predicate the triple predicate.
115         * @param object the triple object.
116         * @see #addTripleLiteral(String, String, String)
117         */
118        void removeTripleLiteral(String subject, String predicate, String object);
119    
120        /**
121         * Adds a triple which object is a collection.
122         *
123         * @param subject the triple subject.
124         * @param predicate the triple predicate.
125         * @param values the triple values list.
126         */
127        void addTripleCollection(Object subject, String predicate, String[] values);
128    
129        /**
130         *  Remove the entire content of the model.
131         */
132        void clearAll();
133    
134    }
