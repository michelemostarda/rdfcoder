019    package com.asemantics.rdfcoder.model;
020    
021    /**
022     * Defines an iterator over triples.
023     *
024     * @author Michele Mostarda (michele.mostarda@gmail.com)
025     */
026    public interface TripleIterator {
027    
028        /**
029         * moves to the next statement.
030         * @return <code>true</code> if the list of statements
031         * has not been reahed, <code>false</code> otherwise.
032         */
033        boolean next();
034    
035        /**
036         * Returns the subject of the current statement.
037         *
038         * @return triple subejct string.
039         */
040        String getSubject();
041    
042        /**
043         * Returns the predicate of the current statement.
044         *
045         * @return triple predicate string.
046         */
047    
048        String getPredicate();
049    
050        /**
051         * Returns the object of the current statement.
052         *
053         * @return triple object string.
054         */
055        String getObject();
056    
057        /**
058         * Coses the statement.
059         */
060        void close();
061    }
