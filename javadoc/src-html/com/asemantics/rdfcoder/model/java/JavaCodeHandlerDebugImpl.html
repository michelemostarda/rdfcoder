021    import com.asemantics.rdfcoder.model.ErrorListener;
022    import com.asemantics.rdfcoder.model.Identifier;
023    import com.asemantics.rdfcoder.parser.javadoc.ClassJavadoc;
024    import com.asemantics.rdfcoder.parser.javadoc.ConstructorJavadoc;
025    import com.asemantics.rdfcoder.parser.javadoc.FieldJavadoc;

028    import org.apache.log4j.Logger;
029    
030    /**
031     * A <code>CodeHandler</code> implementation used to perform debugging.
032     */
033    public class JavaCodeHandlerDebugImpl implements JavaCodeHandler {
034    
035        private static final Logger logger = Logger.getLogger(JavaCodeHandlerDebugImpl.class);
036    
037        public void startParsing(String libraryName, String location) {
038            if(logger.isDebugEnabled()) {
039                logger.debug("Start parsing process of library '" + libraryName + "' at location '" + location + "'");
040            }
041        }
042    
043        public void endParsing() {
044            if(logger.isDebugEnabled()) {
045                logger.debug("End parsing process");
046            }
047        }
048    
049        public void startCompilationUnit(String identifier) {
050            if(logger.isDebugEnabled()) {
051                logger.debug("Start compilation unit: " + identifier);
052            }
053        }
054    
055        public void endCompilationUnit() {
056            if(logger.isDebugEnabled()) {
057                logger.debug("End compilation unit");
058            }
059        }
060    
061        public void startPackage(Identifier pathToPackage) {
062            if(logger.isDebugEnabled()) {
063                logger.debug("Start package: " + pathToPackage);
064            }
065        }
066    
067        public void endPackage() {
068            if(logger.isDebugEnabled()) {
069                logger.debug("End package");
070            }
071        }
072    
073        public void startInterface(
074                Identifier pathToInterface,
075                Identifier[] extendedInterfaces
076        ) {
077            if(logger.isDebugEnabled()) {
078                logger.debug("Start interface " + pathToInterface + " extending < ");
079                for(Identifier extendedInterface : extendedInterfaces) {
080                    logger.debug(" " + extendedInterface + " ");
081                }
082                logger.debug(">");
083            }
084        }
085    
086        public void endInterface() {
087            if(logger.isDebugEnabled()) {
088                logger.debug("End interface");
089            }
090        }
091    
092        public void startClass(
093                JavaCodeModel.JModifier[] modifiers,
094                JavaCodeModel.JVisibility visibility,
095                Identifier pathToClass,
096                Identifier extendedClass,
097                Identifier[] implementedInterfaces
098        ) {
099            if(logger.isDebugEnabled()) {
100                logger.debug(
101                        "Start class " + pathToClass +
102                                " visibility: " + visibility +
103                                " modifiers: " + JavaCodeModel.JModifier.toByte(modifiers)
104                );
105                logger.debug(" extending class " + extendedClass);
106                logger.debug(" implemented interfaces < ");
107                for (Identifier implementedInterface : implementedInterfaces) {
108                    logger.debug(" " + implementedInterface + " ");
109                }
110                logger.debug(">");
111            }
112        }
113    
114        public void endClass() {
115            if(logger.isDebugEnabled()) {
116                logger.debug("End class");
117            }
118        }
119    
120        public void startEnumeration(
121                JavaCodeModel.JModifier[] modifiers,
122                JavaCodeModel.JVisibility visibility,
123                Identifier pathToEnumeration,
124                String[] elements
125        ) {
126            if(logger.isDebugEnabled()) {
127                logger.debug(
128                        "start enumeration " + pathToEnumeration +
129                                " visibility " + visibility  +
130                                " modifiers "  + JavaCodeModel.JModifier.toByte(modifiers)
131                );
132                logger.debug("{");
133                for(String element : elements) {
134                    logger.debug(element);
135                }
136                logger.debug("}");
137            }
138        }
139    
140        public void endEnumeration() {
141            if(logger.isDebugEnabled()) {
142                logger.debug("end enumeration");
143            }
144        }
145    
146        public void attribute(
147                JavaCodeModel.JModifier[] modifiers,
148                JavaCodeModel.JVisibility visibility,
149                Identifier pathToAttribute,
150                JavaCodeModel.JType type,
151                String value
152        ) {
153            if(logger.isDebugEnabled()) {
154                logger.debug(
155                        "attribute " + pathToAttribute +
156                                " visibility " + visibility +
157                                " modifiers " + JavaCodeModel.JModifier.toByte(modifiers)
158                );
159                logger.debug(" of type " + type + " ");
160                logger.debug(" with value " + value);
161            }
162        }
163    
164        public void constructor(
165                JavaCodeModel.JModifier[] modifiers,
166                JavaCodeModel.JVisibility visibility,
167                int signatureHashCode,
168                String[] parameterNames,
169                JavaCodeModel.JType[] parameterTypes,
170                JavaCodeModel.ExceptionType[] exceptions
171        ) {
172            if(logger.isDebugEnabled()) {
173                logger.debug(
174                        "constructor " + signatureHashCode +
175                                " visibility " + visibility +
176                                " modifiers "  + JavaCodeModel.JModifier.toByte(modifiers)
177                );
178                logger.debug("{");
179                printParameters(parameterNames, parameterTypes);
180                logger.debug("}");
181                printExceptions(exceptions);
182            }
183        }
184    
185        public void method(
186                JavaCodeModel.JModifier[] modifiers,
187                JavaCodeModel.JVisibility visibility,
188                Identifier pathToMethod,
189                int signatureHashCode,
190                String[] parameterNames,
191                JavaCodeModel.JType[] parameterTypes,
192                JavaCodeModel.JType returnType,
193                JavaCodeModel.ExceptionType[] exceptions
194        ) {
195            if(logger.isDebugEnabled()) {
196                logger.debug(
197                        "method " + pathToMethod + " signature " + signatureHashCode +
198                                " visibility " + visibility +
199                                " modifiers " + JavaCodeModel.JModifier.toByte(modifiers)
200                );
201                logger.debug(" return type " + returnType);
202                logger.debug("{");
203                printParameters(parameterNames, parameterTypes);
204                logger.debug("}");
205                printExceptions(exceptions);
206            }
207        }
208    
209        public void parseError(String location, String description) {
210            if(logger.isDebugEnabled()) {
211                logger.debug("parseError: location: " + location + " description: " + description);
212            }
213        }
214    
215        public void unresolvedTypes(String[] types) {
216            if(logger.isDebugEnabled()) {
217                logger.debug("unresolvedTypes {");
218                for(String type : types) {
219                    logger.debug(type);
220                }
221                logger.debug("}");
222            }
223        }
224    
225        public void preloadObjectsFromModel(ObjectsTable objectsTable) {
226            throw new UnsupportedOperationException();
227        }
228    
229        public void addErrorListener(ErrorListener errorListener) {
230            throw new UnsupportedOperationException();
231        }
232    
233        public void removeErrorListener(ErrorListener errorListener) {
234            throw new UnsupportedOperationException();
235        }
236    
237        public void classJavadoc(ClassJavadoc entry) {
238            if(logger.isDebugEnabled()) {
239                logger.debug("classJavadoc:" + entry);
240            }
241        }
242    
243        public void fieldJavadoc(FieldJavadoc entry) {
244            if(logger.isDebugEnabled()) {
245                logger.debug("fieldJavadoc:" + entry);
246            }
247        }
248    
249        public void constructorJavadoc(ConstructorJavadoc entry) {
250            if(logger.isDebugEnabled()) {
251                logger.debug("constructorJavadoc: " + entry);
252            }
253        }
254    
255        public void methodJavadoc(MethodJavadoc entry) {
256            if(logger.isDebugEnabled()) {
257                logger.debug("methodJavadoc: " + entry );
258            }
259        }
260    
261        public Identifier generateTempUniqueIdentifier() {
262            throw new UnsupportedOperationException();
263        }
264    
265        public int replaceIdentifierWithQualifiedType(Identifier identifier, Identifier qualifiedType) {
266            if(logger.isDebugEnabled()) {
267                logger.debug("replaceIdentifierWithQualifiedType " + identifier + " => " + qualifiedType);
268            }
269            return 0;
270        }
271    
272        private void printParameters(String[] names, JavaCodeModel.JType[] types) {
273            if(logger.isDebugEnabled()) {
274                for(int i = 0; i < names.length; i++) {
275                    logger.debug(names[i] + ": " + types[i]);
276                }
277            }
278        }
279    
280        private void printExceptions(JavaCodeModel.ExceptionType[] exceptions) {
281            if(exceptions == null || exceptions.length == 0) {
282                return;
283            }
284            if(logger.isDebugEnabled()) {
285                logger.debug(" throws ");
286                logger.debug("<");
287                for (JavaCodeModel.ExceptionType exception : exceptions) {
288                    logger.debug(" " + exception + " ");
289                }
290                logger.debug(">");
291            }
292        }
