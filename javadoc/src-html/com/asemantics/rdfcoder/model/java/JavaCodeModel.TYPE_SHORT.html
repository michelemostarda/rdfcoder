079        /**
080         * Java primitive Type base class. 
081         */
082        protected static abstract class JPrimitiveType implements JType {
083    
084            private Identifier primitiveTypeIdentifier;
085            private String relatedInternal;
086    
087            public Identifier getIdentifier() {
088                if (primitiveTypeIdentifier == null || ! getInternalIdentifier().equals(relatedInternal) ) {
089                    relatedInternal = getInternalIdentifier();
090                    primitiveTypeIdentifier = IdentifierBuilder
091                            .create()
092                            .pushFragment( relatedInternal, PRIMITIVE_KEY)
093                            .setPrefix(CODER_URI)
094                            .build();
095                }
096                return primitiveTypeIdentifier;
097            }
098    
099            public JType cloneType() {
100                return this;
101            }
102    
103            public String toString() {
104                return getInternalIdentifier();
105            }
106            
107        }

193            public ObjectType(Identifier pto) {
194                pathToObject = pto;

197            public void setInternalIdentifier(Identifier id) {

217                if(rdfType.indexOf(CLASS_KEY) == -1) {
218                    throw new IllegalArgumentException("Expected object prefix.");
219                }

221            }

272        /**
273         * The array type.
274         */
275        public static class ArrayType extends StructuredType {
276    
277            private JType type;
278            private int   size;
279    
280            public ArrayType(JType t, int s) {
281                type = t;
282                size = s; 
283            }
284    
285            public JType getType() {
286                return type;
287            }
288    
289            public int getSize() {
290                return size;
291            }
292    
293            public void setInternalIdentifier(Identifier id) {
294                if(type instanceof ObjectType) {
295                    ( (ObjectType) type).setInternalIdentifier(id);
296                } else if( type instanceof ArrayType) {
297                    ( (ArrayType) type).setInternalIdentifier(id);
298                } else {
299                    throw new IllegalStateException();
300                }
301            }
302    
303            public String getInternalIdentifier() {
304                return type.getInternalIdentifier();
305            }
306    
307            public Identifier getIdentifier() {
308                return type.getIdentifier();
309            }
310    
311            public String toString() {
312                return String.format( "%s[%d]", getInternalIdentifier(), getSize() );
313            }
314    
315            private static String ARRAY_PREFIX = toURI(TYPE_ARRAY_STR);
316    
317            public static ArrayType rdfTypeToType(String rdfType) {
318                if(rdfType.indexOf(ARRAY_PREFIX) != 0) {
319                    return null;
320                }
321                int i = rdfType.lastIndexOf(":");
322                int arraySize = Integer.parseInt(rdfType.substring(i + 1));
323                String type = rdfType.substring(ARRAY_PREFIX.length() + 1, i);
324                boolean typeIsObj = type.charAt(0) == '<';
325                if(typeIsObj) {
326                    type = type.substring(1, type.length() -2);
327                }
328                return new ArrayType(
329                        typeIsObj
330                                ?
331                        new ObjectType(IdentifierReader.readIdentifier(type) )
332                                :
333                        javaTypeToJType(type), arraySize
334                );
335            }
336        }

347        /**
348         * Converts a string representation of a RDF rdfType to a <code>JType</code>.
349         * @param rdfType the <code>JType</code> expressed as an identifier.
350         * @return the returned <code>JType</code> object.
351         */
352        public static final JType rdfTypeToJType(String rdfType) {
353    
354            // Primitive types.
355            if(CHAR.getIdentifier().getIdentifier().equals(rdfType)) {
356                return CHAR;
357            } else if(BYTE.getIdentifier().getIdentifier().equals(rdfType)) {
358                return BYTE;
359            } else if(SHORT.getIdentifier().getIdentifier().equals(rdfType)) {
360                return SHORT;
361            } else if(INT.getIdentifier().getIdentifier().equals(rdfType)) {
362                return INT;
363            } else if(LONG.getIdentifier().getIdentifier().equals(rdfType)) {
364                return LONG;
365            } else if(FLOAT.getIdentifier().getIdentifier().equals(rdfType)) {
366                return FLOAT;
367            } else if(DOUBLE.getIdentifier().getIdentifier().equals(rdfType)) {
368                return DOUBLE;
369            }  else if(BOOL.getIdentifier().getIdentifier().equals(rdfType)) {
370                return BOOL;
371            } else if(VOID.getIdentifier().getIdentifier().equals(rdfType)) {
372                return VOID;
373            }
374    
375            /* Complex types. */
376    
377            Object result;
378    
379            // Object type.
380            result = ObjectType.rdfTypeToType(rdfType);
381            if(result != null) {
382                return (ObjectType) result;
383            }
384    
385            // Interface type.
386            result = InterfaceType.rdfTypeToType(rdfType);
387            if(result != null) {
388                return (InterfaceType) result;
389            }
390    
391            // Array type.
392            result = ArrayType.rdfTypeToType(rdfType);
393            if(result != null) {
394                return (ArrayType) result;
395            }
396    
397            throw new IllegalArgumentException("Cannot convert RDF rdfType to a valid JType: '" + rdfType + "'.");
398        }

400        /**
401         * Converts a Java type name to a <code>JType</code>.
402         * @param type the Java type.
403         * @return the returned <code>JType</code> object.
404         */
405        public static final JType javaTypeToJType(String type) {
406            if(CHAR.getInternalIdentifier().equals(type)) {
407                return CHAR;
408            } else if(BYTE.getInternalIdentifier().equals(type)) {
409                return BYTE;
410            } else if(SHORT.getInternalIdentifier().equals(type)) {
411                return SHORT;
412            } else if(INT.getInternalIdentifier().equals(type)) {
413                return INT;
414            } else if(LONG.getInternalIdentifier().equals(type)) {
415                return LONG;
416            } else if(FLOAT.getInternalIdentifier().equals(type)) {
417                return FLOAT;
418            } else if(DOUBLE.getInternalIdentifier().equals(type)) {
419                return DOUBLE;
420            }  else if(BOOL.getInternalIdentifier().equals(type)) {
421                return BOOL;
422            } else if(VOID.getInternalIdentifier().equals(type)) {
423                return VOID;
424            } else {
425                throw new IllegalArgumentException("Cannot convert Java type to a valid JType: '" + type + "'.");
426            }
427        }

431         */
432        public enum JVisibility {
433            PUBLIC {
434                protected String getInternalIdentifier() {
435                    return "public";
436                }
437            },
438            PROTECTED {
439                protected String getInternalIdentifier() {
440                    return "protected";
441                }
442            },
443            DEFAULT {
444                protected String getInternalIdentifier() {
445                    return "default";
446                }
447            },
448            PRIVATE{
449                protected String getInternalIdentifier() {
450                    return "private";
451                }
452            };
453    
454            protected abstract String getInternalIdentifier();
455    
456            public String getIdentifier() {
457                return getInternalIdentifier();
458            }
459    
460            /**
461             * Converts a JVisibility string to an enumerated value.
462             *
463             * @param s
464             * @return the converted type.
465             */
466            public static JVisibility toJVisibility(String s) {
467                for(JVisibility v : JVisibility.values() ) {
468                    if(v.getIdentifier().equals(s)) {
469                        return v;
470                    }
471                }
472                throw new IllegalArgumentException();
473            }
474        }

476        /**
477         * Defines the Java modifiers.
478         */
479        public enum JModifier {
480    
481            /**
482             * Entity is declared abstract.
483             */
484            ABSTRACT {
485                private static final byte FA = 0x1;
486                public byte value() {
487                    return FA;
488                }
489                public boolean isValue(byte b) {
490                    return (b & FA) == FA;
491                }
492            },
493    
494            /**
495             * Entity is declared final.
496             */
497            FINAL {
498                private static final byte FB = 0x2;
499                public byte value() {
500                    return FB;
501                }
502                public boolean isValue(byte b) {
503                    return (b & FB) == FB; 
504                }
505            },
506    
507            /**
508             * Entity is declared static.
509             */
510            STATIC {
511                private static final byte FS = 0x4;
512                public byte value() {
513                    return FS;
514                }
515                public boolean isValue(byte b) {
516                    return (b & FS) == FS;
517                }
518            },
519    
520            /**
521             * Entity is declared volatile.
522             */
523            VOLATILE {
524                private static final byte FV = 0x8;
525                public byte value() {
526                    return FV;
527                }
528                public boolean isValue(byte b) {
529                    return (b & FV) == FV;
530                }
531            },
532    
533            /**
534             * Entity is declared native.
535             */
536            NATIVE {
537                private static final byte FN = 0x10;
538                public byte value() {
539                    return FN;
540                }
541                public boolean isValue(byte b) {
542                    return (b & FN) == FN;
543                }
544            },
545    
546            /**
547             * Entity is declared transient.
548             */
549            TRANSIENT {
550                private static final byte TR = 0x20;
551                public byte value() {
552                    return TR;
553                }
554                public boolean isValue(byte b) {
555                    return (b & TR) == TR;
556                }
557            },
558    
559            /**
560             * Entity is declared synchronized.
561             */
562            SYNCHRONIZED {
563                private static final byte SY = 0x40;
564                public byte value() {
565                    return SY;
566                }
567                public boolean isValue(byte b) {
568                    return (b & SY) == SY;
569                }
570            };
571    
572            public abstract byte    value();
573            public abstract boolean isValue(byte b);
574    
575            public static JModifier[] getModifiers(byte b) {
576                List<JModifier> modifiers = new ArrayList<JModifier>(JModifier.values().length);
577                for( JModifier v : JModifier.values()) {
578                    if(v.isValue(b)) {
579                        modifiers.add(v);
580                    }
581                }
582                return modifiers.toArray(new JModifier[modifiers.size()]);
583            }
584    
585            public static JModifier[] toModifiers(String m) {
586                byte b = Byte.parseByte(m);
587                return getModifiers(b);
588            }
589    
590            public static Byte toByte(JModifier[] modifiers) {
591                byte result = 0;
592                for(JModifier m : modifiers) {
593                    result |= m.value();
594                }
595                return result;
596            }
597        }
