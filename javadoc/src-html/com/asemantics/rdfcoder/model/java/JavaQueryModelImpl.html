052        public String[] getLibraries() {
053            List<String> result = new ArrayList<String>();
054            TripleIterator ti = codeModel.searchTriples(
055                    JavaCodeModel.ASSET,
056                    CodeModelBase.CONTAINS_LIBRARY,
057                    CodeModel.ALL_MATCH
058    
059            );
060            try {
061                while(ti.next()) {
062                    result.add(ti.getObject());
063                }
064            } finally {
065                ti.close();
066            }
067            return result.toArray( new String[result.size()] );
068        }

185        public JPackage[] getAllPackages() {
186            TripleIterator t1 = codeModel.searchTriples(
187                    JavaCodeModel.ALL_MATCH,
188                    JavaCodeModel.SUBCLASSOF,
189                    JavaCodeModel.JPACKAGE
190            );
191            List<JPackage> packages = new ArrayList<JPackage>();
192            String subject;
193            try {
194                while(t1.next()) {
195                    subject = t1.getSubject();
196                    try {
197                        packages.add( JavaCoderFactory.createJPackage(this, IdentifierReader.readIdentifier( subject) ));
198                    } catch (QueryModelException cme) {
199                        throw new RuntimeException("Error while retrieving packages.", cme);
200                    }
201                }
202            } finally {
203                t1.close();
204            }
205            return packages.toArray( new JPackage[packages.size()] );
206        }

427        public JSignature[] getSignatures(Identifier pathToMethod) throws QueryModelException {
428            TripleIterator t1 = codeModel.searchTriples(
429                    pathToMethod.getIdentifier(),
430                    JavaCodeModel.CONTAINS_SIGNATURE,
431                    JavaCodeModel.ALL_MATCH
432            );
433            TripleIterator t2;
434            TripleIterator t3;
435            List<JSignature> signatures = new ArrayList<JSignature>();
436            List<JavaCodeModel.JType> parametersIntoSignature = new ArrayList<JavaCodeModel.JType>();
437            try {
438                while(t1.next()) { //Signatures
439                    t2 = codeModel.searchTriples(t1.getObject(), JavaCodeModel.PARAMETER_TYPE, JavaCodeModel.ALL_MATCH);
440                    parametersIntoSignature.clear();
441                    try {
442                        while(t2.next()) { //Types.
443                            parametersIntoSignature.add( JavaCodeModel.rdfTypeToJType(t2.getSubject()) );
444                        } 
445                    } finally {
446                        t2.close();
447                    }
448                    t3 = codeModel.searchTriples(t1.getObject(), JavaCodeModel.RETURN_TYPE, JavaCodeModel.ALL_MATCH);
449                    try {
450                        if( ! t3.next() ) {
451                            throw new QueryModelException(
452                                    "Cannot find the return type for the method '" + pathToMethod + "'."
453                            );
454                        }
455                        signatures.add(
456                            JavaCoderFactory.createJSignature(
457                                    this,
458                                    IdentifierReader.readIdentifier(t1.getObject()),
459                                    ( parametersIntoSignature.toArray(new JavaCodeModel.JType[parametersIntoSignature.size()]) ),
460                                    JavaCodeModel.rdfTypeToJType( t3.getObject() )
461                                )
462                        );
463                    } finally {
464                        t3.close();
465                    }
466                }
467            } finally {
468                t1.close();
469            }
470            return signatures.toArray( new JSignature[signatures.size()] );
471        }

517                    result = t1.getObject();
