052        public String[] getLibraries() {
053            List<String> result = new ArrayList<String>();
054            TripleIterator ti = codeModel.searchTriples(
055                    JavaCodeModel.ASSET,
056                    CodeModelBase.CONTAINS_LIBRARY,
057                    CodeModel.ALL_MATCH
058    
059            );
060            try {
061                while(ti.next()) {
062                    result.add(ti.getObject());
063                }
064            } finally {
065                ti.close();
066            }
067            return result.toArray( new String[result.size()] );
068        }

185        public JPackage[] getAllPackages() {
186            TripleIterator t1 = codeModel.searchTriples(
187                    JavaCodeModel.ALL_MATCH,
188                    JavaCodeModel.SUBCLASSOF,
189                    JavaCodeModel.JPACKAGE
190            );
191            List<JPackage> packages = new ArrayList<JPackage>();
192            String subject;
193            try {
194                while(t1.next()) {
195                    subject = t1.getSubject();
196                    try {
197                        packages.add( JavaCoderFactory.createJPackage(this, IdentifierReader.readIdentifier( subject) ));
198                    } catch (QueryModelException cme) {
199                        throw new RuntimeException("Error while retrieving packages.", cme);
200                    }
201                }
202            } finally {
203                t1.close();
204            }
205            return packages.toArray( new JPackage[packages.size()] );
206        }

452        public JSignature[] getSignatures(Identifier pathToMethod) throws QueryModelException {
453            TripleIterator t1 = codeModel.searchTriples(
454                    pathToMethod.getIdentifier(),
455                    JavaCodeModel.CONTAINS_SIGNATURE,
456                    JavaCodeModel.ALL_MATCH
457            );
458            TripleIterator t2;
459            TripleIterator t3;
460            List<JSignature> signatures = new ArrayList<JSignature>();
461            List<JavaCodeModel.JType> parametersIntoSignature = new ArrayList<JavaCodeModel.JType>();
462            try {
463                while(t1.next()) { //Signatures
464                    t2 = codeModel.searchTriples(t1.getObject(), JavaCodeModel.PARAMETER_TYPE, JavaCodeModel.ALL_MATCH);
465                    parametersIntoSignature.clear();
466                    try {
467                        while(t2.next()) { //Types.
468                            parametersIntoSignature.add( JavaCodeModel.rdfTypeToJType(t2.getSubject()) );
469                        } 
470                    } finally {
471                        t2.close();
472                    }
473                    t3 = codeModel.searchTriples(t1.getObject(), JavaCodeModel.RETURN_TYPE, JavaCodeModel.ALL_MATCH);
474                    try {
475                        if( ! t3.next() ) {
476                            throw new QueryModelException(
477                                    "Cannot find the return type for the method '" + pathToMethod + "'."
478                            );
479                        }
480                        signatures.add(
481                            JavaCoderFactory.createJSignature(
482                                    this,
483                                    IdentifierReader.readIdentifier(t1.getObject()),
484                                    ( parametersIntoSignature.toArray(new JavaCodeModel.JType[parametersIntoSignature.size()]) ),
485                                    JavaCodeModel.rdfTypeToJType( t3.getObject() )
486                                )
487                        );
488                    } finally {
489                        t3.close();
490                    }
491                }
492            } finally {
493                t1.close();
494            }
495            return signatures.toArray( new JSignature[signatures.size()] );
496        }

498        public JavaCodeModel.JType[] getParameters(Identifier pathToSignature) throws QueryModelException {
499            TripleIterator t1 = codeModel.searchTriples(
500                pathToSignature.getIdentifier(),
501                JavaCodeModel.CONTAINS_PARAMETER,
502                JavaCodeModel.ALL_MATCH
503            );
504    
505            // Retrieve parameter names.
506            List<String> parameterNames = new ArrayList<String>();
507            try {
508                while(t1.next()) {
509                    parameterNames.add(t1.getObject());
510                }
511            } finally{
512                t1.close();
513            }
514    
515            // Retrieve parameter types.
516            List<JavaCodeModel.JType> result = new ArrayList<JavaCodeModel.JType>();
517            for (String parameterName : parameterNames) {
518                TripleIterator t2 = codeModel.searchTriples(
519                        parameterName,
520                        JavaCodeModel.PARAMETER_TYPE,
521                        JavaCodeModel.ALL_MATCH
522                );
523                try {
524                    t2.next(); // Expected to be found.
525                    result.add(JavaCodeModel.rdfTypeToJType( t2.getObject()) );
526                } finally {
527                    t2.close();
528                }
529            }
530    
531            return result.toArray( new JavaCodeModel.JType[result.size()] );
532        }
