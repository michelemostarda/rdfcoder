148    protected void checkLibraryName(String name) {
149        TripleIterator ti = model.searchTriples(JavaCodeModel.ASSET, JavaCodeModel.CONTAINS_LIBRARY, CodeModel.ALL_MATCH);
150        final String targetLibraryName = CodeModelBase.prefixFullyQualifiedName(JavaCodeModel.ASSET_PREFIX, name);
151        try {
152            while(ti.next()) {
153                if( targetLibraryName.equals(ti.getObject()) ) {
154                    throw new CodeHandlerException("a library with name '" + name + "' already exists in the model asset.");
155                }
156            }
157        } finally {
158            ti.close();
159        }
160    }

162    public void startParsing(String libraryName, String libraryLocation) {
163        if(libraryName == null || libraryName.trim().length() == 0) {
164            throw new CodeHandlerException("invalid library name: '" + libraryName + "'");
165        }
166        if(libraryLocation == null || libraryLocation.trim().length() == 0) {
167            throw new CodeHandlerException("invalid library location: '" + libraryLocation + "'");
168        }
169
170        checkLibraryName(libraryName);
171
172        this.libraryName     = libraryName;
173        this.libraryLocation = libraryLocation;
174
175        if(parsingStarted) {
176            throw new CodeHandlerException("Parse process already started.");
177        } else {
178            parsingStarted = true;
179        }
180
181        clearPackagesStack();
182        containersStack.clear();
183    }
184
185    public void endParsing() {
186        if( !parsingStarted) {
187            throw new CodeHandlerException("Parse process not yet started.");
188        }
189        if(! containersStack.isEmpty() ) {
190            throw new CodeHandlerException("There are some classes opened but not closed: " + containersStack.peek());
191        }
192        if(! isPackagesStackEmpty() ) {
193            throw new CodeHandlerException("There are some packages opened but not closed.");
194        }
195        parsingStarted = false;
196
197        String prefixedlibraryName = CodeModelBase.prefixFullyQualifiedName(JavaCodeModel.ASSET_PREFIX, libraryName);
198        String formattedDate = formatLibraryDatetime( new Date() );        
199        model.addTriple(JavaCodeModel.ASSET, JavaCodeModel.CONTAINS_LIBRARY,  prefixedlibraryName);
200        model.addTripleLiteral(prefixedlibraryName, CodeModelBase.LIBRARY_LOCATION, libraryLocation);
201        model.addTripleLiteral(prefixedlibraryName, CodeModelBase.LIBRARY_DATETIME, formattedDate);
202    }

235    public void startInterface(Identifier pathToInterface, Identifier[] extendedInterfaces) {
236        if(pathToInterface == null) {
237            throw new IllegalArgumentException("Invalid pathToInterface");
238        }
239
240        final String pathToInterfaceIdentifier = pathToInterface.getIdentifier();
241
242        checkPackageDiscrepancy(pathToInterface);
243
244        model.addTriple(pathToInterfaceIdentifier, CodeModel.SUBCLASSOF, JavaCodeModel.JINTERFACE);
245        Identifier parentClass = peekContainer();
246        model.addTriple(parentClass.getIdentifier(), JavaCodeModel.CONTAINS_INTERFACE, pathToInterfaceIdentifier);
247        if(extendedInterfaces != null) {
248            for(int i = 0; i < extendedInterfaces.length; i++) {
249                model.addTriple(
250                        pathToInterfaceIdentifier,
251                        JavaCodeModel.EXTENDS_INT,
252                        extendedInterfaces[i].getIdentifier()
253                );
254            }
255        }
256        pushInterfaceOrClass(pathToInterface);
257    }
258
259    public void endInterface() {
260        if(containersStack.isEmpty()) {
261            throw new CodeHandlerException("No interface to end.");
262        }
263
264        popClassOrInterface();
265    }

267    public void startClass(
268            JavaCodeModel.JModifier[] modifiers,
269            JavaCodeModel.JVisibility visibility,
270            Identifier pathToClass,
271            Identifier extededClass,           // Can be null.
272            Identifier[] implementedInterfaces // Can be null.
273    ) {
274        if(modifiers == null || visibility == null || pathToClass == null) {
275            throw new IllegalArgumentException();
276        }
277
278        final String pathToClassIdentifier = pathToClass.getIdentifier();
279
280        checkPackageDiscrepancy(pathToClass);
281
282        model.addTriple(pathToClassIdentifier, CodeModel.SUBCLASSOF, JavaCodeModel.JCLASS);
283        model.addTripleLiteral(
284                pathToClassIdentifier,
285                JavaCodeModel.HAS_MODIFIERS,
286                JavaCodeModel.JModifier.toByte(modifiers).toString()
287        );
288        model.addTripleLiteral(pathToClassIdentifier, JavaCodeModel.HAS_VISIBILITY, visibility.getIdentifier());
289        if(extededClass != null) {
290            model.addTriple(
291                    pathToClassIdentifier,
292                    JavaCodeModel.EXTENDS_CLASS,
293                    extededClass.getIdentifier()
294            );
295        }
296        if(implementedInterfaces != null) {
297            for(int i = 0; i < implementedInterfaces.length; i++) {
298                model.addTriple(
299                        pathToClassIdentifier,
300                        JavaCodeModel.IMPLEMENTS_INT,
301                        implementedInterfaces[i].getIdentifier()
302                );
303            }
304        }
305        Identifier parentClass = peekContainer();
306        model.addTriple(parentClass.getIdentifier(), JavaCodeModel.CONTAINS_CLASS, pathToClassIdentifier);
307        pushInterfaceOrClass(pathToClass);
308    }
309
310    public void endClass() {
311        if( containersStack.isEmpty() ) {
312            throw new CodeHandlerException("No class to end.");
313        } else {
314            popClassOrInterface();
315        }
316    }

319            JavaCodeModel.JModifier[] modifiers,
320            JavaCodeModel.JVisibility visibility,
321            Identifier pathToEnumeration,
322            String[] elements
323    ) {
324        if(modifiers == null || visibility == null || pathToEnumeration == null || elements == null) {
325            throw new IllegalArgumentException();
326        }
327
328        final String pathToEnumerationIdentifier = pathToEnumeration.getIdentifier();
329
330        checkPackageDiscrepancy(pathToEnumeration);
331
332        model.addTriple(pathToEnumerationIdentifier, CodeModel.SUBCLASSOF, JavaCodeModel.JENUMERATION);
333        model.addTripleLiteral(
334                pathToEnumerationIdentifier,
335                JavaCodeModel.HAS_MODIFIERS,
336                JavaCodeModel.JModifier.toByte(modifiers).toString()
337        );
338        model.addTripleLiteral(pathToEnumerationIdentifier, JavaCodeModel.HAS_VISIBILITY, visibility.getIdentifier());
339        for(int i = 0; i < elements.length; i++) {
340            model.addTriple(
341                pathToEnumerationIdentifier,
342                JavaCodeModel.CONTAINS_ELEMENT,
343                IdentifierBuilder.create().pushFragment( elements[i], JavaCodeModel.ELEMENT_KEY).build().getIdentifier()
344            );
345        }
346        Identifier parentClass = peekContainer();
347        model.addTriple(parentClass.getIdentifier(), JavaCodeModel.CONTAINS_ENUMERATION, pathToEnumerationIdentifier);
348        pushInterfaceOrClass(pathToEnumeration);
349    }

359    public void attribute(
360            JavaCodeModel.JModifier[] modifiers,
361            JavaCodeModel.JVisibility visibility,
362            Identifier pathToAttribute,
363            JavaCodeModel.JType type,
364            String value                        // Can be null.
365    ) {
366        if(modifiers == null || visibility == null || pathToAttribute == null || type == null) {
367            throw new IllegalArgumentException();
368        }
369        if(containersStack.isEmpty()) {
370            throw new CodeHandlerException("Must be inside a class to define an attribute.");
371        }
372
373        final String identifier = pathToAttribute.getIdentifier();
374        model.addTriple(identifier, SUBCLASSOF, JavaCodeModel.JATTRIBUTE);
375        model.addTripleLiteral(
376                identifier, JavaCodeModel.HAS_MODIFIERS, JavaCodeModel.JModifier.toByte(modifiers).toString()
377        );
378        model.addTripleLiteral(identifier, JavaCodeModel.HAS_VISIBILITY, visibility.getIdentifier());
379        model.addTripleLiteral(identifier, JavaCodeModel.ATTRIBUTE_TYPE,  type.getIdentifier().getIdentifier());
380        if(value != null) { // Default value defined.
381            model.addTripleLiteral(identifier, JavaCodeModel.ATTRIBUTE_VALUE, value);
382        }
383        Identifier parentClass = peekContainer();
384        model.addTriple(parentClass.getIdentifier(), JavaCodeModel.CONTAINS_ATTRIBUTE, identifier);
385    }

413    public void method(
414            JavaCodeModel.JModifier[] modifiers,
415            JavaCodeModel.JVisibility visibility,
416            Identifier pathToMethod,
417            int signatureHashCode,
418            String[] parameterNames,
419            JavaCodeModel.JType[] parameterTypes,
420            JavaCodeModel.JType returnType,
421            JavaCodeModel.ExceptionType[] exceptions
422    ) {
423        if(modifiers == null || visibility == null || pathToMethod == null) {
424            throw new IllegalArgumentException();
425        }
426        if(containersStack.isEmpty()) {
427            throw new CodeHandlerException("Must be inside a class to define a method.");
428        }
429        // Validating arguments.
430        int paramNamesSize = parameterNames == null ? 0 : parameterNames.length;
431        int paramTypesSize = parameterTypes == null ? 0 : parameterTypes.length;
432        if(paramNamesSize != paramTypesSize) {
433            throw new IllegalArgumentException(
434                "The number of parameter names differs from the number of parameters types: "
435                + paramNamesSize + " <> " + paramTypesSize
436            );
437        }
438        int exceptionsSize = exceptions == null ? 0 : exceptions.length;
439
440        final String identifier = pathToMethod.getIdentifier();
441
442        // Creating structure.
443        model.addTriple(identifier, SUBCLASSOF, JavaCodeModel.JMETHOD);
444        model.addTripleLiteral(
445                identifier,
446                JavaCodeModel.HAS_MODIFIERS,
447                JavaCodeModel.JModifier.toByte(modifiers).toString()
448        );
449        model.addTripleLiteral(identifier, JavaCodeModel.HAS_VISIBILITY, visibility.getIdentifier());
450//        String signature = JavaCodeModel.SIGNATURE_PREFIX + generateSignatureIdentifier(parameterTypes);
451        String signature = IdentifierBuilder
452                .create(pathToMethod)
453                .pushFragment( "_" + signatureHashCode, JavaCodeModel.SIGNATURE_KEY)
454                .build()
455                .getIdentifier();
456        model.addTriple(signature, SUBCLASSOF, JavaCodeModel.JSIGNATURE);
457        String qualifiedParameter; 
458        for(int i = 0; i < paramNamesSize; i++) {
459            qualifiedParameter = qualifyParameterName(pathToMethod, parameterNames[i]);
460            model.addTriple( qualifiedParameter, SUBCLASSOF, JavaCodeModel.JPARAMETER);
461            model.addTripleLiteral(
462                    qualifiedParameter,
463                    JavaCodeModel.PARAMETER_TYPE,
464                    parameterTypes[i].getIdentifier().getIdentifier()
465            );
466            model.addTriple( signature, JavaCodeModel.CONTAINS_PARAMETER, qualifiedParameter);
467        }
468        model.addTripleLiteral(signature, JavaCodeModel.RETURN_TYPE, returnType.getIdentifier().getIdentifier());
469        model.addTriple(identifier, JavaCodeModel.CONTAINS_SIGNATURE, signature);
470        for(int i = 0; i < exceptionsSize; i++) {
471            model.addTriple(identifier, JavaCodeModel.THROWS, exceptions[i].getIdentifier().getIdentifier());
472        }
473        Identifier parentContainer = peekContainer();
474        model.addTriple(parentContainer.getIdentifier(), JavaCodeModel.CONTAINS_METHOD, identifier);
475    }

485    public void preloadObjectsFromModel(ObjectsTable objectsTable) {
486        if(objectsTable == null) {
487            throw new CodeHandlerException("objectsTable cannot be null.");
488        }
489
490        // Preloading classes.
491        TripleIterator t1 = model.searchTriples(
492                JavaCodeModel.ALL_MATCH, JavaCodeModel.SUBCLASSOF, JavaCodeModel.JCLASS
493        );
494        try {
495        Identifier fullyQualifiedObject;
496            while(t1.next()) {
497                fullyQualifiedObject = IdentifierReader.readIdentifier( t1.getSubject() );
498                objectsTable.addObject( fullyQualifiedObject );
499                if(logger.isDebugEnabled()) {
500                    logger.debug("Added object: " + fullyQualifiedObject );
501                }
502            }
503        } finally {
504            t1.close();
505        }
506
507        // Preloading interfaces.
508        TripleIterator t2 = model.searchTriples(
509                JavaCodeModel.ALL_MATCH, JavaCodeModel.SUBCLASSOF, JavaCodeModel.JINTERFACE
510        );
511        try {
512        Identifier fullyQualifiedObject;
513            while(t2.next()) {
514                fullyQualifiedObject = IdentifierReader.readIdentifier( t2.getSubject() );
515                objectsTable.addObject( fullyQualifiedObject );
516                if(logger.isDebugEnabled()) {
517                    logger.debug("Added object: " + fullyQualifiedObject );
518                }
519            }
520        } finally {
521            t2.close();
522        }
523    }

608    /**
609     * Adds a package on the packages stack.
610     *
611     * @param cp current package identifier.
612     */
613    protected void pushPackage(final Identifier cp) {
614        packagesCount++;
615        currentPackage = cp;
616
617        String currentStr;
618        Identifier parent;
619        Identifier current = cp;
620        do {
621            currentStr = current.getIdentifier();
622            model.addTriple(currentStr, SUBCLASSOF, JavaCodeModel.JPACKAGE);
623            parent = current.getParent();
624            model.addTriple(parent.getIdentifier(), JavaCodeModel.CONTAINS_PACKAGE, currentStr);
625            current = parent;
626        } while (current.size() > 1);
627    }

629    /**
630     * Removes last package from the packges stack.
631     */
632    protected void popPackage() {
633        packagesCount--;
634        currentPackage = null;
635    }

645        }
646        return currentPackage;
647    }

691    private void internalConstructor(
692            Identifier pathToClass,
693            JavaCodeModel.JModifier[] modifiers,
694            JavaCodeModel.JVisibility visibility,
695            int signatureHashCode,
696            String[] parameterNames,
697            JavaCodeModel.JType[] parameterTypes,
698            JavaCodeModel.ExceptionType[] exceptions
699    ) {
700        if(modifiers == null || visibility == null) {
701            throw new IllegalArgumentException();
702        }
703
704        // Validating arguments.
705        int paramNamesSize = parameterNames == null ? 0 : parameterNames.length;
706        int paramTypesSize = parameterTypes == null ? 0 : parameterTypes.length;
707        if(paramNamesSize != paramTypesSize) {
708            throw new IllegalArgumentException(
709                "The number of parameter names differs from the number of parameters types: "
710                + paramNamesSize + " <> " + paramTypesSize
711            );
712        }
713        int exceptionsSize = exceptions == null ? 0 : exceptions.length;
714
715        String classIdentifier = pathToClass.getIdentifier();
716        String identifier = IdentifierBuilder
717                .create(pathToClass)
718                .pushFragment( "_" + signatureHashCode, JavaCodeModel.CONSTRUCTOR_KEY).build().getIdentifier();
719        model.addTriple(identifier, CodeModel.SUBCLASSOF, JavaCodeModel.JCONSTRUCTOR);
720        model.addTripleLiteral(
721                identifier,
722                JavaCodeModel.HAS_MODIFIERS,
723                JavaCodeModel.JModifier.toByte(modifiers).toString()
724        );
725        model.addTripleLiteral(identifier, JavaCodeModel.HAS_VISIBILITY, visibility.getIdentifier());
726        String qualifiedParameter;
727        for(int i = 0; i < paramNamesSize; i++) {
728            qualifiedParameter = qualifyParameterName( pathToClass, parameterNames[i]);
729            model.addTriple( qualifiedParameter, CodeModel.SUBCLASSOF, JavaCodeModel.JPARAMETER);
730            model.addTripleLiteral(
731                    qualifiedParameter,
732                    JavaCodeModel.PARAMETER_TYPE,
733                    parameterTypes[i].getIdentifier().getIdentifier()
734            );
735            model.addTriple( identifier, JavaCodeModel.CONTAINS_PARAMETER, qualifiedParameter);
736        }
737        for(int i = 0; i < exceptionsSize; i++) {
738            model.addTriple(identifier, JavaCodeModel.THROWS, exceptions[i].getIdentifier().getIdentifier());
739        }
740        model.addTriple(classIdentifier, JavaCodeModel.CONTAINS_CONSTRUCTOR, identifier);
741    }

759    private void checkPackageDiscrepancy(Identifier pathToElement) {
760        if( ! RDFCoder.checkPackageDiscrepancy() ) { return; }
761
762        Identifier elementContainer = pathToElement.getParent();
763        Identifier currentPackage = peekPackage();
764
765        if( ! currentPackage.equals(elementContainer) ) {
766             notifyPackageDiscrepancy(elementContainer.getIdentifier(), currentPackage.getIdentifier());
767        }
768    }

827    /**
828     * Adds the triples specific for a Javadoc entry.
829     *
830     * @param javadocEntry the input Javadoc entry to be triplified.
831     */
832    private void addJavadocTriples(JavadocEntry javadocEntry) {
833        final String identifier = javadocEntry.getIdentifier().getIdentifier();
834        String shortDescription = javadocEntry.getShortDescription();
835        if(shortDescription.length() > 0)
836            model.addTripleLiteral(
837                    identifier,
838                    JavaCodeModel.HAS_SHORT_COMMENT,
839                    shortDescription
840            );
841        String longDescription = javadocEntry.getLongDescription();
842        if(longDescription.length() > 0)
843            model.addTripleLiteral(
844                    identifier,
845                    JavaCodeModel.HAS_LONG_COMMENT,
846                    longDescription
847            );
848
849        for(String attributeName : javadocEntry.getAttributeNames()) {
850            final String attributeId = IdentifierBuilder.create(javadocEntry.getIdentifier())
851                    .pushFragment(attributeName, JavaCodeModel.JDOC_ATTRIBUTE_PREFIX)
852                    .build()
853                    .getIdentifier();
854            model.addTriple(identifier, JavaCodeModel.HAS_JDOC_ATTR, attributeId);
855            model.addTripleCollection(
856                    attributeId,
857                    JavaCodeModel.HAS_JDOC_ATTR_VALUE,
858                    javadocEntry.getAttributeValues(attributeName)
859            );
860        }
861
862        model.addTripleLiteral(identifier, JavaCodeModel.HAS_ROW   , Integer.toString(javadocEntry.getRow()));
863        model.addTripleLiteral(identifier, JavaCodeModel.HAS_COLUMN, Integer.toString(javadocEntry.getCol()));
864    }
