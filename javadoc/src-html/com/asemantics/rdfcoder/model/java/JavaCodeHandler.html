019    package com.asemantics.rdfcoder.model.java;
020    
021    import com.asemantics.rdfcoder.model.CodeHandler;
022    import com.asemantics.rdfcoder.model.ErrorListener;
023    import com.asemantics.rdfcoder.model.Identifier;
024    import com.asemantics.rdfcoder.parser.ObjectsTable;
025    
026    /**
027     * This interface defines the <i> Java Code Handler</i>.
028     *
029     * A handler is responsible of receiving code analysis
030     * events and transform it in a usable format.
031     *
032     * @author Michele Mostarda (michele.mostarda@gmail.com)
033     */
034    public interface JavaCodeHandler extends CodeHandler, JavadocHandler {
035    
036        /**
037         * Expected format of the library date-time.
038         */
039        public static final String LIBRARY_DATETIME_FORMAT = "yyyy_MM_dd__HH_mm_ss";
040    
041        /**
042         * The package separator character.
043         */
044        public static final String PACKAGE_SEPARATOR = ".";
045    
046        /**
047         * Notifies the begin of a package parsing.
048         * The package is expected to be FULLY QUALIFIED.
049         *
050         * @param pathToPackage the path expressed as list of fragments to the package.
051         */
052        public void startPackage(Identifier pathToPackage);
053    
054        /**
055         * Notifies the end of a package parsing.
056         */
057        public void endPackage();
058    
059        /**
060         * Notifies the begin of an interface definition.
061         *
062         * @param pathToInterface the path expressed as list of fragments to the interface.
063         * @param extendedInterfaces the list of extended interfaces.
064         */
065        public void startInterface(Identifier pathToInterface, Identifier[] extendedInterfaces);
066    
067        /**
068         * Notifies the end of an interface.
069         */
070        public void endInterface();
071    
072        /**
073         * Notifies the begin of a class parsing.
074         * The class is expected to be FULLY QUALIFIED.
075         *
076         * @param modifiers the class modifiers.
077         * @param visibility the class visibility level.
078         * @param pathToClass the path to the class.
079         * @param extendedClass the fully qualified pathToClass of the class extended by the starting class.
080         * @param implementedInterfaces the fully qualified names for the interfaces implemented by the starting class.
081         */
082        public void startClass(
083                JavaCodeModel.JModifier[] modifiers,
084                JavaCodeModel.JVisibility visibility,
085                Identifier pathToClass,
086                Identifier extendedClass,
087                Identifier[] implementedInterfaces
088        );
089    
090        /**
091         * Notifies the end of a class parsing.
092         */
093        public void endClass();
094    
095        /**
096         * Notifies the begin of an enumeration class parsing.
097         *
098         * The enumeration is expected to be FULLY QUALIFIED.
099         *
100         * @param modifiers the enumeration modifiers.
101         * @param visibility the visibility level of the enumeration.
102         * @param pathToEnumeration the fully qualified path to the enumeration.
103         * @param elements the list of the elements of the enumeration.
104         */
105        public void startEnumeration(
106                JavaCodeModel.JModifier[] modifiers,
107                JavaCodeModel.JVisibility visibility,
108                Identifier pathToEnumeration,
109                String[] elements
110        );
111    
112        /**
113         * Notifies the end of the enumeration class.
114         */
115        public void endEnumeration();
116    
117        /**
118         * Notifies the parsing of an attribute.
119         *
120         * The attribute is expected to be FULLY QUALIFIED.
121         *
122         * @param modifiers the attribute modifiers.
123         * @param visibility the attribute visibility level.
124         * @param pathToAttribute the absolute path to the attribute.
125         * @param type the type of the attribute.
126         * @param value the init value for the attribute. 
127         */
128        public void attribute(
129                JavaCodeModel.JModifier[] modifiers,
130                JavaCodeModel.JVisibility visibility,
131                Identifier pathToAttribute,
132                JavaCodeModel.JType type,
133                String value
134        );
135    
136        /**
137         * Notifies the parsing of a constructor.
138         *
139         * The length of paramterSize and parameterNames is the same,
140         * the association between these elements is done positionally.
141         *
142         * @param modifiers the constructor modifiers.
143         * @param visibility the visibility level for the constructor.
144         * @param signatureHashCode the signature hash code. 
145         * @param parameterNames the array of parameter names.
146         * @param parameterTypes the array of parameter types.
147         * @param exceptions exceptions raised by the constructor.
148         *
149         */
150        public void constructor(
151                JavaCodeModel.JModifier[] modifiers,
152                JavaCodeModel.JVisibility visibility,
153                int signatureHashCode,
154                String[] parameterNames,
155                JavaCodeModel.JType[] parameterTypes,
156                JavaCodeModel.ExceptionType[] exceptions
157        );
158    
159        /**
160         * Notifies the parsing of a method.
161         *
162         * The method is expected to be FULLY QUALIFIED.
163         *
164         * The length of parameterSize and parameterNames is the same,
165         * the association between these elements is done positionally.
166         *
167         * @param modifiers the method modifiers.
168         * @param visibility the method visibility level.
169         * @param pathToMethod the fully qualified path to the method.
170         * @param signatureHashCode the signature hash code.
171         * @param parameterNames the array of parameter names.
172         * @param parameterTypes the array of parameter types.
173         * @param returnType the return type of the method.
174         * @param exceptions exceptions raised by the method.
175         *
176         */
177        public void method(
178                JavaCodeModel.JModifier[] modifiers,
179                JavaCodeModel.JVisibility visibility,
180                Identifier pathToMethod,
181                int signatureHashCode,
182                String[] parameterNames,
183                JavaCodeModel.JType[] parameterTypes,
184                JavaCodeModel.JType returnType,
185                JavaCodeModel.ExceptionType[] exceptions
186        );
187    
188        /**
189         * Collects the unresolved types at the end of the parsing.
190         *
191         * @param unresolvedTypes
192         */
193        public void unresolvedTypes(String[] unresolvedTypes);
194    
195        /**
196         * Preloads the given Objects Table with the objects present in the underlying model.
197         *
198         * @param objectsTable
199         */
200        //TODO: LOW - Integrate preloading.
201        public void preloadObjectsFromModel(ObjectsTable objectsTable);
202    
203        /**
204         * Adds an error listener to the code handler.
205         * @param errorListener
206         */
207        public void addErrorListener(ErrorListener errorListener);
208    
209        /**
210         * Removes an error listener from the code handler.
211         * @param errorListener
212         */
213        public void removeErrorListener(ErrorListener errorListener);
214    
215    }
