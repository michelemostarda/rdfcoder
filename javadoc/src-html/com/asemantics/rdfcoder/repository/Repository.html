040        /**
041         * Kinds of resources.
042         */
043        public enum ResourceType {
044            BINARY {
045    
046                String getExtension() {
047                    return "bin";
048                }
049    
050            },
051    
052            XML {
053    
054                String getExtension() {
055                    return "xml";
056                }
057            };
058    
059            /**
060             * Returns the type of the file name on the basis of the extension,
061             * <code>null</code> if no extension matches.
062             *
063             * @param fileName
064             * @return
065             */
066            static ResourceType getType(String fileName) {
067                for(ResourceType rt : ResourceType.values()) {
068                    if(fileName.indexOf(rt.getExtension()) != -1 ) {
069                        return rt;
070                    }
071                }
072                return null;
073            }
074    
075            /**
076             * Returns the name of the resource removing the extension.
077             *
078             * @param file
079             * @return
080             */
081            static String getResourceName(String file) {
082                int separator = file.indexOf(".");
083                return file.substring(0, separator);
084            }
085    
086            /**
087             * Returns the extension associated to the type.
088             * @return
089             */
090            abstract String getExtension();
091    
092            /**
093             * Returns the file name from the resource name.
094             *
095             * @param resourceName
096             * @return
097             */
098            String getFileName( String resourceName ) {
099                return resourceName + "."  + getExtension();
100            }

104        /**
105         * Resource element.
106         */
107        public class Resource {
108    
109            /**
110             * parent directory containing this resource.
111             */
112            private File root;
113    
114            /**
115             * The resource name.
116             */
117            private String name;
118    
119            /**
120             * The resource type.
121             */
122            private ResourceType type;
123    
124            /**
125             * List of readers of this resource.
126             */
127            private List<OutputStream> writers;
128    
129            /**
130             * List of writers of this resource.
131             */
132            private List<InputStream> readers;
133    
134    
135    
136            /**
137             * Constructor.
138             *
139             * @param n
140             * @param rt
141             */
142            Resource(File r, String n, ResourceType rt) {
143                root = r;
144                name = n;
145                type = rt;
146    
147                writers = new ArrayList<OutputStream>();
148                readers = new ArrayList<InputStream>();
149            }
150    
151            /**
152             * Returns the resource name.
153             *
154             * @return the resource name.
155             */
156            public String getName() {
157                return name;
158            }
159    
160            /**
161             * Returns the resource type.
162             * 
163             * @return the type of the resource.
164             */
165            public ResourceType getType() {
166                return type;
167            }
168    
169            /**
170             * Returns the resource location.
171             *
172             * @return the resource location.
173             */
174            public File getLocation() {
175                return new File(root, name + "." + type.getExtension());
176            }
177    
178            /**
179             * Opens an input stream on the resource.
180             *
181             * @return the input stream to read the resource.
182             * @throws FileNotFoundException
183             */
184            public InputStream getInputStream() throws FileNotFoundException, RepositoryException {
185                synchronized (lock) {
186                    if(writers.size() > 0) {
187                        throw new RepositoryException("Cannot open resource in read mode: there is at least a reader on it");
188                    }
189    
190                    // Check there is a single writer.
191                    if(readers.size() > 0) {
192                        throw new RepositoryException("Resource '" + name + "' already accessed in write mode.");
193                    }
194    
195                    InputStream is = new FileInputStream( getLocation() ) {
196        
197                        /**
198                         * Releases the lock file.
199                         *
200                         * @throws IOException
201                         */
202                        public void close() throws IOException {
203                            super.close();
204                            readers.remove(this);
205                        }
206    
207                    };
208                    readers.add(is);
209                    return is;
210                }
211            }
212    
213            /**
214             * Opens an output stream on the resource.
215             *
216             * @param append append flag
217             * @return the output stream to write the resource.
218             * @throws FileNotFoundException if the resource file cannot be found or opened.
219             */
220            public OutputStream getOutputStream(boolean append) throws FileNotFoundException, RepositoryException {
221                synchronized (lock) {
222                    if(writers.size() > 0) {
223                        throw new RepositoryException("Cannot open resource in write mode: there is anoter writer on it");
224                    }
225                    if(readers.size() > 0) {
226                        throw new RepositoryException("Cannot open resource in write mode: there is al least a reader on it");
227                    }
228    
229                    // Creates a lock file for write mode.
230                    try {
231                        createLockFile(name);
232                    } catch (IOException e) {
233                        throw new RepositoryException(e);
234                    }
235    
236                    OutputStream os = new FileOutputStream( getLocation(), append ) {
237    
238                        public void close() throws IOException {
239                            super.close();
240                            writers.remove(this);
241                            removeLockFile(name);
242                            
243                        }
244    
245                    };
246                    writers.add(os);
247                    return os;
248                }
249            }
250    
251            /**
252             *  Opens an output stream on the resource.
253             *
254             * @return the resource output stream.
255             * @throws FileNotFoundException
256             * @throws RepositoryException
257             */
258            public OutputStream getOutputStream() throws FileNotFoundException, RepositoryException {
259                return getOutputStream(false);
260            }
261    
262            /**
263             * Deletes the current resource closing first all input and output streams.
264             */
265            public void delete() {
266                for(InputStream is : readers) {
267                    try {
268                        is.close();
269                    } catch (Exception e) {}
270                }
271                for(OutputStream os : writers) {
272                    try {
273                        os.close();
274                    } catch (Exception e) {}
275                }
276                getLocation().delete();
277    
278            }
279        }

281        /**
282         * Filters all lock files inside a dir.
283         */
284        public static class LockFileFilter implements FilenameFilter {
285            public boolean accept(File dir, String name) {
286                File file = new File(dir, name);
287                if( file.isDirectory() ) { return false; }
288                String postfix = LOCK_FILE_EXT;
289                int lastIndexOf = name.lastIndexOf(postfix);
290                return lastIndexOf > 0 && lastIndexOf == name.length() - postfix.length();
291            }
292        }

356        /**
357         * Loads the resources defined in the <i>target</i>
358         * dir.
359         *
360         * @return the map of loaded resources.
361         */
362        protected Map<String,Resource> loadResources(File target) {
363            if( ! target.exists() ) {
364                throw new IllegalArgumentException("target dir doesn't exist: '" + target.getAbsolutePath() + "'");
365            }
366            if( ! target.isDirectory() ) {
367                throw new IllegalArgumentException("target dir must be a directory");
368            }
369    
370            File[] files = target.listFiles();
371            Map<String,Resource> result = new HashMap<String,Resource>(files.length);
372            ResourceType rt;
373            String fileName;
374            String resourceName;
375            for(File file : files) {
376                if( file.isFile() ) {
377                    fileName = file.getName();
378                    rt = ResourceType.getType(fileName);
379                    if( rt != null ) {
380                        resourceName = ResourceType.getResourceName(fileName);
381                        result.put( resourceName, new Resource(target ,resourceName , rt) );
382                    }
383                }
384            }
385            return result;
386        }

388        /**
389         * Checks the repository folder.
390         *
391         * @param location
392         * @return <code>true</code> if the repo folder has been created,
393         * <code>false</code> otherwise.
394         */
395        protected static boolean checkRepository(File location) throws RepositoryException {
396            if( ! location.exists() ) {
397                if( ! location.mkdir() ) {
398                    throw new RepositoryException(
399                            "Error while creating repository dir at location '" + location.getAbsolutePath() + "'."
400                    );
401                }
402                return true;
403            } else if( ! location.isDirectory() ) {
404                throw new RepositoryException(
405                        "A file with the same name of the repository already exists at location '"+ location.getAbsolutePath() + "'."
406                );
407            }
408            return false;
409        }
410    
411        /**
412         * Singleton restriction constructor.
413         */
414        private Repository(File rl) throws RepositoryException {
415            repositoryLocation = rl;
416            if( ! checkRepository(rl) ) {
417                // Performs a cleanup of repository.
418                removeLockedResources();
419                // Loads existing resources.
420                resources = loadResources(repositoryLocation);
421            } else {
422                resources = new HashMap<String,Resource>();
423            }
424        }

445        /**
446         * Returns <code>true</code> if the repository contains the given resource name,
447         * <code>false</code> otherwise.
448         *
449         * @param resourceName name of resource to find.
450         * @return <code>true</code> if found.
451         */
452        public boolean containsResource(String resourceName) {
453            return resources.containsKey(resourceName);
454        }
455    
456        /**
457         * Returns an existing resource.
458         *
459         * @param resourceName
460         * @return the resource found.
461         * @throws RepositoryException if the resource doens't exist.
462         */
463        public Resource getResource(String resourceName) throws RepositoryException {
464            Resource resource = resources.get(resourceName);
465            if(resource == null) {
466                throw new RepositoryException("Required resource '" + resourceName + "' doesn't exist.");
467            }
468            return resource;
469        }

471        /**
472         * Removes the specified resource.
473         *
474         * @param resourceName
475         * @throws RepositoryException if the resource cannot be found.
476         */
477        public void removeResource(String resourceName) throws RepositoryException {
478            Resource resource = getResource(resourceName);
479            if( ! resource.getLocation().delete() ) {
480                throw new RepositoryException("Cannot delete resource: '" + resourceName + "'");
481            }
482            try {
483                removeLockFile(resourceName);
484            } catch(Exception e) {
485                e.printStackTrace();
486            }
487            resources.remove(resourceName);
488        }

490        /**
491         * Check if the resource name is valid.
492         *
493         * @param rn
494         */
495        static void checkResourceName(String rn) throws RepositoryException {
496            if(rn == null || rn.trim().length() == 0) {
497                throw new RepositoryException("Invalid resource name: '"+ rn + "'");
498            }
499        }
500    
501        /**
502         * Creates a new resource into the repository.
503         *
504         * @param resourceName
505         * @param type
506         * @return the created resource.
507         * @throws RepositoryException if a resurce with the same name exists or the file cannot be created.
508         */
509        public Resource createResource(String resourceName, ResourceType type) throws RepositoryException {
510            checkResourceName(resourceName);
511    
512            if( containsResource(resourceName) ) {
513                throw new RepositoryException(
514                        "Cannot create resource '" + resourceName + "', a resouce with the same name already exists."
515                );
516            }
517    
518            File file    = new File(repositoryLocation, type.getFileName(resourceName));
519            try {
520                file.createNewFile();
521            } catch (IOException e) {
522                throw new RepositoryException("Cannot create file: '" + file.getAbsolutePath() + "'");
523            }
524    
525            Resource resource = new Resource(repositoryLocation, resourceName, type);
526            resources.put(resourceName, resource);
527    
528            return resource;
529        }

531        /**
532         * Returns <code>true</code> if a resource is locked in writing mode,
533         * <code>false</code> otherwise.
534         *
535         * @param resourceName
536         * @return <code>true</code> if the resource is locked.
537         * @throws RepositoryException is resourceName doesn't exist.
538         */
539        public boolean isLocked(String resourceName) throws RepositoryException {
540            if( resources != null && ! containsResource(resourceName) ) {
541                throw new RepositoryException("Repository doesn't contain resource name '" + resourceName + "'");
542            }
543    
544            File resourceFileLock = new File(repositoryLocation, getFileLockName(resourceName));
545            return resourceFileLock.exists();
546        }
547    
548        /**
549         * Returns the name of the open file lock of a resource.
550         *
551         * @param resourceName
552         * @return
553         */
554        private static String getFileLockName(String resourceName) {
555            return "." + resourceName + LOCK_FILE_EXT;
556        }
557    
558        /**
559         * Returns the resource name associated to a given lock file name.
560         *
561         * @param lockFileName
562         * @return
563         */
564        private static String getResourceNameFromLockFile(String lockFileName) {
565            return lockFileName
566                    .substring(1) // "."
567                    .substring(0, lockFileName.length() - 1 - LOCK_FILE_EXT.length() );
568        }

570        /**
571         * Creates a lock file for a resource name.
572         *
573         * @param resourceName
574         */
575        private void createLockFile(String resourceName) throws IOException {
576            File lockFile = new File(repositoryLocation, getFileLockName(resourceName));
577            lockFile.createNewFile();
578        }
579    
580        /**
581         * Removes a lock file.
582         * 
583         * @param resourceName
584         */
585        private void removeLockFile(String resourceName) {
586            File lockFile = new File(repositoryLocation, getFileLockName(resourceName) );
587            lockFile.delete();
588        }
589    
590        /**
591         * Removes all files that are locked and then inconsistent.
592         */
593        private void removeLockedResources() throws RepositoryException {
594            Map<String,Resource> resources = loadResources(repositoryLocation);
595            for(Resource r : resources.values()) {
596                if( isLocked( r.getName() ) ) {
597                    try {
598                        removeLockFile( r.getName());
599                    } catch (Exception e) {
600                        e.printStackTrace();
601                    }
602                    File resourceLocation = r.getLocation();
603                    try {
604                        resourceLocation.delete();
605                    } catch (Exception e) {
606                        e.printStackTrace();
607                    }
608                }
609            }
610    
611        }
