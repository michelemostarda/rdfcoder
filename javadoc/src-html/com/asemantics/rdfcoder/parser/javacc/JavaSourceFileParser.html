156        /**
157         * Processes a single compilation unit.
158         *
159         * @param location
160         * @param ast
161         */
162        public void processCompilationUnit(String location, ASTCompilationUnit ast) {
163    
164            javaCodeHandler = (JavaCodeHandler) getParseHandler();
165    
166            try {
167                javaCodeHandler.startCompilationUnit(location);
168            } catch (Throwable t) {
169                t.printStackTrace();
170            }
171    
172            // Retrieve package.
173            List packs = ast.findChildrenOfType(ASTPackageDeclaration.class);
174            final String packagePathStr;
175            final Identifier packagePath;
176            if (packs.size() > 0) {
177                packagePathStr = ((ASTName) ((ASTPackageDeclaration) packs.get(0)).findChildrenOfType(ASTName.class).get(0)).getName();
178            } else {
179                packagePathStr = ""; // Default package.
180            }
181            packagePath = IdentifierReader.readPackage(packagePathStr);
182    
183            try {
184                javaCodeHandler.startPackage(packagePath);
185            } catch (Throwable t) {
186                t.printStackTrace();
187            }
188    
189            try {
190    
191                // Define inports context.
192                ImportsContext importsContext = new ImportsContext();
193                importsContext.setContextPackage(packagePathStr);
194    
195                // Retrieve imports.
196                List imports = ast.findChildrenOfType(ASTImportDeclaration.class);
197                for (int i = 0; i < imports.size(); i++) {
198                    ASTImportDeclaration importDeclaration = (ASTImportDeclaration) imports.get(i);
199                    String importEntry = ((ASTName) importDeclaration.findChildrenOfType(ASTName.class).get(0)).getName();
200                    if (importDeclaration.isImportOnDemand()) { //Starred package.
201                        try {
202                            importsContext.addStarredPackage(importEntry);
203                        } catch(IllegalArgumentException iae) {
204                            javaCodeHandler.parseError( location, iae.getClass().getName()  + "[" + iae.getMessage() + "]" );
205                        }
206                    } else {
207                        importsContext.addFullyQualifiedObject(importEntry);
208                    }
209                }
210    
211                // Note: enumerations can be present only at first level.
212                extractEnumerations(
213                        packagePath,
214                        javaCodeHandler,
215                        (ASTCompilationUnit) ast.findChildrenOfType(ASTCompilationUnit.class).get(0)
216                );
217    
218                // Process first level entities.
219                processLevel(packagePath, importsContext, ast);
220    
221            } finally {
222                javaCodeHandler.endPackage();
223                javaCodeHandler.endCompilationUnit();
224            }
225        }

227        private void processLevel(Identifier packagePath, ImportsContext importsContext, SimpleNode simpleNode) {
228    
229            // Find classes.
230            List<ASTClassDeclaration> classes = findChildrenAtSameLevel(simpleNode, ASTClassDeclaration.class);
231    
232            // Find interfaces.
233            List<ASTInterfaceDeclaration> interfaces = findChildrenAtSameLevel(simpleNode, ASTInterfaceDeclaration.class);
234    
235            // Process Class.
236            String className;
237            String superClass = null;
238            for (ASTClassDeclaration classDeclaration : classes ) {
239    
240                List<ASTUnmodifiedClassDeclaration> unmodifiedClassDeclarations = classDeclaration.findChildrenOfType(ASTUnmodifiedClassDeclaration.class);
241                for( NamedNode unmodifiedClassDeclaration : unmodifiedClassDeclarations) {
242    
243                    // Class name.
244                    className = unmodifiedClassDeclaration.getName();
245                    final Identifier clazz = packagePath.copy().pushFragment(className, JavaCodeModel.CLASS_KEY).build();
246                    getObjectsTable().addObject(clazz);
247    
248                    // Class visibility.
249                    JavaCodeModel.JVisibility clsOrIntVisibility = retrieveVisibility(classDeclaration);
250    
251                    // Find super class.
252                    List classOrInterface = unmodifiedClassDeclaration.findChildrenOfType(ASTClassOrInterfaceType.class);
253                    if (classOrInterface.size() > 0) {   // Has super Class.
254                        superClass = retrieveObjectName( ((ASTClassOrInterfaceType) classOrInterface.get(0)).findChildrenOfType(ASTIdentifier.class) );
255                    }
256    
257                    javaCodeHandler.startClass(
258                            //TODO: HIGH - verify the subsequest row.
259                            extractModifiers( classDeclaration ),
260                            clsOrIntVisibility,
261                            packagePath.copy().pushFragment(className, JavaCodeModel.CLASS_KEY).build(),
262                            superClass != null ? qualifyType(importsContext, superClass, new JavaCodeModel.ObjectType(null) ): null,
263                            extractImplementedInterfaces(importsContext, unmodifiedClassDeclaration)
264                    );
265                    try {
266                        extractEnumerations(packagePath, javaCodeHandler, unmodifiedClassDeclaration);
267                        extractAttributes  (packagePath, javaCodeHandler, importsContext, unmodifiedClassDeclaration);
268                        extractContructors (javaCodeHandler, importsContext, unmodifiedClassDeclaration);
269                        extractMethods     (packagePath, javaCodeHandler, importsContext, unmodifiedClassDeclaration);
270    
271                        // Recursive invocation.
272                        processLevel(
273                                packagePath.copy().pushFragment(className, JavaCodeModel.CLASS_KEY).build(),
274                                importsContext,
275                                unmodifiedClassDeclaration
276                        );
277    
278                    } finally {
279                        javaCodeHandler.endClass();
280                    }
281    
282                }
283            }
284    
285            // Process interface.
286            String interfaceName = null;
287            for(ASTInterfaceDeclaration interfaceDeclaration : interfaces ) {
288    
289                List<ASTUnmodifiedInterfaceDeclaration> unmodifiedInterfaceDeclarations =
290                        interfaceDeclaration.findChildrenOfType(ASTUnmodifiedInterfaceDeclaration.class);
291    
292                for(NamedNode unmodifiedInterfaceDeclaration : unmodifiedInterfaceDeclarations) {
293    
294                    // Interface name.
295                    interfaceName = unmodifiedInterfaceDeclaration.getName();
296                    final Identifier iface = packagePath.copy().pushFragment(interfaceName, JavaCodeModel.INTERFACE_KEY).build();
297                    getObjectsTable().addObject(iface);
298    
299                    javaCodeHandler.startInterface(
300                            packagePath.copy().pushFragment(interfaceName, JavaCodeModel.INTERFACE_KEY).build(),
301                            extractImplementedInterfaces(importsContext, unmodifiedInterfaceDeclaration)
302                    );
303                    try {
304                        extractEnumerations(packagePath, javaCodeHandler, unmodifiedInterfaceDeclaration);
305                        extractAttributes  (packagePath, javaCodeHandler, importsContext, unmodifiedInterfaceDeclaration);
306                        extractMethods     (packagePath, javaCodeHandler, importsContext, unmodifiedInterfaceDeclaration);
307    
308                        // Recursive invocation.
309                        processLevel(
310                                packagePath.copy().pushFragment(interfaceName, JavaCodeModel.INTERFACE_KEY).build(),
311                                importsContext,
312                                unmodifiedInterfaceDeclaration
313                        );
314    
315                    } finally {
316                        javaCodeHandler.endInterface();
317                    }
318    
319                }
320            }
321    
322        }

325         * List of empty modifiers.
326         */
327        private static final JavaCodeModel.JModifier[] NO_MODIFIERS = new JavaCodeModel.JModifier[0];
328    
329        /**
330         * Extracts the list of modifiers associated to this access node.
331         * @param accessNode
332         * @return
333         */
334        private JavaCodeModel.JModifier[] extractModifiers(AccessNode accessNode) {
335            List<JavaCodeModel.JModifier> modifiers
336                    = new ArrayList<JavaCodeModel.JModifier>(JavaCodeModel.JModifier.values().length);
337            if( accessNode.isAbstract() ) {
338                modifiers.add(JavaCodeModel.JModifier.ABSTRACT);
339            }
340            if( accessNode.isFinal() ) {
341                modifiers.add(JavaCodeModel.JModifier.FINAL);
342            }
343            if( accessNode.isStatic() ) {
344                modifiers.add(JavaCodeModel.JModifier.STATIC);
345            }
346            if( accessNode.isVolatile() ) {
347                modifiers.add(JavaCodeModel.JModifier.VOLATILE);
348            }
349            if( accessNode.isNative() ) {
350                modifiers.add(JavaCodeModel.JModifier.NATIVE);
351            }
352            if( accessNode.isTransient() ) {
353                modifiers.add(JavaCodeModel.JModifier.TRANSIENT);
354            }
355            if( accessNode.isSynchronized() ) {
356                modifiers.add(JavaCodeModel.JModifier.SYNCHRONIZED);
357            }
358            return  modifiers.size() == 0 ? NO_MODIFIERS : modifiers.toArray(new JavaCodeModel.JModifier[modifiers.size()]);
359        }

362        /**
363         * Returns an array containing the fully quelified types of the implemented interfaces
364         * of the current simple node.
365         *
366         * @param importsContext
367         * @param simpleNode
368         * @return
369         */
370        private Identifier[] extractImplementedInterfaces(ImportsContext importsContext, SimpleNode simpleNode) {
371            Identifier[] interfacesArray = null;
372            List genericNameList = simpleNode.findChildrenOfType(ASTGenericNameList.class);
373            for(int i = 0; i < genericNameList.size(); i++) {
374                List classOrInterfaceTypes = ( (ASTGenericNameList) genericNameList.get(i) ).findChildrenOfType(ASTClassOrInterfaceType.class);
375                interfacesArray = new Identifier[classOrInterfaceTypes.size()];
376                for(int j = 0; j < classOrInterfaceTypes.size(); j++) {
377                    List identifierList = ( (ASTClassOrInterfaceType) classOrInterfaceTypes.get(j)).findChildrenOfType(ASTIdentifier.class);
378                    interfacesArray[j] = qualifyType(importsContext, retrieveObjectName(identifierList), new JavaCodeModel.InterfaceType(null) );
379                }
380            }
381            return interfacesArray;
382        }

384        private void extractAttributes(
385                Identifier packagePath,
386                JavaCodeHandler ch,
387                ImportsContext importsContext,
388                NamedNode namedNode
389        ) {
390            List attributes = namedNode.findChildrenOfType(ASTFieldDeclaration.class);
391            String attributeName;
392            JavaCodeModel.JVisibility attributeVisibility;
393            JavaCodeModel.JType attributeType;
394            for (int f = 0; f < attributes.size(); f++) {
395                ASTFieldDeclaration fieldDeclaration = (ASTFieldDeclaration) attributes.get(f);
396    
397                // Retrieve name.
398                ASTVariableDeclaratorId variableDeclaratorId =
399                        (ASTVariableDeclaratorId) fieldDeclaration.findChildrenOfType(ASTVariableDeclaratorId.class).get(0);
400                attributeName = variableDeclaratorId.getName();
401    
402                // Visibility.
403                attributeVisibility = retrieveVisibility(fieldDeclaration);
404                
405                // Retrieve array size.
406                int arraySize;
407                 boolean isArray;
408                List referenceTypes = fieldDeclaration.findChildrenOfType(ASTReferenceType.class);
409                if( referenceTypes.size() > 0 ) {
410                     ASTReferenceType astReferenceType = (ASTReferenceType) referenceTypes.get(0);
411                     arraySize = astReferenceType.getArrayCount();
412                     isArray = arraySize > 0;
413                } else {
414                    arraySize = 0;
415                    isArray = false;
416                }
417    
418                // Retrieve type.
419                List primitiveType = fieldDeclaration.findChildrenOfType(ASTPrimitiveType.class);
420                if (primitiveType.size() > 0) { // Primitive type declared.
421                    JavaCodeModel.JType type = JavaCodeModel.javaTypeToJType(((ASTPrimitiveType) primitiveType.get(0)).getName());
422                    attributeType =
423                            isArray
424                                    ?
425                            new JavaCodeModel.ArrayType(type, arraySize)
426                                    :
427                            type;
428                } else { // Obj identifier declared.
429                    String typeName = retrieveObjectName( fieldDeclaration.findChildrenOfType(ASTIdentifier.class) );
430                    JavaCodeModel.JType type = new JavaCodeModel.ObjectType(
431                            qualifyType(
432                                    importsContext,
433                                    typeName,
434                                    isArray
435                                            ?
436                                            new JavaCodeModel.ArrayType(
437                                                new JavaCodeModel.ObjectType(null),
438                                                arraySize
439                                            )
440                                            :
441                                            new JavaCodeModel.ObjectType(null)
442                            )
443                    );
444                    attributeType = isArray ? new JavaCodeModel.ArrayType(type, variableDeclaratorId.getArrayCount()) : type;
445                }
446    
447                ch.attribute(
448                        extractModifiers(fieldDeclaration),
449                        attributeVisibility,
450                        qualifyAttribute(packagePath, attributeName), attributeType, null);
451            }
452        }

454        private void extractContructors(JavaCodeHandler ch, ImportsContext importsContext, NamedNode namedNode) {
455            // Constructors.
456            List constructors = namedNode.findChildrenOfType(ASTConstructorDeclaration.class);
457            ASTConstructorDeclaration constructorDeclaration;
458            for(int c = 0; c < constructors.size(); c++) {
459    
460                constructorDeclaration = ((ASTConstructorDeclaration) constructors.get(c));
461                List formalParameters = constructorDeclaration.findChildrenOfType(ASTFormalParameter.class);
462    
463                // Extract parameters.
464                ASTFormalParameter formalParameter;
465                List<String> parameterNames = new ArrayList<String>();
466                List<JavaCodeModel.JType> parameterTypes = new ArrayList<JavaCodeModel.JType>();
467                for( int fp = 0; fp < formalParameters.size(); fp++ ) {
468                    formalParameter = (ASTFormalParameter) formalParameters.get(fp);
469                    parameterNames.add(
470                            ( (ASTVariableDeclaratorId) formalParameter.findChildrenOfType(ASTVariableDeclaratorId.class).get(0)).getName()
471                    );
472                    parameterTypes.add( extractFormalparameterType(importsContext, formalParameter) );
473                }
474    
475                // Extract exceptions.
476                JavaCodeModel.ExceptionType[] exceptions = extractExceptions(importsContext, constructorDeclaration);
477    
478                ch.constructor(
479                        extractModifiers(constructorDeclaration),
480                        retrieveVisibility(constructorDeclaration),
481                        c, // TODO: replace index with signature hashcode.
482                        parameterNames.toArray( new String[parameterNames.size()] ),
483                        parameterTypes.toArray( new JavaCodeModel.JType[parameterTypes.size()] ),
484                        exceptions
485                );
486            }
487        }

489        private void extractMethods(
490                Identifier packagePath,
491                JavaCodeHandler ch,
492                ImportsContext importsContext,
493                NamedNode namedNode
494        ) {
495            List methods = namedNode.findChildrenOfType(ASTMethodDeclaration.class);
496            String methodName;
497            JavaCodeModel.JVisibility methodVisibility;
498            boolean returnTypeIsArray;
499            int returnTypeArraySize;
500            JavaCodeModel.JType returnType;
501            List<JavaSourceFileParser.Method> methodsBuffer = new ArrayList<JavaSourceFileParser.Method>();
502            for (int m = 0; m < methods.size(); m++) {
503                ASTMethodDeclaration methodDeclaration = (ASTMethodDeclaration) methods.get(m);
504    
505                // Retrieve method name.
506                methodName = ((ASTMethodDeclarator) methodDeclaration.findChildrenOfType(ASTMethodDeclarator.class).get(0)).getName();
507    
508                // Retrieve method visibility.
509                methodVisibility = retrieveVisibility(methodDeclaration);
510    
511                // Return type is array.
512                List referenceType = methodDeclaration.findChildrenOfType(ASTReferenceType.class);
513                if (referenceType.size() > 0) {
514                    returnTypeArraySize = ((ASTReferenceType) referenceType.get(0)).getArrayCount();
515                    returnTypeIsArray = returnTypeArraySize > 0;
516                } else {
517                    returnTypeArraySize = 0;                
518                    returnTypeIsArray = false;
519                }
520    
521    
522                // Find return type.
523                ASTResultType resultType = (ASTResultType) methodDeclaration.findChildrenOfType(ASTResultType.class).get(0);
524                List primitiveType = resultType.findChildrenOfType(ASTPrimitiveType.class);
525                if (primitiveType.size() > 0) { // Result type is primitive.
526                    JavaCodeModel.JType type = JavaCodeModel.javaTypeToJType(((ASTPrimitiveType) primitiveType.get(0)).getName());
527                    if( returnTypeIsArray ) {
528                        returnType = new JavaCodeModel.ArrayType(type, returnTypeArraySize);
529                    } else {
530                       returnType = type;
531                    }
532                } else if (resultType.findChildrenOfType(ASTIdentifier.class).size() > 0) { // Result type is Object.
533                    Identifier qualifiedType = qualifyType(
534                            importsContext,
535                            retrieveObjectName( resultType.findChildrenOfType(ASTIdentifier.class) ),
536                            returnTypeIsArray
537                                    ?
538                                    new JavaCodeModel.ArrayType(
539                                        new JavaCodeModel.ObjectType(null),
540                                        returnTypeArraySize
541                                    )
542                                    :
543                                    new JavaCodeModel.ObjectType(null)
544                    );
545                    returnType = (
546                            returnTypeIsArray
547                                    ?
548                            new JavaCodeModel.ArrayType( new JavaCodeModel.ObjectType(qualifiedType), returnTypeArraySize)
549                                    :
550                            new JavaCodeModel.ObjectType(qualifiedType)
551                    );
552                } else { // Void type.
553                    returnType = JavaCodeModel.VOID;
554                }
555    
556                // Extract formal parameters.
557                List formalParameters = methodDeclaration.findChildrenOfType(ASTFormalParameter.class);
558                ASTFormalParameter formalParameter;
559                List<String> parameterNames = new ArrayList();
560                List<JavaCodeModel.JType> parameterTypes = new ArrayList();
561                for (int p = 0; p < formalParameters.size(); p++) {
562                    formalParameter = (ASTFormalParameter) formalParameters.get(p);
563                    // Parameter identifier.
564                    parameterNames.add(((ASTVariableDeclaratorId) formalParameter.findChildrenOfType(ASTVariableDeclaratorId.class).get(0)).getName());
565                    // Parameter type.
566                    parameterTypes.add(extractFormalparameterType(importsContext, formalParameter));
567                }
568    
569                // Extract exceptions.
570                JavaCodeModel.ExceptionType[] exceptions = extractExceptions(importsContext, methodDeclaration);
571    
572                // Storing method.
573                methodsBuffer.add(
574                        new JavaSourceFileParser.Method(
575                                extractModifiers(methodDeclaration),
576                                methodVisibility,
577                                qualifyMethod(packagePath, methodName),
578                                parameterNames.toArray(new String[parameterNames.size()]),
579                                parameterTypes.toArray(new JavaCodeModel.JType[parameterTypes.size()]),
580                                returnType,
581                                exceptions
582    
583                        )
584                );
585            }
586    
587            // Group signatures by name.
588            Map<Identifier, List<Method>> methodsMap = new HashMap<Identifier, List<Method>>();
589            Iterator<Method> iter = methodsBuffer.iterator();
590            JavaSourceFileParser.Method curr;
591            while (iter.hasNext()) {
592                curr = iter.next();
593                List signatures = methodsMap.get(curr.methodPath);
594                if (signatures == null) {
595                    signatures = new ArrayList();
596                    methodsMap.put(curr.methodPath, signatures);
597                }
598                signatures.add(curr);
599            }
600    
601            // Store methods.
602            Iterator<Map.Entry<Identifier, List<JavaSourceFileParser.Method>>> entries = methodsMap.entrySet().iterator();
603            Map.Entry<Identifier, List<JavaSourceFileParser.Method>> entry;
604            while (entries.hasNext()) {
605                entry = entries.next();
606                for (int i = 0; i < entry.getValue().size(); i++) {
607                    ch.method(
608                            entry.getValue().get(i).modifiers,
609                            entry.getValue().get(i).visibility,
610                            entry.getKey(),
611                            i, // TODO: replace index with signature hashcode.
612                            entry.getValue().get(i).parameterNames,
613                            entry.getValue().get(i).parameterTypes,
614                            entry.getValue().get(i).returnType,
615                            entry.getValue().get(i).exceptions
616                    );
617                }
618            }
619        }

621        private void extractEnumerations(Identifier packagePath, JavaCodeHandler ch, SimpleNode simpleNode) {
622            List enumDeclarations = simpleNode.findChildrenOfType(ASTEnumDeclaration.class);
623            ASTEnumDeclaration enumDeclaration;
624            for(int ed = 0; ed < enumDeclarations.size(); ed++) {
625                enumDeclaration = (ASTEnumDeclaration) enumDeclarations.get(ed);
626                 JavaCodeModel.JVisibility visibility = retrieveVisibility(enumDeclaration);
627                ASTIdentifier indentifier = (ASTIdentifier) enumDeclaration.findChildrenOfType(ASTIdentifier.class).get(0);
628                String identifier = indentifier.getName();
629                List enumElements = enumDeclaration.findChildrenOfType(ASTEnumElement.class);
630                List elements = new ArrayList();
631                for(int ee = 0; ee < enumElements.size(); ee++ ) {
632                    elements.add( ((ASTEnumElement) enumElements.get(ee)).getName() );
633                }
634                ch.startEnumeration(
635                        extractModifiers(enumDeclaration),
636                        visibility,
637                        qualifyEnumeration(packagePath, identifier),
638                        (String[]) elements.toArray(new String[elements.size()])
639                );
640                //TODO: HIGH - extract fields + methods.
641                ch.endEnumeration();
642                elements.clear();
643            }
644        }

646        private JavaCodeModel.JType extractFormalparameterType(ImportsContext importsContext, ASTFormalParameter formalParameter) {
647    
648            // Retrieve array size.
649            int arraySize;
650            List referenceTypes = formalParameter.findChildrenOfType(ASTReferenceType.class);
651            if( referenceTypes.size() > 0 ) {
652                ASTReferenceType referenceType = (ASTReferenceType) referenceTypes.get(0);
653                arraySize = referenceType.getArrayCount();
654            } else {
655                arraySize = 0;
656            }
657    
658            // Retrieve type.
659            List primitiveType = formalParameter.findChildrenOfType(ASTPrimitiveType.class);
660            if (primitiveType.size() > 0) { // Result type is Primitive.
661                return JavaCodeModel.javaTypeToJType(((ASTPrimitiveType) primitiveType.get(0)).getName());
662            } else { // Result type is Object.
663                String typeName =  retrieveObjectName( formalParameter.findChildrenOfType(ASTIdentifier.class) );
664                if(arraySize > 0) {
665                    return new JavaCodeModel.ArrayType(
666                            new JavaCodeModel.ObjectType(
667                                qualifyType(
668                                        importsContext,
669                                        typeName,
670                                        new JavaCodeModel.ArrayType(new JavaCodeModel.ObjectType(null), arraySize)
671                                )
672                            ),
673                             arraySize
674                    );
675                } else {
676                    return new JavaCodeModel.ObjectType(
677                            qualifyType(
678                                importsContext,
679                                typeName,
680                                new JavaCodeModel.ObjectType(null)
681                            )
682                    );
683                }
684            }
685        }

687        private JavaCodeModel.ExceptionType[] extractExceptions(ImportsContext importsContext,AccessNode accessNode) {
688            List nameLists = accessNode.findChildrenOfType(ASTNameList.class);
689            JavaCodeModel.ExceptionType[] exceptions = null;
690            if(nameLists.size() > 0) {
691                ASTNameList nameList = (ASTNameList) nameLists.get(0);
692                List namesList = nameList.findChildrenOfType(ASTName.class);
693                String exceptionName;
694                exceptions = new JavaCodeModel.ExceptionType[namesList.size()];
695                for( int i = 0; i < namesList.size(); i++ ) {
696                    ASTName name = (ASTName) namesList.get(i);
697                    exceptionName = retrieveObjectName(name.findChildrenOfType(ASTIdentifier.class));
698                    exceptions[i] = new JavaCodeModel.ExceptionType(
699                        qualifyType(importsContext, exceptionName, new JavaCodeModel.ExceptionType(null))
700                    );
701                }
702            }
703            return exceptions;
704        }

722        public void parse(File file) throws ParserException, IOException {
723            FileInputStream fis;
724            try {
725                fis = new FileInputStream(file);
726            } catch (FileNotFoundException fnfe) {
727                throw new ParserException(fnfe, file.getAbsolutePath());
728            }
729            JavaParser jp = new JavaParser(fis);
730            ASTCompilationUnit ast;
731            try {
732                ast = jp.CompilationUnit();
733                processCompilationUnit(file.getAbsolutePath(), ast);
734            } catch (ParseException pe) {
735                throw new ParserException( pe, file.getAbsolutePath() );
736            } finally {
737                fis.close();
738            }
739        }

741        /**
742           * Returns the relative depth of a descendant in respect with the parent.
743           * @param parent
744           * @param descendant
745           * @return
746           */
747          private int childRelativeDepth(Node parent, Node descendant) {
748              Node childParent = descendant;
749              int depth = 0;
750              while(childParent != parent) {
751                  childParent = childParent.jjtGetParent();
752                  depth++;
753                  if(childParent == null) {
754                      throw new IllegalStateException();
755                  }
756              }
757              return depth;
758          }

760          /**
761           * Returns a list of descendant of simpleNode of type maching the given clazz
762           * where all results are at the same depth level of the first child found.
763           *
764           * @param simpleNode
765           * @param clazz
766           * @return
767           */
768          private List findChildrenAtSameLevel(SimpleNode simpleNode, Class clazz) {
769              List<SimpleNode> children = simpleNode.findChildrenOfType(clazz);
770              if(children.size() == 0) {
771                  return children;
772              }
773              final int depthLevel = childRelativeDepth(simpleNode, children.get(0));
774              List sameDepth = new ArrayList();
775              for(SimpleNode child : children) {
776                  if( childRelativeDepth(simpleNode, child ) == depthLevel ) {
777                      sameDepth.add(child);
778                  }
779              }
780              return sameDepth;
781          }

798           * Returns the JVisibility of an access node.
799           *
800           * @param an
801           * @return
802           */
803          private JavaCodeModel.JVisibility retrieveVisibility(AccessNode an) {
804              if (an.isPublic()) {
805                  return JavaCodeModel.JVisibility.PUBLIC;
806              } else if (an.isProtected()) {
807                  return JavaCodeModel.JVisibility.PROTECTED;
808              } else if (an.isPackage()) {
809                  return JavaCodeModel.JVisibility.DEFAULT;
810              } else {
811                  return JavaCodeModel.JVisibility.PRIVATE;
812              }

815          /**
816           * Retruns the qualified typeName for a typeName string.
817           *
818           * @param typeName
819           * @return
820           */
821          private Identifier qualifyType(ImportsContext importsContext, String typeName, JavaCodeModel.JType type) {
822              Identifier qualifiedObject = importsContext.qualifyType(getObjectsTable(), typeName);
823              if (qualifiedObject == null) {
824                  Identifier tempId = javaCodeHandler.generateTempUniqueIdentifier();
825                  if(type instanceof JavaCodeModel.ObjectType) {
826                      ((JavaCodeModel.ObjectType) type).setInternalIdentifier(tempId);
827                  } else if( type instanceof JavaCodeModel.InterfaceType) {
828                      ((JavaCodeModel.InterfaceType) type).setInternalIdentifier(tempId);
829                  } else if( type instanceof JavaCodeModel.ArrayType) {
830                      ((JavaCodeModel.ArrayType) type).setInternalIdentifier(tempId);
831                  } else {
832                      throw new IllegalStateException();
833                  }
834                  getObjectsTable().addUnresolvedType(typeName, type, importsContext);
835                  return tempId;
836              }
837              return qualifiedObject;
838          }
