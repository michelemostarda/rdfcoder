217        public JavaCodeHandler createStatisticsCodeHandler(JavaCodeHandler ch) {
218            StatisticsJavaCodeHandler sch = new StatisticsJavaCodeHandler(ch);
219            statisticsCodeHandlers.add(sch);
220            return sch;
221        }
222    
223        public void detachHandlers() {
224            Iterator<StatisticsJavaCodeHandler> handlers = statisticsCodeHandlers.iterator();
225            while (handlers.hasNext()) {
226                handlers.next().dispose();
227                handlers.remove();
228            }
229        }
230    
231        public void reset() {
232            detachHandlers();
233    
234            startParsingTime = endParsingTime = 0;
235    
236            parseErrors = 0;
237            parsedFiles = 0;
238            parsedClasses = 0;
239            parsedInterfaces = 0;
240            parsedAttributes = 0;
241            parsedConstructors = 0;
242            parsedMethods = 0;
243            parsedEnumarations = 0;
244            generatedTempIds = 0;
245            replacedEntries = 0;
246            unresolved = null;
247    
248            errorMessages.delete(0, errorMessages.length());
249        }
250    
251        public void clean() {
252            reset();
253        }

255        public String toStringReport() {
256            StringBuilder sb = new StringBuilder();
257            sb.append("parsing time (secs):").append( (endParsingTime - startParsingTime) / 1000 ).append("\n"); 
258            sb.append("parsed files: ").append(parsedFiles).append("\n");
259            sb.append("parsed classes: ").append(parsedClasses).append("\n");
260            sb.append("parsed interfaces: ").append(parsedInterfaces).append("\n");
261            sb.append("parsed attributes: ").append(parsedAttributes).append("\n");
262            sb.append("parsed constructors: ").append(parsedConstructors).append("\n");
263            sb.append("parsed methods: ").append(parsedMethods).append("\n");
264            sb.append("parsed enumerations: ").append(parsedEnumarations).append("\n");
265            sb.append("javadoc entries: ").append(javadocEntries).append("\n");
266            sb.append("classes javadoc: ").append(classesJavadoc).append("\n");
267            sb.append("fields javadoc: ").append(fieldsJavadoc).append("\n");
268            sb.append("constructors javadoc: ").append(constructorsJavadoc).append("\n");
269            sb.append("methods javadoc: ").append(methodsJavadoc).append("\n");
270            sb.append("generated temporary identifiers: ").append(generatedTempIds).append("\n");
271            sb.append("replaced entries: ").append(replacedEntries).append("\n");
272    
273            sb.append("unresolved [").append( unresolved != null ? unresolved.length : 0).append("] {\n");
274            if( unresolved != null ) {
275                for( String u : unresolved) {
276                    sb.append(u).append("\n");
277                }
278            }
279            sb.append("}\n");
280    
281    
282            sb.append("parse errors[").append(parseErrors).append("] {\n");
283            sb.append( errorMessages );
284            sb.append("}\n");
285    
286            return sb.toString();
287        }
288    
289        public String toString() {
290            return toStringReport();
291        }

293        protected class StatisticsJavaCodeHandler implements JavaCodeHandler {
294    
295            private JavaCodeHandler wrapped;

297            StatisticsJavaCodeHandler(JavaCodeHandler ch) {
298                this.wrapped = ch;
299            }

302                wrapped = null;
303            }

305            public void startParsing(String libraryName, String location) {
306                if(wrapped == null) { throw new IllegalStateException(); }
307                startParsingTime = System.currentTimeMillis();
308                wrapped.startParsing(libraryName, location);
309            }
310    
311            public void endParsing() {
312                if(wrapped == null) { throw new IllegalStateException(); }
313                wrapped.endParsing();
314                endParsingTime = System.currentTimeMillis();
315            }
