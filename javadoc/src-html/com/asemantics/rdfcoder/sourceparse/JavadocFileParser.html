193        /**
194         * Contains all data relative the parsing of a class.
195         */
196        private class JDClassData {
197    
198            String classIdentifier;
199    
200            int parentesisCounter = 0;
201    
202            JDClassData(String ci) {
203                classIdentifier = ci;
204            }
205    
206            void increment() {
207                parentesisCounter++;
208            }
209    
210            void decrement() {
211                parentesisCounter--;
212            }
213    
214            boolean endOfClass() {
215                return parentesisCounter == 0;
216            }
217    
218            public String toString() {
219                return classIdentifier;
220            }
221        }

233        /**
234         * The back tracking buffer size.
235         */
236        private static final int BACKTRACK_BUFFER_SIZE = 3;
237    
238        /**
239         * Contains the list of tokens to be processed.
240         */
241        private class TokensPipe {
242    
243            /**
244             * Internal queue.
245             */
246            private Queue<JDToken> queue;
247    
248            /**
249             * Back tracking buffer.
250             */
251            private List<JDToken> backTrackBuffer = new ArrayList<JDToken>();
252    
253            /**
254             * Constructor.
255             */
256            TokensPipe() {
257                queue = new LinkedList<JDToken>();
258            }
259    
260            /**
261             * Adds a token to the pipe.
262             *
263             * @param token
264             */
265            void addToken(JDToken token) {
266                queue.add(token);
267            }
268    
269            /**
270             * Returns the next token to be processed.
271             *
272             * @return
273             */
274            public JDToken nextToken() {
275                JDToken headToken = queue.poll();
276                if(headToken != null) {
277                    if(backTrackBuffer.size() == BACKTRACK_BUFFER_SIZE) {
278                        backTrackBuffer.remove(0);
279                    }
280                    backTrackBuffer.add(headToken);
281                }
282                return headToken;
283            }
284    
285            /**
286             * Returns the token with backtrack <i>b</i>.
287             * @param b
288             * @return
289             */
290            public JDToken backtrack(int b) {
291                if(b <= 0) {
292                    throw new IllegalArgumentException();
293                }
294                if(b > backTrackBuffer.size()) {
295                    return null;
296                }
297                return backTrackBuffer.get(b - 1);
298            }
299    
300            public void clear() {
301                queue.clear();
302                backTrackBuffer.clear();
303            }
304        }

395        /**
396         * Parses a compilation unit.
397         * 
398         * @param inputStream
399         * @param compilationUnitPath
400         * @throws ParserException
401         */
402        public void parse(InputStream inputStream, String compilationUnitPath) throws ParserException {
403    
404            final JavadocHandler javadocHandler = (JavadocHandler) getParseHandler();
405    
406            tokensPipe.clear();
407            classes.clear();
408            methodName = null;
409            attribute = null;
410            entryText    .delete(0, entryText.length());
411            attributeText.delete(0, attributeText.length());
412            attributesMap.clear();
413            parameters   .clear();
414            lastEntry = null;
415    
416            try {
417                getParseHandler().startCompilationUnit(compilationUnitPath);
418            } catch (Throwable t) {
419                t.printStackTrace();
420            }
421    
422            JDToken token;
423            try {
424    
425                // Main cycle.
426                while( true ) {
427    
428                    // Extract tokens from imput stream.
429                    extractTokens(inputStream, tokensPipe);
430    
431                    // Check if end of file has been reached.
432                    if( (token = tokensPipe.nextToken() ) == JD_EOF) {
433                        break;
434                    }
435    
436                    // Tokens to be processed when inside Javadoc block.
437                    if(insideComment) {
438    
439                        // Closing Javadoc, the entity is emitted.
440                        if(token == JD_CLOSE_COMMENT) {
441    
442                            // Stores the last attribute if any.
443                            storeCurrentAttribute();
444    
445                            insideComment = false;
446                            String entryText = this.entryText.toString();
447                            int splitPoint = entryText.indexOf(".");
448                            String shortComment, longComment;
449                            if(splitPoint != -1) {
450                                shortComment = entryText.substring(0, splitPoint).trim();
451                                longComment = entryText.substring(splitPoint + 1).trim();
452                            } else {
453                                shortComment = entryText;
454                                longComment  = "";
455                            }
456    
457                            // Creates last entry data.
458                            JavadocEntry entry = new JavadocEntry(
459                                    shortComment,
460                                    longComment,
461                                    new HashMap<String,List<String>>(attributesMap),
462                                    lastOpenComment.getRow(),
463                                    lastOpenComment.getCol()
464                            );
465                            lastEntry = entry;
466    
467                            // Notifies last entry to listener.
468                            try {
469                                // A Javadoc entry has been found.
470                                javadocHandler.parsedEntry(entry);
471                            } catch(Throwable t) {
472                                t.printStackTrace();
473                            }
474    
475                        // Process attribute.
476                        } else if( token instanceof JDAttribute) {
477    
478                            // Storing previous attribute.
479                            storeCurrentAttribute();
480    
481                            attribute = token.getValue();
482                            attributeText.delete(0, attributeText.length());
483    
484                        // Process attribute words.
485                        } else if( attribute != null && token instanceof JDWord ) {
486    
487                            attributeText.append(token.getValue());
488    
489                        // Process entry comment words.
490                        } else if( token instanceof JDWord ) {
491    
492                            entryText.append(token.getValue());
493    
494                        }
495    
496                    // Tokens to be processed when outside javadoc blocks.
497                    } else {
498    
499                        // Javadoc commend opening.
500                        if( token instanceof JDOpenComment ) {
501    
502                            lastOpenComment = (JDOpenComment) token;
503                            insideComment = true;
504                            entryText.delete(0, entryText.length());
505                            attributesMap.clear();
506                            attributeText.delete(0, attributeText.length());
507    
508                        // Class declaration beginning.
509                        } else if(token instanceof JDClassDeclarator) {
510    
511                            classExpected = true;
512    
513                        // Class signature handling.
514                        } else if(classExpected && token instanceof JDWord) {
515    
516    
517                            classExpected = false;
518                            classes.push( new JDClassData( token.getValue() ) );
519                            if(lastEntry != null) {
520                                String containerPath = getContainerPath();
521                                try {
522                                    javadocHandler.classJavadoc( lastEntry, containerPath );
523                                } catch(Throwable t) {
524                                    t.printStackTrace();
525                                }
526                                lastEntry = null;
527                            }
528    
529                        // Opening class or method.
530                        } else if( ! classes.isEmpty() && token == JD_OPEN_CLASS_OR_METHOD) {
531    
532                            classes.peek().increment();
533    
534                        // Closing class or method.
535                        } else if(  ! classes.isEmpty() && token == JD_CLOSE_CLASS_OR_METHOD) {
536    
537                            classes.peek().decrement();
538                            if(classes.peek().endOfClass()) {
539                                classes.pop();
540                            }
541    
542                        // Begin parameters list.
543                        } else if(token == JD_OPEN_PARAMETERS) {
544    
545                            parameters.clear();
546                            methodName = tokensPipe.backtrack(2).getValue();
547    
548                        // Ends parameters list.
549                        } else if(token == JD_CLOSE_PARAMETERS) {
550    
551                            // Adds last parameter.
552                            parameters.add( new JDParameter( tokensPipe.backtrack(1).getValue(), tokensPipe.backtrack(2).getValue() ) );
553    
554                            // Notifies to the listener the method javadoc recognition.
555                            if(lastEntry != null) {
556                                String containerPath = getContainerPath();
557                                String pathToMethod    = (containerPath.equals("") ? "" : containerPath + JavaCodeHandler.PACKAGE_SEPARATOR) + methodName;
558                                String[] signature   = toSignature(parameters);
559                                try {
560                                    javadocHandler.methodJavadoc(lastEntry, pathToMethod, signature);
561                                } catch (Throwable t) {
562                                    t.printStackTrace();
563                                }
564                                methodName = null;
565                                lastEntry  = null;
566                            }
567    
568                        // Parameters separator.
569                        } else if(token == JD_PARAMETERS_SEPARATOR) {
570    
571                            parameters.add( new JDParameter( tokensPipe.backtrack(1).getValue(), tokensPipe.backtrack(2).getValue() ) );
572    
573                        }
574    
575                    }
576                }
577    
578            } catch(IOException ioe) {
579                throw new ParserException(ioe, compilationUnitPath);
580            }
581    
582            try {
583                getParseHandler().endCompilationUnit();
584            } catch (Throwable t) {
585                t.printStackTrace();
586            }
587        }

589        /**
590         * Parses a file.
591         * @param file
592         * @throws ParserException
593         */
594        public void parse(File file) throws ParserException {
595            FileInputStream fileInputStream = null;
596            try {
597                fileInputStream = new FileInputStream(file);
598                parse(fileInputStream, file.getAbsolutePath());
599                fileInputStream.close();
600            } catch (FileNotFoundException fnfe) {
601                throw new ParserException(fnfe, file.getAbsolutePath());
602            } catch (IOException ioe) {
603                throw new ParserException(ioe, file.getAbsolutePath());
604            }
605        }

607        /**
608         *
609         * Tokenizer: extracts token from the given <i>char stream</i>.
610         *
611         * @param inputStream
612         * @param tokensPipe
613         * @throws IOException
614         */
615        private void extractTokens(InputStream inputStream, TokensPipe tokensPipe) throws IOException {
616            int intc;
617            char c;
618    
619            tokenBuffer.delete(0, tokenBuffer.length());
620            while( (intc = inputStream.read()) != -1 ) {
621                c = (char) intc;
622    
623                // Manage row / col position.
624                if(c == '\n') {
625                    col = 0;
626                    row++;
627                } else {
628                    col++;
629                }
630    
631                // Manage special characters.
632                if(c == ' ' || c == '\n' || c == '{' || c == '}' || c == '(' || c == ')' || c == ',') {
633    
634                    // Tokens.
635    
636                    String bufferContent = tokenBuffer.toString();
637    
638                    // Handles inside <i>*</i> comment.
639                    if("*".equals(bufferContent)) {
640                        tokenBuffer.delete(0, tokenBuffer.length());
641                        continue;
642                    } 
643    
644                    // Handles begin comment.
645                    if("/**".equals(bufferContent.trim())) {
646                        tokensPipe.addToken( new JDOpenComment(row, col) );
647                        return;
648                    }
649    
650                    // Handles end comment.
651                    if("*/".equals(bufferContent)) {
652                        tokensPipe.addToken( JD_CLOSE_COMMENT );
653                        return;
654                    }
655    
656                    // Handles class key.
657                    if("class".equals(bufferContent) ) {
658                        tokensPipe.addToken( new JDClassDeclarator(row, col) );
659                        return;
660                    }
661    
662                    // Handles attribute.
663                    if(bufferContent.length() > 1 && bufferContent.charAt(0) == '@') {
664                        tokensPipe.addToken( new JDAttribute(bufferContent) );
665                        return;
666                    }
667    
668                    // Handles word separators.
669                    tokensPipe.addToken( new JDWord(bufferContent + " ") );
670    
671                    if( c == '{' ) {
672                        tokensPipe.addToken(JD_OPEN_CLASS_OR_METHOD);
673                    } else if(c == '}') {
674                        tokensPipe.addToken(JD_CLOSE_CLASS_OR_METHOD);
675                    } else if(c == '(') {
676                        tokensPipe.addToken(JD_OPEN_PARAMETERS);
677                    } else if(c == ')') {
678                        tokensPipe.addToken(JD_CLOSE_PARAMETERS);
679                    } else if(c == ',') {
680                        tokensPipe.addToken(JD_PARAMETERS_SEPARATOR);
681                    }
682                    return;
683    
684                } else {
685                    tokenBuffer.append(c);
686                }
687            }
688            tokensPipe.addToken( JD_EOF );
689            return;
690        }

692        /**
693         * Stores the current attribute.
694         */
695        private void storeCurrentAttribute() {
696            if(attribute != null) {
697                String key = attribute.trim();
698                List<String> list = attributesMap.get(key);
699                if(list == null) {
700                    list = new ArrayList<String>();
701                    attributesMap.put(key, list);
702                }
703                list.add(attributeText.toString().trim());
704            }
705        }

707        /**
708         * Returns the container path.
709         *
710         * @return
711         */
712        private String getContainerPath() {
713            return CoderUtils.join(classes, CoderUtils.PACKAGE_SEPARATOR);
714        }

716        /**
717         * Converts the parameters list to a method signature.
718         *
719         * @param parameters
720         * @return
721         */
722        private String[] toSignature(List<JDParameter> parameters) {
723            String[] signature = new String[parameters.size()];
724            for(int i = 0; i < parameters.size(); i++) {
725                signature[i] = parameters.get(i).name;
726            }
727            return signature;
728        }
