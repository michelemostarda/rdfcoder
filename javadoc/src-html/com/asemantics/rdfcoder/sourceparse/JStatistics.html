195        public JavaCodeHandler createStatisticsCodeHandler(JavaCodeHandler ch) {
196            StatisticsJavaCodeHandler sch = new StatisticsJavaCodeHandler(ch);
197            statisticsCodeHandlers.add(sch);
198            return sch;
199        }
200    
201        public void reset() {
202            Iterator<StatisticsJavaCodeHandler> handlers = statisticsCodeHandlers.iterator();
203            while(handlers.hasNext()) {
204                handlers.next().dispose();
205                handlers.remove();
206            }
207    
208            parseErrors = 0;
209            parsedFiles = 0;
210            parsedClasses = 0;
211            parsedInterfaces = 0;
212            parsedAttributes = 0;
213            parsedConstructors = 0;
214            parsedMethods = 0;
215            parsedEnumarations = 0;
216            generatedTempIds = 0;
217            replacedEntries = 0;
218            unresolved = null;
219    
220            errorMessages.delete(0, errorMessages.length());
221        }
222    
223        public void clean() {
224            reset();
225        }
226    
227        public String toStringReport() {
228            StringBuilder sb = new StringBuilder();
229            sb.append("parsing time (secs):").append( (endParsingTime - startParsingTime) / 1000 ).append("\n"); 
230            sb.append("parsed files: ").append(parsedFiles).append("\n");
231            sb.append("parsed classes: ").append(parsedClasses).append("\n");
232            sb.append("parsed interfaces: ").append(parsedInterfaces).append("\n");
233            sb.append("parsed attributes: ").append(parsedAttributes).append("\n");
234            sb.append("parsed constructors: ").append(parsedConstructors).append("\n");
235            sb.append("parsed methods: ").append(parsedMethods).append("\n");
236            sb.append("parsed enumerations: ").append(parsedEnumarations).append("\n");
237            sb.append("javadoc entries: ").append(javadocEntries).append("\n");
238            sb.append("classes javadoc: ").append(classesJavadoc).append("\n");
239            sb.append("methods javadoc: ").append(methodsJavadoc).append("\n");
240            sb.append("generated temporary indentifiers: ").append(generatedTempIds).append("\n");
241            sb.append("replaced entries: ").append(replacedEntries).append("\n");
242    
243            sb.append("unresolved [").append( unresolved != null ? unresolved.length : 0).append("] {\n");
244            if( unresolved != null ) {
245                for( String u : unresolved) {
246                    sb.append(u).append("\n");
247                }
248            }
249            sb.append("}\n");
250    
251    
252            sb.append("parse errors[").append(parseErrors).append("] {\n");
253            sb.append( errorMessages );
254            sb.append("}\n");
255    
256            return sb.toString();
257        }
258    
259        public String toString() {
260            return toStringReport();
261        }

263        protected class StatisticsJavaCodeHandler implements JavaCodeHandler {
264    
265            private JavaCodeHandler wrapped;
266    
267            StatisticsJavaCodeHandler(JavaCodeHandler ch) {
268                this.wrapped = ch;
269            }
270    
271            void dispose() {
272                wrapped = null;
273            }
274    
275            public void startParsing(String libraryName, String location) {
276                if(wrapped == null) { throw new IllegalStateException(); }
277                startParsingTime = System.currentTimeMillis();
278                wrapped.startParsing(libraryName, location);
279            }
280    
281            public void endParsing() {
282                if(wrapped == null) { throw new IllegalStateException(); }
283                wrapped.endParsing();
284                endParsingTime = System.currentTimeMillis();
285            }
286    
287            public void startCompilationUnit(String identifier) {
288                if(wrapped == null) { throw new IllegalStateException(); }
289                wrapped.startCompilationUnit(identifier);
290                parsedFiles++;
291            }
292    
293            public void endCompilationUnit() {
294                if(wrapped == null) { throw new IllegalStateException(); }
295                wrapped.endCompilationUnit();
296            }
297    
298            public void startPackage(Identifier pathToPackage) {

304                if(wrapped == null) { throw new IllegalStateException(); }
305                wrapped.endPackage();
306            }
307    
308            public void startInterface(Identifier pathToInterface, Identifier[] extendedInterfaces) {
309                if(wrapped == null) { throw new IllegalStateException(); }

311            }
312

314                if(wrapped == null) { throw new IllegalStateException(); }

317            }

367                    int overloadIndex,

370                    JavaCodeModel.ExceptionType[] exceptions

389                parsedMethods++;

392            public void parseError(String location, String description) {
393                errorMessages.append("location='" + location + "' description='" + description + "'\n" );
394                parseErrors++;
395            }

399                unresolved = types;

402            }

422                javadocEntries++;
423            }

426                if(wrapped == null) { throw new IllegalStateException(); }

428                classesJavadoc++;

431            public void methodJavadoc(JavadocEntry entry, String pathToMethod, String[] signature) {
432                if(wrapped == null) { throw new IllegalStateException(); }
433                wrapped.methodJavadoc(entry, pathToMethod, signature);
434                methodsJavadoc++;
435            }
436    
437            public Identifier generateTempUniqueIdentifier() {
438                if(wrapped == null) { throw new IllegalStateException(); }
439                generatedTempIds++;
440                return wrapped.generateTempUniqueIdentifier();
441            }
442    
443            public int replaceIdentifierWithQualifiedType(Identifier identifier, Identifier qualifiedType) {
444                if(wrapped == null) { throw new IllegalStateException(); }
445                int replaced = wrapped.replaceIdentifierWithQualifiedType(identifier, qualifiedType);
446                replacedEntries += replaced;
447                return replaced;
448            }
