215    /**
216     * Constructor.
217     *
218     * @param file the initial location.
219     * @throws IOException
220     * @throws IllegalAccessException
221     * @throws InvocationTargetException
222     */
223    public AbstractCommandLine(File file)
224            throws IOException, IllegalAccessException, InvocationTargetException, ProfileException {
225        if(file == null) {
226            throw new IllegalArgumentException("file cannot be null");
227        }
228        currentDirectory = file;
229
230        modelHandlers = new HashMap<String, ModelHandler>();
231        toBeSaved = new ArrayList<String>();
232
233        // Java profile initialization.
234        rdfCoder = new RDFCoder("./rdfcoder-repository");
235        rdfCoder.setDebug( debug );
236        rdfCoder.registerProfile(JAVA_PROFILE, "com.asemantics.rdfcoder.JavaProfile");
237
238        // Model handler initialization.
239        createModelHandler(DEFAULT_CODE_MODEL);
240        selectedModel = DEFAULT_CODE_MODEL;
241
242        // Init console reader and history.
243        consoleReader = new ConsoleReader();
244        if(!HISTORY_FILE.exists()) {
245            if( ! HISTORY_FILE.createNewFile() ) {
246                throw new IllegalStateException(
247                    String.format("An error occurred while attempting to create the history file %s", HISTORY_FILE)
248                );
249            }
250        }
251        history = new History(HISTORY_FILE);
252        consoleReader.setHistory(history);
253        consoleReader.setUseHistory(true);
254        configureCommandCompletors(consoleReader);
255    }

280        File newDirectory;

291            );

389    // TODO: retrieve the code model used do populate the Object Table and add it to the list of available models.
390    protected ModelHandler createModelHandler(String modelName) throws ProfileException {
391        if(modelHandlers.containsKey(modelName)) {
392            throw new IllegalArgumentException("a model with name " + modelName + " already exists.");
393        }
394
395        Model model = rdfCoder.createModel(modelName);
396        final JavaProfile jprofile = (JavaProfile) model.getProfile(JAVA_PROFILE);
397
398        if (!jreIntializationDone) {
399            initOrLoadJRE(jprofile);
400            jreIntializationDone = true;
401        }
402
403        ModelHandler mh = new ModelHandler(model, jprofile);
404        modelHandlers.put(modelName, mh);
405        return mh;
406    }

437    /**
438     * Performs a SPARQL query on the specified model.
439     *
440     * @param modelName
441     * @param qry
442     * @param ps
443     */
444    protected void performQueryOnModel(String modelName, String qry, PrintStream ps) {
445        if( ! modelHandlers.containsKey(modelName)) {
446            throw new IllegalArgumentException("model with name " + modelName + " doesn't exist.");
447        }
448
449        ModelHandler mh = modelHandlers.get(modelName);
450        CodeModel cm = mh.model.getCodeModelBase();
451        if( ! (cm instanceof SPARQLQuerableCodeModel) ) {
452            throw new IllegalArgumentException("code model " + modelName + " is not a QuerableCodeModel instance.");
453        }
454        SPARQLQuerableCodeModel qcm = (SPARQLQuerableCodeModel) cm;
455        QueryResult qr = null;
456        try {
457            qr = qcm.performQuery(qry);
458            qr.toTabularView(ps);
459        } catch (SPARQLException e) {
460            throw new IllegalArgumentException("Cannot perform SPARQL query.");
461        } finally {
462            if(qr != null) { qr.close(); }
463        }
464    }

617    /**
618     * Loads a list of libraries and reports the result operations on the given PrintStream.
619     *
620     * @param args
621     * @param ps
622     * @throws IOException
623     */
624    protected void loadLibraries(String[] args, PrintStream ps) throws IOException {
625        String libraryName;
626
627        // Validate arguments first.
628        List<Library> libraries = new ArrayList<Library>();
629        for(int i = 0; i < args.length; i += 2) {
630            libraryName = args[i];
631            validLibraryName(libraryName);
632            String resource = args[i + 1];
633            File resourceFile;
634            LibraryType type;
635            if( (resourceFile = validJar(resource)) != null ) {
636                type = LibraryType.JAR_FILE;
637            } else if ( (resourceFile = validSource(resource) ) != null ) {
638                type = LibraryType.SOURCE_DIR;
639            } else if ( (resourceFile = validJavadoc(resource)) != null ) {
640                type = LibraryType.JAVADOC_DIR;
641            } else if ( (resourceFile = validClass(resource)  ) != null ) {
642                type = LibraryType.CLASS_DIR;
643            } else {
644                throw new IllegalArgumentException("Cannot find resource: '" + args[i + 1] + "'");
645            }
646            libraries.add(new Library(libraryName, resourceFile, type));
647        }
648
649        JavaProfile jprofile = getJavaProfile();
650        final List<JStatistics> statistics = new ArrayList<JStatistics>();
651        for(int i = 0; i < libraries.size(); i++) {
652            final Library currentLibrary = libraries.get(i);
653            if( currentLibrary.type == LibraryType.JAR_FILE ) {
654
655                try {
656                    System.out.print("loading " + currentLibrary.location.getAbsolutePath() + " ...");
657                    statistics.add(
658                            jprofile.loadJar( currentLibrary.name, currentLibrary.location.getAbsolutePath() )
659                    );
660                    System.out.println(" done");
661                } catch (Exception e) {
662                    throw new IllegalArgumentException(
663                            String.format("Error while loading JAR: '%s'", currentLibrary.location),
664                            e
665                    );
666                }
667
668            } else if( libraries.get(i).type == LibraryType.SOURCE_DIR ) {
669
670                try {
671                    System.out.format( "loading '%s' ...", currentLibrary.location.getAbsolutePath() );
672                    statistics.add(
673                        jprofile.loadSources( currentLibrary.name, currentLibrary.location.getAbsolutePath() )
674                    );
675                    System.out.println(" done");
676                } catch(Exception e) {
677                    throw new IllegalArgumentException(
678                            String.format("Error while reading SRC dir: '%s'", libraries.get(i).location),
679                            e
680                    );
681                }
682
683            } else if(libraries.get(i).type == LibraryType.JAVADOC_DIR ) {
684
685                    try {
686                        System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
687                        statistics.add(
688                            jprofile.loadJavadoc( libraries.get(i).name, libraries.get(i).location.getAbsolutePath() )
689                        );
690                        System.out.println(" done");
691                    } catch (Exception e) {
692                        throw new IllegalArgumentException(
693                              "Error while reading JAVADOC dir: '" + libraries.get(i).location + "'", e
694                        );
695                    }
696
697            } else if(libraries.get(i).type == LibraryType.CLASS_DIR ) {
698
699                try {
700                    System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
701//                        classDirectoryParser.parseDirectory( libraries.get(i).name, libraries.get(i).location );
702                    statistics.add(
703                        jprofile.loadClasses( currentLibrary.name, currentLibrary.location.getAbsolutePath() )
704                    );
705                    System.out.println(" done");
706                } catch (Exception e) {
707                    throw new IllegalArgumentException(
708                            String.format("Error while reading CLASS dir: '%s'", libraries.get(i).location),
709                            e
710                    );
711                }
712            } else {
713                throw new IllegalStateException(
714                        String.format("Cannot load library: '%s'", libraries.get(i).name)
715                );
716            }
717        }
718
719        for(JStatistics stats : statistics) {
720            ps.print( stats.toStringReport() );
721        }
722
723        toBeSaved.add(selectedModel);
724    }

792        Map<String,String> parameters = new HashMap<String,String>();
793        String server = retrieveParam(CodeStorage.DB_SERVER, args);
794        if( server == null ) {
795            throw new IllegalArgumentException(
796                    CodeStorage.FS_FILENAME + " or " + CodeStorage.DB_SERVER + " must be specified."
797            );
798        }
799        String port = retrieveParam(CodeStorage.DB_PORT, args);
800        if( port == null ) {
801            throw new IllegalArgumentException(CodeStorage.DB_PORT + " must be specified.");
802        }
803        String name = retrieveParam(CodeStorage.DB_NAME, args);
804        if( name == null ) {
805            throw new IllegalArgumentException(CodeStorage.DB_NAME + " must be specified.");
806        }
807        String username = retrieveParam(CodeStorage.DB_USERNAME, args);
808        if( username == null ) {
809            throw new IllegalArgumentException(CodeStorage.DB_USERNAME + " must be specified.");
810        }
811        String password = retrieveParam(CodeStorage.DB_PASSWORD, args);
812        if( password == null ) {
813            throw new IllegalArgumentException(CodeStorage.DB_PASSWORD + " must be specified.");
814        }
815        parameters.put(CodeStorage.DB_SERVER,   server);
816        parameters.put(CodeStorage.DB_PORT,     port);
817        parameters.put(CodeStorage.DB_NAME,     name);
818        parameters.put(CodeStorage.DB_USERNAME, username);
819        parameters.put(CodeStorage.DB_PASSWORD, password);
820        return parameters;
821    }

823    /**
824     * Processes a command line input.
825     *
826     * @param args
827     * @throws IllegalAccessException
828     * @throws InvocationTargetException
829     * @return returns <code>true</code> if the command has been processed, <code>false</code> otherwise.
830     *
831     */
832    protected boolean processCommand(String[] args) throws IllegalAccessException, InvocationTargetException {
833        if(args.length == 0) {
834            return true;
835        }
836
837        loadCommands();
838
839        String commandName = args[0];
840        String[] commandArgs = new String[args.length -1];
841        int i;
842        for(i = 1; i < args.length; i++) {
843            commandArgs[i - 1] = args[i];
844        }
845
846        final Command command = commands.get(commandName);
847        if(command == null) {
848            throw new IllegalArgumentException( String.format("unknown command: '%s'", commandName) );
849        }
850        final Method target = command.target;
851
852        try {
853            target.invoke(this, (Object) commandArgs);
854        } catch (InvocationTargetException ite) {
855            if (ite.getCause() instanceof IllegalArgumentException) {
856                IllegalArgumentException iae = (IllegalArgumentException) ite.getCause();
857                handleIllegalArgumentException(iae, commandName);
858                return false;
859            } else {
860                throw ite;
861            }
862        }
863        return true;
864    }

879     * @return
880     * @throws IOException
881     */
882    protected boolean processLine(String line) throws IOException {

892            handleGenericException(t);
893        }
894        return true;
895    }

911    /**
912     * Extracts arguments from a command line input.
913     *
914     * @param cl
915     * @return the list of arguments.
916     */
917    protected static String[] extractArgs(String cl) {
918        argsBuffer.clear();
919        cl += " ";
920        boolean insideQuotes = false;
921        int begin = 0;
922        int c;
923        for(c = 0; c < cl.length(); c++) {
924            if( cl.charAt(c) == '"' ) {
925                if(insideQuotes) {
926                    insideQuotes = false;
927                    if(c - begin > 0) {
928                        argsBuffer.add(cl.substring(begin, c));
929                    }
930                } else {
931                    insideQuotes = true;
932                }
933                begin = c + 1;
934            }
935            if(insideQuotes) {
936                if(c == cl.length() - 1) {
937                    throw new IllegalArgumentException("not found quotes closure.");
938                }
939                continue;
940            }
941            if( cl.charAt(c) == ' ' ) {
942                if(c - begin > 0) {
943                    argsBuffer.add( cl.substring(begin, c) );
944                }
945                begin = c + 1;
946            } else if(c == cl.length() - 1 && c - begin > 0) {
947                argsBuffer.add( cl.substring(begin, c + 1) );
948            }
949        }
950        return argsBuffer.toArray(new String[argsBuffer.size()]);
951    }

1053    /**
1054     * Loads all the commands defined in the command line.
1055     */
1056    private void loadCommands() {
1057        if(commands != null) {
1058            return;
1059        }
1060
1061        commands = new HashMap<String, Command>();
1062        Method[] methods = this.getClass().getMethods();
1063        Command command;
1064        try {
1065            for (Method method : methods) {
1066                final String methodName = method.getName();
1067                if (methodName.indexOf(COMMAND_PREFIX) == 0) {
1068                    String commandName = methodName.substring(COMMAND_PREFIX.length());
1069                    command = new Command(
1070                            commandName,
1071                            method,
1072                            extractShortCommandDescription(commandName),
1073                            extractLongCommandDescription(commandName)
1074                    );
1075                    commands.put(commandName, command);
1076                }
1077            }
1078        } catch (Exception e) {
1079            throw new RuntimeException("An error occurred while loading commands.", e);
1080        }
1081    }

1084     * Handles an IllegalArgumentException raised by a command method transforming it
1085     * in a error message.
1086     *
1087     * @param iae
1088     * @param cmd
1089     */
1090    private void handleIllegalArgumentException(IllegalArgumentException iae, String cmd) {
1091        System.out.println("ERROR: " + iae.getMessage() );
1092        if(debug) { iae.printStackTrace(); }
1093        Throwable cause = iae.getCause();
1094        int causeLevel = 0;
1095        while(cause != null) {
1096            System.out.println("[" + (causeLevel++) + "] with cause: '" + iae.getCause().getMessage() + "'");
1097            if(debug) { iae.getCause().printStackTrace(); }
1098            cause = cause.getCause();
1099        }
1100
1101        try {
1102            if(cmd == null) {
1103                printUsage(System.out);
1104            } else {
1105                System.out.println(getLongCommandDescription(cmd));
1106            }
1107        } catch (IllegalAccessException e) {
1108            e.printStackTrace();
1109        } catch (InvocationTargetException e) {
1110            e.printStackTrace();
1111        }
1112    }

1170            for(String m : toBeSaved ) {
1171                System.out.println("\t" + m);

1178                    return false;

1181        }
1182        return true;
