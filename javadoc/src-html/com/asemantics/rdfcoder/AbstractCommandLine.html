215        /**
216         * Constructor.
217         *
218         * @param file the initial location.
219         * @throws IOException
220         * @throws IllegalAccessException
221         * @throws InvocationTargetException
222         */
223        public AbstractCommandLine(File file)
224                throws IOException, IllegalAccessException, InvocationTargetException, ProfileException {
225            if(file == null) {
226                throw new IllegalArgumentException("file cannot be null");
227            }
228            currentDirectory = file;
229    
230            modelHandlers = new HashMap<String, ModelHandler>();
231            toBeSaved = new ArrayList<String>();
232    
233            // Java profile initialization.
234            rdfCoder = new RDFCoder("./rdfcoder-repository");
235            rdfCoder.setDebug( debug );
236            rdfCoder.registerProfile(JAVA_PROFILE, "com.asemantics.rdfcoder.JavaProfile");
237    
238            // Model handler initialization.
239            createModelHandler(DEFAULT_CODE_MODEL);
240            selectedModel = DEFAULT_CODE_MODEL;
241    
242            // Init console reader and history.
243            consoleReader = new ConsoleReader();
244            if(!HISTORY_FILE.exists()) {
245                if( ! HISTORY_FILE.createNewFile() ) {
246                    throw new IllegalStateException(
247                        String.format("An error occurred while attempting to create the history file %s", HISTORY_FILE)
248                    );
249                }
250            }
251            history = new History(HISTORY_FILE);
252            consoleReader.setHistory(history);
253            consoleReader.setUseHistory(true);
254            configureCommandCompletors(consoleReader);
255        }

280            File newDirectory;
281            try {
282                newDirectory = toAbsolutePath( candidateLocation.getPath() ).getCanonicalFile();

288            } else {
289                throw new IllegalArgumentException(

291                );

385         *
386         * @param modelName
387         * @return
388         */
389        // TODO: retrieve the code model used do populate the Object Table and add it to the list of available models.
390        protected ModelHandler createModelHandler(String modelName) throws ProfileException {
391            if(modelHandlers.containsKey(modelName)) {
392                throw new IllegalArgumentException("a model with name " + modelName + " already exists.");
393            }
394    
395            Model model = rdfCoder.createModel(modelName);
396            final JavaProfile jprofile = (JavaProfile) model.getProfile(JAVA_PROFILE);
397    
398            if (!jreIntializationDone) {
399                initOrLoadJRE(jprofile);
400                jreIntializationDone = true;
401            }
402    
403            ModelHandler mh = new ModelHandler(model, jprofile);
404            modelHandlers.put(modelName, mh);
405            return mh;
406        }

437        /**
438         * Performs a SPARQL query on the specified model.
439         *
440         * @param modelName
441         * @param qry
442         * @param ps
443         */
444        protected void performQueryOnModel(String modelName, String qry, PrintStream ps) {
445            if( ! modelHandlers.containsKey(modelName)) {
446                throw new IllegalArgumentException("model with name " + modelName + " doesn't exist.");
447            }
448    
449            ModelHandler mh = modelHandlers.get(modelName);
450            CodeModel cm = mh.model.getCodeModelBase();
451            if( ! (cm instanceof SPARQLQuerableCodeModel) ) {
452                throw new IllegalArgumentException("code model " + modelName + " is not a QuerableCodeModel instance.");
453            }
454            SPARQLQuerableCodeModel qcm = (SPARQLQuerableCodeModel) cm;
455            QueryResult qr = null;
456            try {
457                qr = qcm.performQuery(qry);
458                qr.toTabularView(ps);
459            } catch (SPARQLException e) {
460                throw new IllegalArgumentException("Cannot perform SPARQL query.");
461            } finally {
462                if(qr != null) { qr.close(); }
463            }
464        }

617        /**
618         * Loads a list of libraries and reports the result operations on the given PrintStream.
619         *
620         * @param args
621         * @param ps
622         * @throws IOException
623         */
624        protected void loadLibraries(String[] args, PrintStream ps) throws IOException {
625            String libraryName;
626    
627            // Validate arguments first.
628            List<Library> libraries = new ArrayList<Library>();
629            for(int i = 0; i < args.length; i += 2) {
630                libraryName = args[i];
631                validLibraryName(libraryName);
632                String resource = args[i + 1];
633                File resourceFile;
634                LibraryType type;
635                if( (resourceFile = validJar(resource)) != null ) {
636                    type = LibraryType.JAR_FILE;
637                } else if ( (resourceFile = validSource(resource) ) != null ) {
638                    type = LibraryType.SOURCE_DIR;
639                } else if ( (resourceFile = validJavadoc(resource)) != null ) {
640                    type = LibraryType.JAVADOC_DIR;
641                } else if ( (resourceFile = validClass(resource)  ) != null ) {
642                    type = LibraryType.CLASS_DIR;
643                } else {
644                    throw new IllegalArgumentException("Cannot find resource: '" + args[i + 1] + "'");
645                }
646                libraries.add(new Library(libraryName, resourceFile, type));
647            }
648    
649            JavaProfile jprofile = getJavaProfile();
650            final List<JStatistics> statistics = new ArrayList<JStatistics>();
651            for(int i = 0; i < libraries.size(); i++) {
652                final Library currentLibrary = libraries.get(i);
653                if( currentLibrary.type == LibraryType.JAR_FILE ) {
654    
655                    try {
656                        System.out.print("loading " + currentLibrary.location.getAbsolutePath() + " ...");
657                        statistics.add(
658                                jprofile.loadJar( currentLibrary.name, currentLibrary.location.getAbsolutePath() )
659                        );
660                        System.out.println(" done");
661                    } catch (Exception e) {
662                        throw new IllegalArgumentException(
663                                String.format("Error while loading JAR: '%s'", currentLibrary.location),
664                                e
665                        );
666                    }
667    
668                } else if( libraries.get(i).type == LibraryType.SOURCE_DIR ) {
669    
670                    try {
671                        System.out.format( "loading '%s' ...", currentLibrary.location.getAbsolutePath() );
672                        statistics.add(
673                            jprofile.loadSources( currentLibrary.name, currentLibrary.location.getAbsolutePath() )
674                        );
675                        System.out.println(" done");
676                    } catch(Exception e) {
677                        throw new IllegalArgumentException(
678                                String.format("Error while reading SRC dir: '%s'", libraries.get(i).location),
679                                e
680                        );
681                    }
682    
683                } else if(libraries.get(i).type == LibraryType.JAVADOC_DIR ) {
684    
685                    // TODO: implement this.
686                    throw new UnsupportedOperationException();
687    
688    //                    try {
689    //                        System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
690    //                        javadocDirectoryParser.parseDirectory( libraries.get(i).name, libraries.get(i).location );
691    //                        System.out.println(" done");
692    //                    } catch (Exception e) {
693    //                        throw new IllegalArgumentException(
694    //                              "Error while reading JAVADOC dir: '" + libraries.get(i).location + "'", e
695    //                        );
696    //                    }
697    
698                } else if(libraries.get(i).type == LibraryType.CLASS_DIR ) {
699    
700                    try {
701                        System.out.print("loading " + libraries.get(i).location.getAbsolutePath() + " ...");
702    //                        classDirectoryParser.parseDirectory( libraries.get(i).name, libraries.get(i).location );
703                        statistics.add(
704                            jprofile.loadClasses( currentLibrary.name, currentLibrary.location.getAbsolutePath() )
705                        );
706                        System.out.println(" done");
707                    } catch (Exception e) {
708                        throw new IllegalArgumentException(
709                                String.format("Error while reading CLASS dir: '%s'", libraries.get(i).location),
710                                e
711                        );
712                    }
713                } else {
714                    throw new IllegalStateException(
715                            String.format("Cannot load library: '%s'", libraries.get(i).name)
716                    );
717                }
718            }
719    
720            for(JStatistics stats : statistics) {
721                ps.print( stats.toStringReport() );
722            }
723    
724            toBeSaved.add(selectedModel);
725        }

822        /**
823         * Processes a command line input.
824         *
825         * @param args
826         * @throws IllegalAccessException
827         * @throws InvocationTargetException
828         * @return returns <code>true</code> if the command has been processed, <code>false</code> otherwise.
829         *
830         */
831        protected boolean processCommand(String[] args) throws IllegalAccessException, InvocationTargetException {
832            if(args.length == 0) {
833                return true;
834            }
835    
836            loadCommands();
837    
838            String commandName = args[0];
839            String[] commandArgs = new String[args.length -1];
840            int i;
841            for(i = 1; i < args.length; i++) {
842                commandArgs[i - 1] = args[i];
843            }
844    
845            final Command command = commands.get(commandName);
846            if(command == null) {
847                throw new IllegalArgumentException( String.format("unknown command: '%s'", commandName) );
848            }
849            final Method target = command.target;
850    
851            try {
852                target.invoke(this, (Object) commandArgs);
853            } catch (InvocationTargetException ite) {
854                if (ite.getCause() instanceof IllegalArgumentException) {
855                    IllegalArgumentException iae = (IllegalArgumentException) ite.getCause();
856                    handleIllegalArgumentException(iae, commandName);
857                    return false;
858                } else {
859                    throw ite;
860                }
861            }
862            return true;
863        }

881        protected boolean processLine(String line) throws IOException {

892            }
893            return true;
894        }

910        /**
911         * Extracts arguments from a command line input.
912         *
913         * @param cl
914         * @return the list of arguments.
915         */
916        protected static String[] extractArgs(String cl) {
917            argsBuffer.clear();
918            cl += " ";
919            boolean insideQuotes = false;
920            int begin = 0;
921            int c;
922            for(c = 0; c < cl.length(); c++) {
923                if( cl.charAt(c) == '"' ) {
924                    if(insideQuotes) {
925                        insideQuotes = false;
926                        if(c - begin > 0) {
927                            argsBuffer.add(cl.substring(begin, c));
928                        }
929                    } else {
930                        insideQuotes = true;
931                    }
932                    begin = c + 1;
933                }
934                if(insideQuotes) {
935                    if(c == cl.length() - 1) {
936                        throw new IllegalArgumentException("not found quotes closure.");
937                    }
938                    continue;
939                }
940                if( cl.charAt(c) == ' ' ) {
941                    if(c - begin > 0) {
942                        argsBuffer.add( cl.substring(begin, c) );
943                    }
944                    begin = c + 1;
945                } else if(c == cl.length() - 1 && c - begin > 0) {
946                    argsBuffer.add( cl.substring(begin, c + 1) );
947                }
948            }
949            return argsBuffer.toArray(new String[argsBuffer.size()]);
950        }

1052        /**
1053         * Loads all the commands defined in the command line.
1054         */
1055        private void loadCommands() {
1056            if(commands != null) {
1057                return;
1058            }
1059    
1060            commands = new HashMap<String, Command>();
1061            Method[] methods = this.getClass().getMethods();
1062            Command command;
1063            try {
1064                for (Method method : methods) {
1065                    final String methodName = method.getName();
1066                    if (methodName.indexOf(COMMAND_PREFIX) == 0) {
1067                        String commandName = methodName.substring(COMMAND_PREFIX.length());
1068                        command = new Command(
1069                                commandName,
1070                                method,
1071                                extractShortCommandDescription(commandName),
1072                                extractLongCommandDescription(commandName)
1073                        );
1074                        commands.put(commandName, command);
1075                    }
1076                }
1077            } catch (Exception e) {
1078                throw new RuntimeException("An error occurred while loading commands.", e);
1079            }
1080        }

1084         * in a error message.
1085         *
1086         * @param iae
1087         * @param cmd
1088         */
1089        private void handleIllegalArgumentException(IllegalArgumentException iae, String cmd) {
1090            System.out.println("ERROR: " + iae.getMessage() );
1091            if(debug) { iae.printStackTrace(); }
1092            Throwable cause = iae.getCause();
1093            int causeLevel = 0;
1094            while(cause != null) {
1095                System.out.println("[" + (causeLevel++) + "] with cause: '" + iae.getCause().getMessage() + "'");
1096                if(debug) { iae.getCause().printStackTrace(); }
1097                cause = cause.getCause();
1098            }
1099    
1100            try {
1101                if(cmd == null) {
1102                    printUsage(System.out);
1103                } else {
1104                    System.out.println(getLongCommandDescription(cmd));
1105                }
1106            } catch (IllegalAccessException e) {
1107                e.printStackTrace();
1108            } catch (InvocationTargetException e) {
1109                e.printStackTrace();
1110            }
1111        }
