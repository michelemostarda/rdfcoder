019    package com.asemantics.model;
020    
021    import com.asemantics.sourceparse.ObjectsTable;
022    
023    /**
024     * This interface represents a code handler.
025     *
026     * An handler is domething able to receive code analysis
027     * events and use it to perform some analysis.
028     *
029     * @author Michele Mostarda (michele.mostarda@gmail.com)
030     */
031    //TODO: rename in JavaCodeHandler.
032    public interface CodeHandler extends ParseHandler, JavadocHandler, BackTrackingSupport {
033    
034        /**
035         * Expected format of the library datetime.
036         */
037        public static final String LIBRARY_DATETIME_FORMAT = "yyyy_MM_dd__HH_mm_ss";
038    
039        /**
040         * The package separator character.
041         */
042        public static final String PACKAGE_SEPARATOR = ".";
043    
044        /**
045         * Nofifies the begin of a package parsing.
046         * The package is expected to be FULLY QUALIFIED.
047         *
048         * @param pathToPackage the abosoute path to the package.
049         */
050        public void startPackage(String pathToPackage);
051    
052        /**
053         * Notifies the end of a package parsing.
054         */
055        public void endPackage();
056    
057        /**
058         * Notifies the begin of an interface definition.
059         *
060         * @param pathToInterface the pathToInterface of the interface.
061         * @param extendedInterfaces the list of extended interfaces.
062         */
063        public void startInterface(String pathToInterface, String[] extendedInterfaces);
064    
065        /**
066         * Notifies the end of an interface.
067         */
068        public void endInterface();
069    
070        /**
071         * Notifies the begin of a class parsing.
072         * The class is expected to be FULLY QUALIFIED.
073         *
074         * @param modifiers the class modifiers.
075         * @param visibility the class visibility level.
076         * @param pathToClass the fully qualified path to the class.
077         * @param extendedClass the fully qualified pathToClass of the class extended by the starting class.
078         * @param implementedInterfaces the fully qualified names for the interfaces implemented by the starting class.
079         */
080        public void startClass(
081                JavaCodeModel.JModifier[] modifiers,
082                JavaCodeModel.JVisibility visibility,
083                String pathToClass,
084                String extendedClass,
085                String[] implementedInterfaces
086        );
087    
088        /**
089         * Notifies the end of a class parsing.
090         */
091        public void endClass();
092    
093        /**
094         * Notifies the begin of an enumeration class parsing.
095         *
096         * The enumeration is expected to be FULLY QUALIFIED.
097         *
098         * @param modifiers the enumeration modifiers.
099         * @param visibility the visibility level of the enumeration.
100         * @param pathToEnumeration the fully qualified path to the enumeration.
101         * @param elements the list of the elements of the enumeration.
102         */
103        public void startEnumeration(
104                JavaCodeModel.JModifier[] modifiers,
105                JavaCodeModel.JVisibility visibility,
106                String pathToEnumeration,
107                String[] elements
108        );
109    
110        /**
111         * Notifies the end of the enumeration class.
112         */
113        public void endEnumeration();
114    
115        /**
116         * Notifies the parsing of an attribute.
117         *
118         * The attribute is expected to be FULLY QUALIFIED.
119         *
120         * @param modifiers the attribute modifiers.
121         * @param visibility the attribute visibility level.
122         * @param pathToAttribute the absolute path to the attribute.
123         * @param type the type of the attribute.
124         * @param value the init value for the attribute. 
125         */
126        public void attribute(
127                JavaCodeModel.JModifier[] modifiers,
128                JavaCodeModel.JVisibility visibility,
129                String pathToAttribute,
130                JavaCodeModel.JType type,
131                String value
132        );
133    
134        /**
135         * Notifies the parsing of a constructor.
136         *
137         * The length of paramterSize and parameterNames is the same,
138         * the association between these elements is done positionally.
139         *
140         * @param modifiers the constructor modifiers.
141         * @param visibility the visibility level for the constructor.
142         * @param overloadIndex the overload index for the constructor.
143         * @param parameterNames the array of parameter names.
144         * @param parameterTypes the array of parameter types.
145         * @param exceptions exceptions raised by the contructor.
146         *
147         */
148        public void constructor(
149                JavaCodeModel.JModifier[] modifiers,
150                JavaCodeModel.JVisibility visibility,
151                int overloadIndex,
152                String[] parameterNames,
153                JavaCodeModel.JType[] parameterTypes,
154                JavaCodeModel.ExceptionType[] exceptions
155        );
156    
157        /**
158         * Notifies the parsing of a method.
159         *
160         * The method is expected to be FULLY QUALIFIED.
161         *
162         * The length of paramterSize and parameterNames is the same,
163         * the association between these elements is done positionally.
164         *
165         * @param modifiers the method modifiers.
166         * @param visibility the method visibility level.
167         * @param pathToMethod the fully qualified path to the method.
168         * @param overloadIndex the index of the current overload for the method.
169         * @param parameterNames the array of parameter names.
170         * @param parameterTypes the array of parameter types.
171         * @param returnType the return type of the method.
172         * @param exceptions exceptions raised by the method.
173         *
174         */
175        public void method(
176                JavaCodeModel.JModifier[] modifiers,
177                JavaCodeModel.JVisibility visibility,
178                String pathToMethod,
179                int overloadIndex,
180                String[] parameterNames,
181                JavaCodeModel.JType[] parameterTypes,
182                JavaCodeModel.JType returnType,
183                JavaCodeModel.ExceptionType[] exceptions
184        );
185    
186    
187        /**
188         * Collects the unresolved types at the end of the parsing.
189         *
190         * @param unresolvedTypes
191         */
192        public void unresolvedTypes(String[] unresolvedTypes);
193    
194        /**
195         * Preloads the given Objects Table with the objects present in the underlying model.
196         *
197         * @param objectsTable
198         */
199        //TODO: LOW - integrate preloading.
200        public void preloadObjectsFromModel(ObjectsTable objectsTable);
201    
202        /**
203         * Adds an error listener to the code handler.
204         * @param errorListener
205         */
206        public void addErrorListener(ErrorListener errorListener);
207    
208        /**
209         * Removes an error listener from the code handler.
210         * @param errorListener
211         */
212        public void removeErrorListener(ErrorListener errorListener);
213    
214    }
