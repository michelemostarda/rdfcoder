186        private class JDClassData {
187    
188            String classIdentifier;
189    
190            int parentesisCounter = 0;
191    
192            JDClassData(String ci) {
193                classIdentifier = ci;
194            }
195    
196            void increment() {
197                parentesisCounter++;
198            }
199    
200            void decrement() {
201                parentesisCounter--;
202            }
203

205                return parentesisCounter == 0;
206            }

210            }

223        /**
224         * The back tracking buffer size.
225         */
226        private static final int BACKTRACK_BUFFER_SIZE = 3;
227    
228        /**
229         * Contains the list of tokens to be processed.
230         */
231        private class TokensPipe {
232    
233            /**
234             * Internal queue.
235             */
236            private Queue<JDToken> queue;
237    
238            /**
239             * Back tracking buffer.
240             */
241            private List<JDToken> backTrackBuffer = new ArrayList<JDToken>();
242    
243            /**
244             * Constructor.
245             */
246            TokensPipe() {
247                queue = new LinkedList<JDToken>();
248            }
249    
250            /**
251             * Adds a token to the pipe.
252             *
253             * @param token
254             */
255            void addToken(JDToken token) {
256                queue.add(token);
257            }
258    
259            /**
260             * Returns the next token to be processed.
261             *
262             * @return
263             */
264            public JDToken nextToken() {
265                JDToken headToken = queue.poll();
266                if(headToken != null) {
267                    if(backTrackBuffer.size() == BACKTRACK_BUFFER_SIZE) {
268                        backTrackBuffer.remove(0);
269                    }
270                    backTrackBuffer.add(headToken);
271                }
272                return headToken;
273            }
274    
275            /**
276             * Returns the token with backtrack <i>b</i>.
277             * @param b
278             * @return
279             */
280            public JDToken backtrack(int b) {
281                if(b <= 0) {
282                    throw new IllegalArgumentException();
283                }
284                if(b > backTrackBuffer.size()) {
285                    return null;
286                }
287                return backTrackBuffer.get(b - 1);
288            }
289    
290            public void clear() {
291                queue.clear();
292                backTrackBuffer.clear();
293            }
294        }

385        /**
386         * Parses a compilation unit.
387         * 
388         * @param inputStream
389         * @param compilationUnitPath
390         * @throws ParserException
391         */
392        public void parse(InputStream inputStream, String compilationUnitPath) throws ParserException {
393    
394            final JavadocHandler javadocHandler = (JavadocHandler) getParseHandler();
395    
396            tokensPipe.clear();
397            classes.clear();
398            methodName = null;
399            attribute = null;
400            entryText    .delete(0, entryText.length());
401            attributeText.delete(0, attributeText.length());
402            attributesMap.clear();
403            parameters   .clear();
404            lastEntry = null;
405    
406            try {
407                getParseHandler().startCompilationUnit(compilationUnitPath);
408            } catch (Throwable t) {
409                t.printStackTrace();
410            }
411    
412            JDToken token;
413            try {
414    
415                // Main cycle.
416                while( true ) {
417    
418                    // Extract tokens from imput stream.
419                    extractTokens(inputStream, tokensPipe);
420    
421                    // Check if end of file has been reached.
422                    if( (token = tokensPipe.nextToken() ) == JD_EOF) {
423                        break;
424                    }
425    
426                    // Tokens to be processed when inside Javadoc block.
427                    if(insideComment) {
428    
429                        // Closing Javadoc, the entity is emitted.
430                        if(token == JD_CLOSE_COMMENT) {
431    
432                            // Stores the last attribute if any.
433                            storeCurrentAttribute();
434    
435                            insideComment = false;
436                            String entryText = this.entryText.toString();
437                            int splitPoint = entryText.indexOf(".");
438                            String shortComment, longComment;
439                            if(splitPoint != -1) {
440                                shortComment = entryText.substring(0, splitPoint).trim();
441                                longComment = entryText.substring(splitPoint + 1).trim();
442                            } else {
443                                shortComment = entryText;
444                                longComment  = "";
445                            }
446    
447                            // Creates last entry data.
448                            JavadocEntry entry = new JavadocEntry(
449                                    shortComment,
450                                    longComment,
451                                    new HashMap<String,List<String>>(attributesMap),
452                                    lastOpenComment.getRow(),
453                                    lastOpenComment.getCol()
454                            );
455                            lastEntry = entry;
456    
457                            // Notifies last entry to listener.
458                            try {
459                                // A Javadoc entry has been found.
460                                javadocHandler.parsedEntry(entry);
461                            } catch(Throwable t) {
462                                t.printStackTrace();
463                            }
464    
465                        // Process attribute.
466                        } else if( token instanceof JDAttribute) {
467    
468                            // Storing previous attribute.
469                            storeCurrentAttribute();
470    
471                            attribute = token.getValue();
472                            attributeText.delete(0, attributeText.length());
473    
474                        // Process attribute words.
475                        } else if( attribute != null && token instanceof JDWord ) {
476    
477                            attributeText.append(token.getValue());
478    
479                        // Process entry comment words.
480                        } else if( token instanceof JDWord ) {
481    
482                            entryText.append(token.getValue());
483    
484                        }
485    
486                    // Tokens to be processed when outside javadoc blocks.
487                    } else {
488    
489                        // Javadoc commend opening.
490                        if( token instanceof JDOpenComment ) {
491    
492                            //System.out.println("OPEN COMMENT!!");
493    
494                            lastOpenComment = (JDOpenComment) token;
495                            insideComment = true;
496                            entryText.delete(0, entryText.length());
497                            attributesMap.clear();
498                            attributeText.delete(0, attributeText.length());
499    
500                        // Class declaration beginning.
501                        } else if(token instanceof JDClassDeclarator) {
502    
503                            classExpected = true;
504    
505                        // Class signature handling.
506                        } else if(classExpected && token instanceof JDWord) {
507    
508    
509                            classExpected = false;
510                            classes.push( new JDClassData( token.getValue() ) );
511                            if(lastEntry != null) {
512                                String containerPath = getContainerPath();
513                                try {
514                                    javadocHandler.classJavadoc( lastEntry, containerPath );
515                                } catch(Throwable t) {
516                                    t.printStackTrace();
517                                }
518                                lastEntry = null;
519                            }
520    
521                        // Opening class or method.
522                        } else if( ! classes.isEmpty() && token == JD_OPEN_CLASS_OR_METHOD) {
523    
524                            classes.peek().increment();
525    
526                        // Closing class or method.
527                        } else if(  ! classes.isEmpty() && token == JD_CLOSE_CLASS_OR_METHOD) {
528    
529                            classes.peek().decrement();
530                            if(classes.peek().endOfClass()) {
531                                classes.pop();
532                            }
533    
534                        // Begin parameters list.
535                        } else if(token == JD_OPEN_PARAMETERS) {
536    
537                            parameters.clear();
538                            methodName = tokensPipe.backtrack(2).getValue();
539    
540                        // Ends parameters list.
541                        } else if(token == JD_CLOSE_PARAMETERS) {
542    
543                            // Adds last parameter.
544                            parameters.add( new JDParameter( tokensPipe.backtrack(1).getValue(), tokensPipe.backtrack(2).getValue() ) );
545    
546                            // Notifies to the listener the method javadoc recognition.
547                            if(lastEntry != null) {
548                                String containerPath = getContainerPath();
549                                String pathToMethod    = (containerPath.equals("") ? "" : containerPath + CodeHandler.PACKAGE_SEPARATOR) + methodName;
550                                String[] signature   = toSignature(parameters);
551                                try {
552                                    javadocHandler.methodJavadoc(lastEntry, pathToMethod, signature);
553                                } catch (Throwable t) {
554                                    t.printStackTrace();
555                                }
556                                methodName = null;
557                                lastEntry  = null;
558                            }
559    
560                        // Parameters separator.
561                        } else if(token == JD_PARAMETERS_SEPARATOR) {
562    
563                            parameters.add( new JDParameter( tokensPipe.backtrack(1).getValue(), tokensPipe.backtrack(2).getValue() ) );
564    
565                        }
566    
567                    }
568                }
569    
570                for(JDClassData clazz : classes) {
571                    System.out.println(clazz);
572                }
573    
574            } catch(IOException ioe) {
575                throw new ParserException(ioe, compilationUnitPath);
576            }
577    
578            try {
579                getParseHandler().endCompilationUnit();
580            } catch (Throwable t) {
581                t.printStackTrace();
582            }
583        }

585        /**
586         * Parses a file.
587         * @param file
588         * @throws ParserException
589         */
590        public void parse(File file) throws ParserException {
591            FileInputStream fileInputStream = null;
592            try {
593                fileInputStream = new FileInputStream(file);
594                parse(fileInputStream, file.getAbsolutePath());
595                fileInputStream.close();
596            } catch (FileNotFoundException fnfe) {
597                throw new ParserException(fnfe, file.getAbsolutePath());
598            } catch (IOException ioe) {
599                throw new ParserException(ioe, file.getAbsolutePath());
600            }
601        }

603        /**
604         *
605         * Tokenizer: extracts token from the given <i>char stream</i>.
606         *
607         * @param inputStream
608         * @param tokensPipe
609         * @throws IOException
610         */
611        private void extractTokens(InputStream inputStream, TokensPipe tokensPipe) throws IOException {
612            int intc;
613            char c;
614    
615            tokenBuffer.delete(0, tokenBuffer.length());
616            while( (intc = inputStream.read()) != -1 ) {
617                c = (char) intc;
618    
619                // Manage row / col position.
620                if(c == '\n') {
621                    col = 0;
622                    row++;
623                } else {
624                    col++;
625                }
626    
627                // Manage special characters.
628                if(c == ' ' || c == '\n' || c == '{' || c == '}' || c == '(' || c == ')' || c == ',') {
629    
630                    // Tokens.
631    
632                    String bufferContent = tokenBuffer.toString();
633                    //System.out.println("BUFFER CONTENT: '" + bufferContent + "'");
634    
635                    // Handles inside <i>*</i> comment.
636                    if("*".equals(bufferContent)) {
637                        tokenBuffer.delete(0, tokenBuffer.length());
638                        continue;
639                    } 
640    
641                    // Handles begin comment.
642                    if("/**".equals(bufferContent.trim())) {
643                        tokensPipe.addToken( new JDOpenComment(row, col) );
644                        //System.out.println("OPEN COMMENT TOKEN");
645                        return;
646                    }
647    
648                    // Handles end comment.
649                    if("*/".equals(bufferContent)) {
650                        tokensPipe.addToken( JD_CLOSE_COMMENT );
651                        return;
652                    }
653    
654                    // Handles class key.
655                    if("class".equals(bufferContent) ) {
656                        tokensPipe.addToken( new JDClassDeclarator(row, col) );
657                        return;
658                    }
659    
660                    // Handles attribute.
661                    if(bufferContent.length() > 1 && bufferContent.charAt(0) == '@') {
662                        tokensPipe.addToken( new JDAttribute(bufferContent) );
663                        return;
664                    }
665    
666                    // Handles word separators.
667                    //System.out.println("ADDTOKEN: '" + bufferContent + " '");
668                    tokensPipe.addToken( new JDWord(bufferContent + " ") );
669    
670                    if( c == '{' ) {
671                        tokensPipe.addToken(JD_OPEN_CLASS_OR_METHOD);
672                    } else if(c == '}') {
673                        tokensPipe.addToken(JD_CLOSE_CLASS_OR_METHOD);
674                    } else if(c == '(') {
675                        tokensPipe.addToken(JD_OPEN_PARAMETERS);
676                    } else if(c == ')') {
677                        tokensPipe.addToken(JD_CLOSE_PARAMETERS);
678                    } else if(c == ',') {
679                        tokensPipe.addToken(JD_PARAMETERS_SEPARATOR);
680                    }
681                    return;
682    
683                } else {
684                    tokenBuffer.append(c);
685                }
686            }
687            tokensPipe.addToken( JD_EOF );
688            return;
689        }

691        /**
692         * Stores the current attribute.
693         */
694        private void storeCurrentAttribute() {
695            if(attribute != null) {
696                String key = attribute.trim();
697                //System.out.println("key:" + key);
698                List<String> list = attributesMap.get(key);
699                if(list == null) {
700                    list = new ArrayList<String>();
701                    attributesMap.put(key, list);
702                }
703                //System.out.println("list " + key + "::" + list);
704                list.add(attributeText.toString().trim());
705            }
706        }

708        /**
709         * Returns the container path.
710         *
711         * @return
712         */
713        private String getContainerPath() {
714            return CoderUtils.join(classes, CoderUtils.PACKAGE_SEPARATOR);
715        }

717        /**
718         * Converts the parameters list to a method signature.
719         *
720         * @param parameters
721         * @return
722         */
723        private String[] toSignature(List<JDParameter> parameters) {
724            String[] signature = new String[parameters.size()];
725            for(int i = 0; i < parameters.size(); i++) {
726                signature[i] = parameters.get(i).name;
727            }
728            return signature;
729        }
