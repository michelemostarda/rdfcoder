033    /**
034     * This class defines a table to resolve objects.
035     *
036     * //TODO: LOW - implement serialization / deserialization of UnresolvedFileEntry(es). 
037     */
038    public class ObjectsTable implements Serializable {
039    
040        private final int BLOCK_SIZE = 100;
041    
042        /**
043         * Contains in a memory optimized way the
044         * list of type entries inside a package.
045         */
046        protected class PackageEntry implements Serializable {
047    
048            String[] objects = new String[BLOCK_SIZE];
049    
050            int index = 0;
051    
052            void add(String objectName) {
053                if( contains(objectName) ) {
054                    return;
055                }
056    
057                if( objects.length == index ) {
058                    String[] newObjects = new String[objects.length + BLOCK_SIZE];
059                    System.arraycopy(objects, 0, newObjects, 0, objects.length);
060                    objects = newObjects;
061                }
062                objects[index++] = objectName;
063            }
064    
065            /**
066             * Returns <code>true</code> if the package contains the object,
067             * <code>false</code> otherwise.
068             * @param objectName
069             * @return
070             */
071            boolean contains(String objectName) {
072                for(int c = 0; c < index; c++) {
073                    if(objects[c].equals(objectName)) {
074                        return true;
075                    }
076                }
077                return false;
078            }
079        }
080    
081        /**
082         * Contains the name of the unresolved type
083         * and the relative import context.
084         */
085        protected class UnresolvedTypeEntry implements Serializable {
086    
087            /**
088             * The name of the unresolved type.
089             */
090            String typeName;
091    
092            /**
093             * The usage done for the unresolved entry.
094             */
095            JavaCodeModel.JType type;
096    
097            /**
098             * The context of the imports in which the
099             * type is defined.
100             */
101            ImportsContext importsContext;
102    
103            /**
104             * The temporary identifier currently representing the type.
105             */
106            UnresolvedTypeEntry(String tn, JavaCodeModel.JType t, ImportsContext ic) {
107                typeName            = tn;
108                type                = t;
109                importsContext      = ic;
110            }
111        }
112    
113        /**
114         * This map contains as keys the package prefixes and as
115         * values package entries.
116         */
117        private Map<String, PackageEntry> packagesToContents;
118    
119        /**
120         * This map contains as keys the unresolved type names
121         * and as values unresolved type entries.
122         */
123        private List<UnresolvedTypeEntry> unresolvedTypeEntries;
124    
125        public ObjectsTable() {
126            packagesToContents = new HashMap<String, PackageEntry>();
127            unresolvedTypeEntries = new ArrayList<UnresolvedTypeEntry>();
128        }
129    
130        /**
131         * Adds an object to the objects table.
132         * @param objectPackage the package containing the object.
133         * @param objectName the object name without qualification.
134         */
135        public void addObject(String objectPackage, String objectName) {
136            if(
137                objectPackage == null
138                    ||
139                objectName    == null || objectName.trim().length()    == 0
140            ) {
141                throw new IllegalArgumentException();
142            }
143            if( objectName.indexOf(CodeHandler.PACKAGE_SEPARATOR) != -1 ) {
144                throw new IllegalArgumentException("The object name must be NOT qualified.");
145            }
146    
147            //System.out.println("objectPackage: " + objectPackage);
148            PackageEntry pe = packagesToContents.get(objectPackage);
149            if(pe == null) {
150                pe = new PackageEntry();
151                packagesToContents.put(objectPackage, pe);
152            }
153            pe.add(objectName);
154        }
155    
156        /**
157         * Adds a fully qualified object to the objects table.
158         * @param fullyQualifiedName the fully qualified name.
159         */
160        public void addObject(String fullyQualifiedName) {
161            if(fullyQualifiedName == null) {
162                throw new IllegalArgumentException("fullyQualifidName cannot be null.");
163            }
164    
165            int i = fullyQualifiedName.lastIndexOf(CodeHandler.PACKAGE_SEPARATOR);
166            if(i == -1) {
167                addObject("", fullyQualifiedName); // Default package.
168            } else {
169                addObject( fullyQualifiedName.substring(0, i), fullyQualifiedName.substring(i + 1 ) );
170            }
171        }
172    
173        /**
174         * Checks if an object name is defined inside a package.
175         *
176         * @param objectPackage the package containing the object.
177         * @param type the object type.
178         * @return <code>true</code> if found.
179         */
180        public boolean checkObject(String objectPackage, String type) {
181            PackageEntry pe = packagesToContents.get(objectPackage);
182            if(pe != null) {
183                return pe.contains(type);
184            }
185            return false;
186        }
187    
188        /**
189         * Registes a type as unresolved.
190         *
191         * @param type the unresolved type name.
192         * @param ic context of imports.
193         */
194        public void addUnresolvedType(String tn, JavaCodeModel.JType type, ImportsContext ic) {
195            if(tn == null || tn.trim().length() == 0 || type == null || ic == null) {
196                throw new IllegalArgumentException();
197            }
198    
199            unresolvedTypeEntries.add( new UnresolvedTypeEntry(tn, type, ic) );
200        }
201    
202        /**
203         * @return the list of unresolved types.
204         */
205        public UnresolvedTypeEntry[] getUnresolvedTypeEntries() {
206            return unresolvedTypeEntries.toArray( new UnresolvedTypeEntry[unresolvedTypeEntries.size()] );
207        }
208    
209        /**
210         * @return the number of unresolved types.
211         */
212        public int unresolvedCount() {
213            return unresolvedTypeEntries.size();
214        }
215    
216        /**
217         * Processes all the unresolved identifiers.
218         *
219         * @param ch the code handler user to process the identifiers.
220         */
221        public Set<String> processTemporaryIdentifiers(CodeHandler ch) {
222            Iterator<UnresolvedTypeEntry> uei = unresolvedTypeEntries.iterator();
223            UnresolvedTypeEntry ue;
224            String qualifiedType;
225            Set<String> definitivelyUnresolved = new HashSet();
226            while(uei.hasNext()) {
227                ue = uei.next();
228                qualifiedType = ue.importsContext.qualifyType(this, ue.typeName);
229                String prefixedIdentifier = ue.type.getIdentifier();
230                if(qualifiedType != null) {
231                    ch.replaceIdentifierWithQualifiedType(prefixedIdentifier, qualifiedType);
232                    uei.remove();
233                } else {
234                    ch.replaceIdentifierWithQualifiedType(
235                            prefixedIdentifier,
236                            JavaCodeModel.UNQUALIFIED_PREFIX + ue.typeName
237                    );
238                    definitivelyUnresolved.add( ue.typeName );
239                }
240            }
241            return definitivelyUnresolved;
242        }

244        /**
245         * Clears the content of the object table.
246         */
247        public void clear() {
248            packagesToContents.clear();
249            unresolvedTypeEntries.clear();
250        }

252        /**
253         * Preloads the Objects table with the object symbols found into the sourceDir directory.
254         *
255         * @param sourceDir source directory.
256         */
257        public void preloadSourceDir(File sourceDir) {
258            if(sourceDir == null || ! sourceDir.exists() ) {
259                throw new IllegalArgumentException();
260            }
261    
262            Stack<File> stack = new Stack<File>();
263            stack.push(sourceDir);
264            recursivePreload(
265                    new CoderUtils.JavaSourceFilenameFilter(),
266                    sourceDir.getAbsolutePath().length() + 1,
267                    CoderUtils.JAVA_SOURCE_FILE_EXT.length(),
268                    stack
269            );
270        }
271    
272        /**
273         * Preloads the Objects Table with the object symbols found into the class directory.
274         *
275         * @param classDir
276         */
277        public void preloadClassDir(File classDir) {
278            if(classDir == null || ! classDir.exists() ) {
279                throw new IllegalArgumentException();
280            }
281    
282            Stack<File> stack = new Stack<File>();
283            stack.push(classDir);
284            recursivePreload(
285                    new CoderUtils.JavaClassFilenameFilter(),
286                    classDir.getAbsolutePath().length() + 1,
287                    CoderUtils.JAVA_SOURCE_CLASS_EXT.length(),
288                    stack
289            );
290        }
291    
292        /**
293         * Preloads he Object Table with the objects defined in the given jar.
294         *
295         * @param jar
296         * @throws IOException
297         */
298        public void preloadJar(File jar) throws IOException {
299            if(jar == null || ! jar.exists()) {
300                throw new IllegalArgumentException();
301            }
302    
303            JarFile jarFile = new JarFile(jar);
304            try {
305                Enumeration<JarEntry> entries = jarFile.entries();
306                JarEntry entry;
307                while(entries.hasMoreElements()) {
308                    entry = entries.nextElement();
309                    if(entry.isDirectory()) { continue; }  // is not a file.
310                    String fullyQualifiedObject = entry.getName();
311                    if( fullyQualifiedObject.indexOf(CoderUtils.JAVA_SOURCE_CLASS_EXT) == -1 ) { continue; } // is not a .class
312                    fullyQualifiedObject = fullyQualifiedObject.substring(0, fullyQualifiedObject.length() - CoderUtils.JAVA_SOURCE_CLASS_EXT.length());
313                    fullyQualifiedObject = fullyQualifiedObject.replaceAll("/", CodeHandler.PACKAGE_SEPARATOR);
314                    System.out.println("Add object: " + fullyQualifiedObject);
315                    addObject(fullyQualifiedObject);
316                }
317            } finally {
318                jarFile.close();
319            }
320        }
321    
322        public void preloadClasspath(ClassPath classPath) throws IOException {
323            if(classPath == null) {
324                throw new IllegalArgumentException();
325            }
326    
327            for(File sourceDir : classPath.sourceDirs) {
328                preloadSourceDir(sourceDir);
329            }
330            for(File classDir : classPath.classDirs) {
331                preloadClassDir(classDir);
332            }
333            for(File jarFile : classPath.jarFiles) {
334                preloadJar(jarFile);
335            }
336        }
337    
338        /**
339         * Loads the content of an object table into another.
340         *
341         * @param other
342         */
343        public void load(ObjectsTable other) {
344            if( other == null ) {
345                throw new NullPointerException();
346            }
347    
348            packagesToContents.putAll   ( other.packagesToContents    );
349            unresolvedTypeEntries.addAll( other.unresolvedTypeEntries );
350        }
351    
352        private void recursivePreload(final FilenameFilter ff, final int rootLength, final int extSize, Stack<File> stack) {
353            if(stack.isEmpty()) { return; }
354    
355            File current = stack.pop();
356    
357            File[] all = current.listFiles();
358            for(File f : all) {
359                if( f.isDirectory() &&  ! f.isHidden() ) {
360                    stack.push(f);
361                }
362            }
363    
364            File[] list = current.listFiles(ff);
365            for(File f : list) {
366                String fullyQualifiedObject = f.getAbsolutePath();
367                fullyQualifiedObject = fullyQualifiedObject.substring(rootLength, fullyQualifiedObject.length() - extSize);
368                fullyQualifiedObject = fullyQualifiedObject
369                                            .replaceAll("/"   , CodeHandler.PACKAGE_SEPARATOR)
370                                            .replaceAll("\\\\", CodeHandler.PACKAGE_SEPARATOR)
371                                            .replaceAll("\\$" , CodeHandler.PACKAGE_SEPARATOR); // Inner classes replacing.
372                System.out.println("Add object: " + fullyQualifiedObject);
373                addObject(fullyQualifiedObject);
374            }
375    
376            recursivePreload(ff, rootLength, extSize, stack);
377        }
