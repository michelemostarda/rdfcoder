042         * Kinds of resources.

044        public enum ResourceType {
045            BINARY {
046    
047                String getExtension() {
048                    return "bin";
049                }
050    
051            },
052    
053            XML {
054    
055                String getExtension() {
056                    return "xml";
057                }
058            };
059    
060            /**
061             * Returns the type of the file name on the basis of the extension,
062             * <code>null</code> if no extension matches.
063             *
064             * @param fileName
065             * @return
066             */
067            static ResourceType getType(String fileName) {
068                for(ResourceType rt : ResourceType.values()) {
069                    if(fileName.indexOf(rt.getExtension()) != -1 ) {
070                        return rt;
071                    }
072                }
073                return null;
074            }
075    
076            /**
077             * Returns the name of the resource removing the extension.
078             *
079             * @param file
080             * @return
081             */
082            static String getResourceName(String file) {
083                int separator = file.indexOf(".");
084                return file.substring(0, separator);
085            }
086    
087            /**
088             * Returns the extension associated to the type.
089             * @return
090             */
091            abstract String getExtension();
092    
093            /**
094             * Returns the file name from the resource name.
095             *
096             * @param resourceName
097             * @return
098             */
099            String getFileName( String resourceName ) {
100                return resourceName + "."  + getExtension();
101            }

105        /**
106         * Resource element.
107         */
108        public class Resource {
109    
110            /**
111             * parent directory containing this resource.
112             */
113            private File root;
114    
115            /**
116             * The resource name.
117             */
118            private String name;
119    
120            /**
121             * The resource type.
122             */
123            private ResourceType type;
124    
125            /**
126             * List of readers of this resource.
127             */
128            private List<OutputStream> writers;
129    
130            /**
131             * List of writers of this resource.
132             */
133            private List<InputStream> readers;
134    
135    
136    
137            /**
138             * Constructor.
139             *
140             * @param n
141             * @param rt
142             */
143            Resource(File r, String n, ResourceType rt) {
144                root = r;
145                name = n;
146                type = rt;
147    
148                writers = new ArrayList<OutputStream>();
149                readers = new ArrayList<InputStream>();
150            }
151    
152            /**
153             * Returns the resource name.
154             *
155             * @return the resource name.
156             */
157            public String getName() {
158                return name;
159            }
160    
161            /**
162             * Returns the resource type.
163             * 
164             * @return the type of the resource.
165             */
166            public ResourceType getType() {
167                return type;
168            }
169    
170            /**
171             * Returns the resource location.
172             *
173             * @return the resource location.
174             */
175            public File getLocation() {
176                return new File(root, name + "." + type.getExtension());
177            }
178    
179            /**
180             * Opens an input stream on the resource.
181             *
182             * @return the input stream to read the resource.
183             * @throws FileNotFoundException
184             */
185            public InputStream getInputStream() throws FileNotFoundException, RepositoryException {
186                synchronized (lock) {
187                    if(writers.size() > 0) {
188                        throw new RepositoryException("Cannot open resource in read mode: there is at least a reader on it");
189                    }
190    
191                    // Check there is a single writer.
192                    if(readers.size() > 0) {
193                        throw new RepositoryException("Resource '" + name + "' already accessed in write mode.");
194                    }
195    
196                    InputStream is = new FileInputStream( getLocation() ) {
197        
198                        /**
199                         * Releases the lock file.
200                         *
201                         * @throws IOException
202                         */
203                        public void close() throws IOException {
204                            super.close();
205                            readers.remove(this);
206                        }
207    
208                    };
209                    readers.add(is);
210                    return is;
211                }
212            }
213    
214            /**
215             * Opens an output stream on the resource.
216             *
217             * @param append append flag
218             * @return the output stream to write the resource.
219             * @throws FileNotFoundException if the resource file cannot be found or opened.
220             */
221            public OutputStream getOutputStream(boolean append) throws FileNotFoundException, RepositoryException {
222                synchronized (lock) {
223                    if(writers.size() > 0) {
224                        throw new RepositoryException("Cannot open resource in write mode: there is anoter writer on it");
225                    }
226                    if(readers.size() > 0) {
227                        throw new RepositoryException("Cannot open resource in write mode: there is al least a reader on it");
228                    }
229    
230                    // Creates a lock file for write mode.
231                    try {
232                        createLockFile(name);
233                    } catch (IOException e) {
234                        throw new RepositoryException(e);
235                    }
236    
237                    OutputStream os = new FileOutputStream( getLocation(), append ) {
238    
239                        public void close() throws IOException {
240                            super.close();
241                            writers.remove(this);
242                            removeLockFile(name);
243                            
244                        }
245    
246                    };
247                    writers.add(os);
248                    return os;
249                }
250            }
251    
252            /**
253             *  Opens an output stream on the resource.
254             *
255             * @return the resource output stream.
256             * @throws FileNotFoundException
257             * @throws RepositoryException
258             */
259            public OutputStream getOutputStream() throws FileNotFoundException, RepositoryException {
260                return getOutputStream(false);
261            }
262    
263            /**
264             * Deletes the current resource closing first all input and output streams.
265             */
266            public void delete() {
267                for(InputStream is : readers) {
268                    try {
269                        is.close();
270                    } catch (Exception e) {}
271                }
272                for(OutputStream os : writers) {
273                    try {
274                        os.close();
275                    } catch (Exception e) {}
276                }
277                getLocation().delete();
278    
279            }
280        }

282        /**
283         * Filters all lock files inside a dir.
284         */
285        public static class LockFileFilter implements FilenameFilter {
286            public boolean accept(File dir, String name) {
287                File file = new File(dir, name);
288                if( file.isDirectory() ) { return false; }
289                String postfix = LOCK_FILE_EXT;
290                int lastIndexOf = name.lastIndexOf(postfix);
291                return lastIndexOf > 0 && lastIndexOf == name.length() - postfix.length();
292            }
293        }

357        /**
358         * Loads the resources defined in the <i>target</i>
359         * dir.
360         *
361         * @return the map of loaded resources.
362         */
363        protected Map<String,Resource> loadResources(File target) {
364            if( ! target.exists() ) {
365                throw new IllegalArgumentException("target dir doesn't exist: '" + target.getAbsolutePath() + "'");
366            }
367            if( ! target.isDirectory() ) {
368                throw new IllegalArgumentException("target dir must be a directory");
369            }
370    
371            File[] files = target.listFiles();
372            Map<String,Resource> result = new HashMap<String,Resource>(files.length);
373            ResourceType rt;
374            String fileName;
375            String resourceName;
376            for(File file : files) {
377                if( file.isFile() ) {
378                    fileName = file.getName();
379                    rt = ResourceType.getType(fileName);
380                    if( rt != null ) {
381                        resourceName = ResourceType.getResourceName(fileName);
382                        result.put( resourceName, new Resource(target ,resourceName , rt) );
383                    }
384                }
385            }
386            return result;
387        }

389        /**
390         * Checks the repository folder.
391         *
392         * @param location
393         * @return <code>true</code> if the repo folder has been created,
394         * <code>false</code> otherwise.
395         */
396        protected static boolean checkRepository(File location) throws RepositoryException {
397            if( ! location.exists() ) {
398                if( ! location.mkdir() ) {
399                    throw new RepositoryException(
400                            "Error while creating repository dir at location '" + location.getAbsolutePath() + "'."
401                    );
402                }
403                return true;
404            } else if( ! location.isDirectory() ) {
405                throw new RepositoryException(
406                        "A file with the same name of the repository already exists at location '"+ location.getAbsolutePath() + "'."
407                );
408            }
409            return false;
410        }
411    
412        /**
413         * Singleton restriction constructor.
414         */
415        private Repository(File rl) throws RepositoryException {
416            repositoryLocation = rl;
417            if( ! checkRepository(rl) ) {
418                // Performs a cleanup of repository.
419                removeLockedResources();
420                // Loads existing resources.
421                resources = loadResources(repositoryLocation);
422            } else {
423                resources = new HashMap<String,Resource>();
424            }
425        }

454            return resources.containsKey(resourceName);
455        }

457        /**
458         * Returns an existing resource.
459         *
460         * @param resourceName
461         * @return the resource found.
462         * @throws RepositoryException if the resource doens't exist.
463         */
464        public Resource getResource(String resourceName) throws RepositoryException {
465            Resource resource = resources.get(resourceName);
466            if(resource == null) {
467                throw new RepositoryException("Required resource '" + resourceName + "' doesn't exist.");
468            }
469            return resource;
470        }

472        /**
473         * Removes the specified resource.
474         *
475         * @param resourceName
476         * @throws RepositoryException if the resource cannot be found.
477         */
478        public void removeResource(String resourceName) throws RepositoryException {
479            Resource resource = getResource(resourceName);
480            if( ! resource.getLocation().delete() ) {
481                throw new RepositoryException("Cannot delete resource: '" + resourceName + "'");
482            }
483            try {
484                removeLockFile(resourceName);
485            } catch(Exception e) {
486                e.printStackTrace();
487            }
488            resources.remove(resourceName);
489        }

491        /**
492         * Check if the resource name is valid.
493         *
494         * @param rn
495         */
496        static void checkResourceName(String rn) throws RepositoryException {
497            if(rn == null || rn.trim().length() == 0) {
498                throw new RepositoryException("Invalid resource name: '"+ rn + "'");
499            }
500        }

502        /**
503         * Creates a new resource into the repository.
504         *
505         * @param resourceName
506         * @param type
507         * @return the created resource.
508         * @throws RepositoryException if a resurce with the same name exists or the file cannot be created.
509         */
510        public Resource createResource(String resourceName, ResourceType type) throws RepositoryException {
511            checkResourceName(resourceName);
512    
513            if( containsResource(resourceName) ) {
514                throw new RepositoryException(
515                        "Cannot create resource '" + resourceName + "', a resouce with the same name already exists."
516                );
517            }
518    
519            File file    = new File(repositoryLocation, type.getFileName(resourceName));
520            try {
521                file.createNewFile();
522            } catch (IOException e) {
523                throw new RepositoryException("Cannot create file: '" + file.getAbsolutePath() + "'");
524            }
525    
526            Resource resource = new Resource(repositoryLocation, resourceName, type);
527            resources.put(resourceName, resource);
528    
529            return resource;
530        }

532        /**
533         * Returns <code>true</code> if a resource is locked in writing mode,
534         * <code>false</code> otherwise.
535         *
536         * @param resourceName
537         * @return <code>true</code> if the resource is locked.
538         * @throws RepositoryException is resourceName doesn't exist.
539         */
540        public boolean isLocked(String resourceName) throws RepositoryException {
541            if( resources != null && ! containsResource(resourceName) ) {
542                throw new RepositoryException("Repository doesn't contain resource name '" + resourceName + "'");
543            }
544    
545            File resourceFileLock = new File(repositoryLocation, getFileLockName(resourceName));
546            return resourceFileLock.exists();
547        }
548    
549        /**
550         * Returns the name of the open file lock of a resource.
551         *
552         * @param resourceName
553         * @return
554         */
555        private static String getFileLockName(String resourceName) {
556            return "." + resourceName + LOCK_FILE_EXT;
557        }
558    
559        /**
560         * Returns the resource name associated to a given lock file name.
561         *
562         * @param lockFileName
563         * @return
564         */
565        private static String getResourceNameFromLockFile(String lockFileName) {
566            return lockFileName
567                    .substring(1) // "."
568                    .substring(0, lockFileName.length() - 1 - LOCK_FILE_EXT.length() );
569        }

571        /**
572         * Creates a lock file for a resource name.
573         *
574         * @param resourceName
575         */
576        private void createLockFile(String resourceName) throws IOException {
577            File lockFile = new File(repositoryLocation, getFileLockName(resourceName));
578            lockFile.createNewFile();
579        }
580    
581        /**
582         * Removes a lock file.
583         * 
584         * @param resourceName
585         */
586        private void removeLockFile(String resourceName) {
587            File lockFile = new File(repositoryLocation, getFileLockName(resourceName) );
588            lockFile.delete();
589        }
590    
591        /**
592         * Removes all files that are locked and then inconsistent.
593         */
594        private void removeLockedResources() throws RepositoryException {
595            Map<String,Resource> resources = loadResources(repositoryLocation);
596            for(Resource r : resources.values()) {
597                if( isLocked( r.getName() ) ) {
598                    try {
599                        removeLockFile( r.getName());
600                    } catch (Exception e) {
601                        e.printStackTrace();
602                    }
603                    File resourceLocation = r.getLocation();
604                    try {
605                        resourceLocation.delete();
606                    } catch (Exception e) {
607                        e.printStackTrace();
608                    }
609                }
610            }

612        }
