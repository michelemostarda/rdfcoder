264        /**
265         * The array type.
266         */
267        public static class ArrayType extends StructuredType {
268    
269            private JType type;
270            private int   size;
271    
272            public ArrayType(JType t, int s) {
273                type = t;
274                size = s; 
275            }
276    
277            public JType getType() {
278                return type;
279            }
280    
281            public int getSize() {
282                return size;
283            }
284    
285            public void setInternalIdentifier(Identifier id) {
286                if(type instanceof ObjectType) {
287                    ( (ObjectType) type).setInternalIdentifier(id);
288                } else if( type instanceof ArrayType) {
289                    ( (ArrayType) type).setInternalIdentifier(id);
290                } else {
291                    throw new IllegalStateException();
292                }
293            }
294    
295            public String getInternalIdentifier() {
296                return type.getInternalIdentifier();
297            }
298    
299            public Identifier getIdentifier() {
300                return type.getIdentifier();
301            }
302    
303            public String getIdentifier(String internalRepresentation) {
304                return toIdentifier(internalRepresentation);
305            }
306    
307            //TODO: MED - check this.
308            private String toIdentifier(String id) {
309                return toURI(
310                        TYPE_ARRAY_STR + ":" +
311                        ( type instanceof JPrimitiveType ?
312                                type.getInternalIdentifier()
313                                    :
314                                "<" + id + ">" // Mark object type.
315                        ) +
316                        ":" + size
317                );
318            }
319    
320            public String toString() {
321                return getInternalIdentifier() + "["+ getSize() + "]";
322            }
323    
324            private static String ARRAY_PREFIX = toURI(TYPE_ARRAY_STR);
325    
326            public static ArrayType rdfTypeToType(String rdfType) {
327                if(rdfType.indexOf(ARRAY_PREFIX) != 0) {
328                    return null;
329                }
330                int i = rdfType.lastIndexOf(":");
331                int arraySize = Integer.parseInt(rdfType.substring(i + 1));
332                String type = rdfType.substring(ARRAY_PREFIX.length() + 1, i);
333                boolean typeIsObj = type.charAt(0) == '<';
334                if(typeIsObj) {
335                    type = type.substring(1, type.length() -2);
336                }
337                return new ArrayType( typeIsObj ? new ObjectType(IdentifierReader.readIdentifier(type) ) : javaTypeToJType(type), arraySize );
338            }
339        }

350        /**
351         * Converts a string representation of a RDF rdfType to a <code>JType</code>.
352         * @param rdfType the <code>JType</code> expressed as an identifier.
353         * @return the returned <code>JType</code> object.
354         */
355        public static final JType rdfTypeToJType(String rdfType) {
356    
357            // Primitive types.
358            if(CHAR.getIdentifier().getIdentifier().equals(rdfType)) {
359                return CHAR;
360            } else if(BYTE.getIdentifier().getIdentifier().equals(rdfType)) {
361                return BYTE;
362            } else if(SHORT.getIdentifier().getIdentifier().equals(rdfType)) {
363                return SHORT;
364            } else if(INT.getIdentifier().getIdentifier().equals(rdfType)) {
365                return INT;
366            } else if(LONG.getIdentifier().getIdentifier().equals(rdfType)) {
367                return LONG;
368            } else if(FLOAT.getIdentifier().getIdentifier().equals(rdfType)) {
369                return FLOAT;
370            } else if(DOUBLE.getIdentifier().getIdentifier().equals(rdfType)) {
371                return DOUBLE;
372            }  else if(BOOL.getIdentifier().getIdentifier().equals(rdfType)) {
373                return BOOL;
374            } else if(VOID.getIdentifier().getIdentifier().equals(rdfType)) {
375                return VOID;
376            }
377    
378            /* Complex types. */
379    
380            Object result;
381    
382            // Object type.
383            result = ObjectType.rdfTypeToType(rdfType);
384            if(result != null) {
385                return (ObjectType) result;
386            }
387    
388            // Interface type.
389            result = InterfaceType.rdfTypeToType(rdfType);
390            if(result != null) {
391                return (InterfaceType) result;
392            }
393    
394            // Array type.
395            result = ArrayType.rdfTypeToType(rdfType);
396            if(result != null) {
397                return (ArrayType) result;
398            }
399    
400            throw new IllegalArgumentException("Cannot convert RDF rdfType to a valid JType: '" + rdfType + "'.");
401        }

403        /**
404         * Converts a Java type name to a <code>JType</code>.
405         * @param type the Java type.
406         * @return the returned <code>JType</code> object.
407         */
408        public static final JType javaTypeToJType(String type) {
409            if(CHAR.getInternalIdentifier().equals(type)) {
410                return CHAR;
411            } else if(BYTE.getInternalIdentifier().equals(type)) {
412                return BYTE;
413            } else if(SHORT.getInternalIdentifier().equals(type)) {
414                return SHORT;
415            } else if(INT.getInternalIdentifier().equals(type)) {
416                return INT;
417            } else if(LONG.getInternalIdentifier().equals(type)) {
418                return LONG;
419            } else if(FLOAT.getInternalIdentifier().equals(type)) {
420                return FLOAT;
421            } else if(DOUBLE.getInternalIdentifier().equals(type)) {
422                return DOUBLE;
423            }  else if(BOOL.getInternalIdentifier().equals(type)) {
424                return BOOL;
425            } else if(VOID.getInternalIdentifier().equals(type)) {
426                return VOID;
427            } else {
428                throw new IllegalArgumentException("Cannot convert Java type to a valid JType: '" + type + "'.");
429            }
430        }

435        public enum JVisibility {
436            PUBLIC {
437                protected String getInternalIdentifier() {
438                    return "public";
439                }
440            },
441            PROTECTED {
442                protected String getInternalIdentifier() {
443                    return "protected";
444                }
445            },
446            DEFAULT {
447                protected String getInternalIdentifier() {
448                    return "default";
449                }
450            },
451            PRIVATE{
452                protected String getInternalIdentifier() {
453                    return "private";
454                }
455            };

463            /**
464             * Converts a JVisibility string to an enumerated value.
465             *
466             * @param s
467             * @return the converted type.
468             */
469            public static JVisibility toJVisibility(String s) {
470                for(JVisibility v : JVisibility.values() ) {
471                    if(v.getIdentifier().equals(s)) {
472                        return v;
473                    }
474                }
475                throw new IllegalArgumentException();
476            }

479        /**
480         * Defines the Java modifiers.
481         */
482        public enum JModifier {
483    
484            /**
485             * Entity is declared abstract.
486             */
487            ABSTRACT {
488                private static final byte FA = 0x1;
489                public byte value() {
490                    return FA;
491                }
492                public boolean isValue(byte b) {
493                    return (b & FA) == FA;
494                }
495            },
496    
497            /**
498             * Entity is declared final.
499             */
500            FINAL {
501                private static final byte FB = 0x2;
502                public byte value() {
503                    return FB;
504                }
505                public boolean isValue(byte b) {
506                    return (b & FB) == FB; 
507                }
508            },
509    
510            /**
511             * Entity is declared static.
512             */
513            STATIC {
514                private static final byte FS = 0x4;
515                public byte value() {
516                    return FS;
517                }
518                public boolean isValue(byte b) {
519                    return (b & FS) == FS;
520                }
521            },
522    
523            /**
524             * Entity is declared volatile.
525             */
526            VOLATILE {
527                private static final byte FV = 0x8;
528                public byte value() {
529                    return FV;
530                }
531                public boolean isValue(byte b) {
532                    return (b & FV) == FV;
533                }
534            },
535    
536            /**
537             * Entity is declared native.
538             */
539            NATIVE {
540                private static final byte FN = 0x10;
541                public byte value() {
542                    return FN;
543                }
544                public boolean isValue(byte b) {
545                    return (b & FN) == FN;
546                }
547            },
548    
549            /**
550             * Entity is declared transient.
551             */
552            TRANSIENT {
553                private static final byte TR = 0x20;
554                public byte value() {
555                    return TR;
556                }
557                public boolean isValue(byte b) {
558                    return (b & TR) == TR;
559                }
560            },
561    
562            /**
563             * Entity is declared synchronized.
564             */
565            SYNCHRONIZED {
566                private static final byte SY = 0x40;
567                public byte value() {
568                    return SY;
569                }
570                public boolean isValue(byte b) {
571                    return (b & SY) == SY;
572                }
573            };
574    
575            //TODO: manage CONST modifier.
576    
577            public abstract byte    value();
578            public abstract boolean isValue(byte b);
579    
580            public static JModifier[] getModifiers(byte b) {
581                List<JModifier> modifiers = new ArrayList<JModifier>(JModifier.values().length);
582                for( JModifier v : JModifier.values()) {
583                    if(v.isValue(b)) {
584                        modifiers.add(v);
585                    }
586                }
587                return modifiers.toArray(new JModifier[modifiers.size()]);
588            }
589    
590            public static JModifier[] toModifiers(String m) {
591                byte b = Byte.parseByte(m);
592                return getModifiers(b);
593            }
594    
595            public static Byte toByte(JModifier[] modifiers) {
596                byte result = 0;
597                for(JModifier m : modifiers) {
598                    result |= m.value();
599                }
600                return result;
601            }
602        }
