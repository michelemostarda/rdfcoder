019    package com.asemantics.rdfcoder.model;
020    
021    import com.asemantics.rdfcoder.model.java.JavaCodeModel;
022    
023    import java.util.ArrayList;
024    import java.util.Iterator;
025    import java.util.List;
026    
027    /**
028     * Contains base methods common to every <i>Code Model</i>.
029     */
030    public abstract class CodeModelBase implements CodeModel, BackTrackingSupport {
031    
032    
033        /* BEGIN: Asset classes and properties. */
034    
035        /**
036         * The asset term.
037         */
038        public static final String ASSET_KEY        = "asset";
039    
040        /**
041         * Asset subject.
042         */
043        public static final String ASSET            = toPrefix(ASSET_KEY);
044    
045        /**
046         * Asset prefix.
047         */
048        public static final String ASSET_PREFIX     = toPrefix(ASSET_KEY);
049    
050        /**
051         * The model asset contains library.
052         */
053        public static final String CONTAINS_LIBRARY = toPropertyURI("contains_library");
054    
055        /**
056         * Location of a library.
057         */
058        public static final String LIBRARY_LOCATION = toPropertyURI("library_location");
059    
060        /**
061         * Date of parsing of the library.
062         */
063        public static final String LIBRARY_DATETIME = toPropertyURI("library_date");
064    
065        /* END: Asset classes and properties. */
066    
067    
068        /**
069         * Marks unqualified types.
070         */
071        public static final String UNQUALIFIED_PREFIX = toPrefix("unqualified");
072    
073        /**
074         * URI-fier method.
075         *
076         * @param prop
077         * @return the URI form as string.
078         */
079        protected static String toURI(String prop) {
080            return CODER_URI + prop;
081        }
082    
083        /**
084         * URI-fier method.
085         *
086         * @param prop
087         * @return the URI form as string.
088         */
089        protected static String toPropertyURI(String prop) {
090            return CODER_URI_BASE + "/" + prop;
091        }
092    
093        /**
094         * URI-fier method for class specific names.
095         *
096         * @param prop
097         * @return the URI form as string with the first two class chars upper case.
098         */
099        protected static String toClassURI(String prop) {
100            return CODER_URI + prop.substring(0, 2).toUpperCase() + prop.substring(2);
101        }
102    
103        /**
104         * Prefix-fier method.
105         * @param pref
106         * @return the prefix.
107         */
108        protected static final String toPrefix(String pref) {
109            return pref + PREFIX_SEPARATOR;
110        }
111    
112        /**
113         * Unique identifier discriminator counter.
114         */
115        private long _counter = 0;
116    
117        /**
118         * Generates a temporary unique identifier.
119         */
120        public Identifier generateTempUniqueIdentifier() {
121            return IdentifierBuilder
122                    .create()
123                    .pushFragment(System.currentTimeMillis() + "_" + _counter++, TEMPORARY_TYPE_ID_PREFIX)
124                    .build();
125        }
126    
127        /**
128         * Prefixes a full qualifier with the given prefix validating both first.
129         *
130         * @param prefix
131         * @param path
132         * @return the prefixed string.
133         */
134        public static String prefixFullyQualifiedName(String prefix, String path) {
135            if(prefix == null || prefix.trim().length() == 0 || path == null) {
136                throw new IllegalArgumentException();
137            }
138    
139            int prefixSeparatorIndex = path.indexOf(PREFIX_SEPARATOR);
140            // Invalid path: prefix is ""
141            if(prefixSeparatorIndex == 0) {
142                throw new IllegalArgumentException("Invalid path: '" + path + "'");
143            }
144            // Removes the last prefix.
145            if(prefixSeparatorIndex > 0) {
146                path = path.substring(prefixSeparatorIndex + 1);
147            }
148    
149            return prefix + path;
150        }
151    
152        /**
153         * Replaces temporary identifiers with final types.
154         *
155         * @param identifier
156         * @param qualifiedType
157         * @return the number of replaced identifiers.
158         * @see #generateTempUniqueIdentifier()
159         */
160        public int replaceIdentifierWithQualifiedType(final Identifier identifier, final Identifier qualifiedType) {
161            int effectedTriples;
162    
163            /*
164            if( RDFCoder.assertions() && identifier.indexOf(CodeModel.PREFIX_SEPARATOR) == -1) {
165                throw new IllegalArgumentException("identifier: " + identifier + " qualified type: " + qualifiedType);
166            }
167            */
168    
169            final String identifierStr = identifier.getIdentifier();
170    
171            List newTriples = new ArrayList();
172            String[] nextTriple;
173    
174            // Replacing all subjects.
175    
176            // Creates new triples image.
177            TripleIterator ti = searchTriples(identifierStr, CodeModel.ALL_MATCH, JavaCodeModel.ALL_MATCH);
178            while(ti.next()) {
179                newTriples.add( new String[] { ti.getPredicate(), ti.getObject() } );
180            }
181            effectedTriples = newTriples.size();
182    
183            // Delete old triples.
184            Iterator<String[]> newTriplesIter = newTriples.iterator();
185            while(newTriplesIter.hasNext()) {
186                nextTriple = newTriplesIter.next();
187                removeTriple( identifierStr, nextTriple[0], nextTriple[1] );
188            }
189            // Add new ones.
190            newTriplesIter = newTriples.iterator();
191            while(newTriplesIter.hasNext()) {
192                nextTriple = newTriplesIter.next();
193                removeTriple( identifierStr, nextTriple[0], nextTriple[1] );
194            }
195    
196            // Replacing all objects.
197    
198            newTriples.clear();
199    
200            // Creates new triples image.
201            ti = searchTriples(CodeModel.ALL_MATCH, CodeModel.ALL_MATCH, identifierStr);
202            while(ti.next()) {
203                newTriples.add( new String[] {ti.getSubject(), ti.getPredicate() } );
204            }
205            effectedTriples += newTriples.size();
206    
207            // Delete old triples.
208            newTriplesIter = newTriples.iterator();
209            while(newTriplesIter.hasNext()) {
210                nextTriple = newTriplesIter.next();
211                removeTriple( nextTriple[0], nextTriple[1], identifierStr );
212            }
213            // Add new ones.
214            newTriplesIter = newTriples.iterator();
215            while(newTriplesIter.hasNext()) {
216                nextTriple = newTriplesIter.next();
217                removeTriple( nextTriple[0], nextTriple[1], qualifiedType.getIdentifier() );
218            }
219    
220            return effectedTriples;
221        }
222    
223    }
