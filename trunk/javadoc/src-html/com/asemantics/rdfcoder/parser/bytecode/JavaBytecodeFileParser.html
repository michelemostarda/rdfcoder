087        /**
088         * Parses a <i>.class</i> file.
089         * 
090         * @param classloader
091         * @param inputStream
092         * @param fileName
093         * @throws IOException
094         */
095        public void parse(JavaBytecodeClassLoader classloader, InputStream inputStream, String fileName)
096        throws IOException, ParserException {
097            containersStack.clear();
098    
099            // Parse bytecode.
100            ClassParser classParser = new ClassParser( inputStream, fileName );
101            JavaClass javaClass = classParser.parse();
102    
103            final JavaCodeHandler javaCodeHandler = (JavaCodeHandler) getParseHandler();
104            ObjectsTable objectsTable = getObjectsTable();
105    
106            // Package.
107            javaCodeHandler.startPackage(IdentifierReader.readPackage( javaClass.getPackageName() ) );
108            try {
109    
110                // Objects table update.
111                objectsTable.addObject( IdentifierReader.readFullyQualifiedClass(javaClass.getClassName()) );
112    
113                // Recursive processing.
114                Set processed = new HashSet();
115                processClass(javaCodeHandler, classloader, processed, javaClass);
116    
117            } catch(Exception e) {
118                logger.error("An error occurred while parsing bytecode file.", e);
119                throw new ParserException("An error occurred while parsing bytecode file.", e);
120            } finally {
121                // Close package.
122                javaCodeHandler.endPackage();
123            }
124        }
125    
126        public void parse(JavaBytecodeClassLoader classloader, File file) throws IOException, ParserException {
127            if(file == null) {
128                throw new NullPointerException();
129            }
130            if( ! file.exists() ) {
131                throw new IllegalArgumentException("cannot find file: " + file.getAbsolutePath());
132            }
133    
134            InputStream inputStream = new FileInputStream(file);
135            parse(classloader, inputStream, file.getAbsolutePath());
136            inputStream.close();
137        }

139        public void parse(File file) throws IOException, ParserException {
140            //TODO: HIGH - check this method: JavaBytecodeClassLoader must not be null.
141            parse(null, file);
142        }

144        protected void processClass(
145                JavaCodeHandler javaCodeHandler,
146                JavaBytecodeClassLoader classloader,
147                Set<String> processed,
148                JavaClass javaClass
149        ) throws ParserException {
150    
151            if(logger.isDebugEnabled()) {
152                logger.debug("Processing class " + javaClass);
153            }
154    
155            containersStack.push( javaClass );
156    
157            javaCodeHandler.startCompilationUnit(javaClass.toString());
158    
159            // The stack is currectly handled despite errors.
160            try {
161    
162                Field[] fields = javaClass.getFields();
163                List<String> enumElements = new ArrayList<String>();
164    
165                // Extracts class modifiers.
166                JavaCodeModel.JModifier[] modifiers = extractModifiers(javaClass);
167    
168                if( javaClass.isEnum() ) { // Enumeration.
169                    //TODO: MED - Check this solution.
170                    for(int i = 0; i < fields.length - 1; i++) {   // Skip last element containing $VALUE
171                        if( fields[i].getType().toString().equals( javaClass.getClassName() ) ) { // Excluding real enumElements.
172                            enumElements.add( fields[i].getName() );
173                        }
174                    }
175                    javaCodeHandler.startEnumeration(
176                            modifiers,
177                            toVisibility( javaClass ),
178                            toQualifiedEnumName( javaClass ),
179                            enumElements.toArray( new String[ enumElements.size() ] )
180                    );
181                } else if( javaClass.isClass() ) { // Class.
182                    javaCodeHandler.startClass(
183                       modifiers,
184                       toVisibility(javaClass),
185                       toQualifiedClassName(javaClass),
186                       toQualifiedSuperClassName(javaClass),
187                       toQualifiedInterfaces(javaClass)
188                    );
189                } else if( javaClass.isInterface() ) { // Interface.
190                    javaCodeHandler.startInterface(
191                       toQualifiedInterfaceName(javaClass),
192                       toQualifiedInterfaces(javaClass)
193                    );
194                } else {
195                    throw new IllegalStateException();
196                }
197    
198                // Fields.
199                for (Field field : fields) {
200                    if (enumElements.contains(field.getName())) {
201                        continue;
202                    }
203                    javaCodeHandler.attribute(
204                            extractModifiers(field),
205                            toVisibility(field),
206                            toQualifiedAttribute(field),
207                            bcelTypeToJType(field.getType()),
208                            toConstantValue(field.getConstantValue())
209                    );
210                }
211                enumElements.clear();
212    
213    
214                // Methods.
215                Method[] methods = javaClass.getMethods();
216                for(int i = 0; i < methods.length; i++) {
217                    if( INIT_METHOD.equals(methods[i].getName()) ) { // Constructors.
218                        javaCodeHandler.constructor(
219                            extractModifiers(methods[i]),
220                            toVisibility( methods[i] ),
221                            methods[i].getSignature().hashCode(),
222                            toParameterNames( methods[i].getArgumentTypes()  ),
223                            toParameterTypes( methods[i].getArgumentTypes()  ),
224                            toExceptionTypes( methods[i].getExceptionTable() )
225                        );
226                    } else {
227                        javaCodeHandler.method(
228                                extractModifiers(methods[i]),
229                                toVisibility(methods[i]),
230                                toQualifiedMethod(methods[i]),
231                                methods[i].getSignature().hashCode(),
232                                toParameterNames(methods[i].getArgumentTypes()),
233                                toParameterTypes(methods[i].getArgumentTypes()),
234                                bcelTypeToJType(methods[i].getReturnType()),
235                                toExceptionTypes(methods[i].getExceptionTable())
236                        );
237                    }
238                }
239    
240                // Inner classes. If class loader is not defined the inner classes are ignored.
241                if(classloader != null) {
242                    String innerClassName;
243                    InnerClass[] innerClasses = innerClasses(javaClass);
244                    ConstantPool constantPool = javaClass.getConstantPool();
245                    for (InnerClass innerClass1 : innerClasses) {
246                        innerClassName = constantPool.getConstantString(
247                                innerClass1.getInnerClassIndex(),
248                                Constants.CONSTANT_Class
249                        );
250                        if (processed.contains(innerClassName)) {
251                            continue;
252                        } else {
253                            processed.add(innerClassName);
254                        }
255                        if (logger.isDebugEnabled()) {
256                            logger.debug("Processing inner class: " + innerClassName);
257                        }
258                        innerClassName = Utility.compactClassName(innerClassName);
259                        JavaClass innerClass;
260                        try {
261                            innerClass = classloader.loadClass(innerClassName);
262                            if (innerClass == null) {
263                                if (logger.isDebugEnabled()) {
264                                    logger.debug("Cannot load class: " + innerClassName);
265                                }
266                            }
267                        } catch (IOException ioe) {
268                            javaCodeHandler.parseError(javaClass.getClassName(), "[" + ioe.getClass().getName() + "]: " + ioe.getMessage());
269                            ioe.printStackTrace();
270                        }
271                    }
272                }
273    
274            } catch(Exception e) {
275                e.printStackTrace(); // TODO: HIGH - Remove me.
276                throw new ParserException( String.format("Error while parsing class %s", javaClass.getClassName() ), e);   
277            } finally {
278    
279                // End container.
280                if( javaClass.isClass() ) {
281                    javaCodeHandler.endClass();
282                } else if( javaClass.isInterface() ) {
283                    javaCodeHandler.endInterface();
284                } else if(javaClass.isEnum()) {
285                    javaCodeHandler.endEnumeration();
286                } else {
287                    throw new IllegalStateException();
288                }
289    
290                containersStack.pop();
291                javaCodeHandler.endCompilationUnit();
292            }
293        }

295        private InnerClass[] innerClasses(JavaClass javaClass) {
296            Attribute[] attributes = javaClass.getAttributes();
297            for (Attribute attribute : attributes) {
298                if (attribute instanceof InnerClasses) {
299                    InnerClasses innerClasses = (InnerClasses) attribute;
300                    return innerClasses.getInnerClasses();
301                }
302            }
303            return INNER_CLASS_EMPTY;
304        }

306        private JavaCodeModel.JVisibility toVisibility(AccessFlags af) {
307            if( af.isPublic() ) {
308                return JavaCodeModel.JVisibility.PUBLIC;
309            } else if( af.isProtected() ) {
310                return JavaCodeModel.JVisibility.PROTECTED;
311            } else if( af.isPrivate() ) {
312                return JavaCodeModel.JVisibility.PRIVATE;
313            } else {
314                return JavaCodeModel.JVisibility.DEFAULT;
315            }
316        }

318        private Identifier getContainerPath() {
319            if( containersStack.size() == 0 ) {
320                throw new IllegalStateException();
321            }
322            return IdentifierReader.readFullyQualifiedClass( containersStack.peek().getClassName() );
323        }

325        private Identifier toQualifiedClassName(JavaClass javaClass) {
326            return IdentifierReader.readFullyQualifiedClass( javaClass.getClassName() );
327        }

330            return IdentifierReader.readFullyQualifiedClass( javaClass.getClassName() );
331        }

334            return IdentifierReader.readFullyQualifiedInterface( javaClass.getClassName() );
335        }
336    
337        private Identifier toQualifiedEnumName(JavaClass javaClass) {
338            return IdentifierReader.readFullyQualifiedEnumeration( javaClass.getClassName() );
339        }

341        private Identifier[] toQualifiedInterfaces(JavaClass javaClass) {
342            JavaClass[] interfaces;
343            try {
344                interfaces = javaClass.getAllInterfaces();
345            } catch (ClassNotFoundException e) {
346                return null;
347            }
348            Identifier[] result = new Identifier[interfaces.length];
349            for(int i = 0; i < interfaces.length; i++) {
350                result[i] = IdentifierReader.readFullyQualifiedInterface( interfaces[i].getClassName() );
351            }
352            return result;
353        }
354    
355        private Identifier toQualifiedAttribute(Field field) {
356            return getContainerPath().copy().pushFragment(field.getName(), JavaCodeModel.ATTRIBUTE_KEY).build();
357        }
358    
359        private Identifier toQualifiedMethod(Method method) {
360            return getContainerPath().copy().pushFragment(method.getName(), JavaCodeModel.METHOD_KEY).build();
361        }

363        private String toConstantValue(ConstantValue cv) {
364            if(cv == null) {
365                return null;
366            }
367            ConstantPool cp =  cv.getConstantPool();
368            Constant c = cp.getConstant( cv.getConstantValueIndex() );
369            return c.toString();
370        }

372        private JavaCodeModel.JType bcelTypeToJType(org.apache.bcel.generic.Type type) {
373    
374            // Convert basic types.
375            if(type == org.apache.bcel.generic.Type.VOID ) {
376                 return JavaCodeModel.VOID;
377            } else if( type == org.apache.bcel.generic.Type.BOOLEAN ) {
378                return JavaCodeModel.BOOL;
379            } else if( type == org.apache.bcel.generic.Type.INT ) {
380                return JavaCodeModel.INT;
381            } else if( type == org.apache.bcel.generic.Type.SHORT ) {
382                return JavaCodeModel.SHORT;
383            } else if( type == org.apache.bcel.generic.Type.BYTE ) {
384                return JavaCodeModel.BYTE;
385            } else if( type == org.apache.bcel.generic.Type.LONG ) {
386                return JavaCodeModel.LONG;
387            } else if( type == org.apache.bcel.generic.Type.DOUBLE) {
388                return JavaCodeModel.DOUBLE;
389            } else if( type == org.apache.bcel.generic.Type.FLOAT ) {
390                return JavaCodeModel.FLOAT;
391            } else if( type == org.apache.bcel.generic.Type.CHAR ) {
392                return JavaCodeModel.CHAR;
393            }
394    
395            // Convert object types.
396            if( type instanceof org.apache.bcel.generic.ObjectType) {
397                org.apache.bcel.generic.ObjectType bcelOT = (org.apache.bcel.generic.ObjectType) type;
398                return new JavaCodeModel.ObjectType( IdentifierReader.readFullyQualifiedClass(bcelOT.getClassName() ) );
399            }
400    
401            // Array types.
402            if ( type instanceof org.apache.bcel.generic.ArrayType ) {
403                org.apache.bcel.generic.ArrayType arrayType = (org.apache.bcel.generic.ArrayType) type;
404                return new JavaCodeModel.ArrayType(
405                        bcelTypeToJType(arrayType.getBasicType() ),
406                        arrayType.getDimensions()
407                );
408            }
409    
410            throw new IllegalArgumentException("cannot convert " + type);
411    
412        }

414        private String toParameterName(org.apache.bcel.generic.Type type ) {
415            // Generate name for basic type.
416            if( type == org.apache.bcel.generic.Type.BOOLEAN ) {
417                return "b";
418            } else if( type == org.apache.bcel.generic.Type.INT ) {
419                return "i";
420            } else if( type == org.apache.bcel.generic.Type.SHORT ) {
421                return "s";
422            } else if( type == org.apache.bcel.generic.Type.BYTE ) {
423                return "e";
424            } else if( type == org.apache.bcel.generic.Type.LONG ) {
425                return "l";
426            } else if( type == org.apache.bcel.generic.Type.DOUBLE) {
427                return "d";
428            } else if( type == org.apache.bcel.generic.Type.FLOAT ) {
429                return "f";
430            } else if( type == org.apache.bcel.generic.Type.CHAR ) {
431                return "c";
432            }
433    
434            // Generate name for object type.
435            if( type instanceof org.apache.bcel.generic.ObjectType) {
436                org.apache.bcel.generic.ObjectType bcelOT = (org.apache.bcel.generic.ObjectType) type;
437                String cn = bcelOT.getClassName();
438                int i = cn.lastIndexOf(".");
439                String result = cn.substring(i + 1);
440                if("String".equals(result)) {
441                    return "s";
442                } else if("Object".equals(result)) {
443                    return "o";
444                } else {
445                    return result;
446                }
447            }
448    
449            // Generate name for array type.
450            if ( type instanceof org.apache.bcel.generic.ArrayType ) {
451                org.apache.bcel.generic.ArrayType arrayType = (org.apache.bcel.generic.ArrayType) type;
452                String typeName = toParameterName( arrayType.getBasicType() );
453                return typeName + "s";
454            }
455    
456            throw new IllegalArgumentException("cannot generate name for " + type);
457        }

460        private String[] toParameterNames( org.apache.bcel.generic.Type[] type ) {
461            HashMap<String,Integer> namesMap = new HashMap<String,Integer>();
462            String[] names = new String[type.length];
463            String designedName;
464            for(int i = 0; i < type.length; i++) {
465                designedName = toParameterName( type[i] ).toLowerCase();
466                Integer occurrences = namesMap.get(designedName);
467                if(occurrences == null) { occurrences = 0; }
468                occurrences++;
469                namesMap.put(designedName, occurrences);
470                names[i] = designedName + ( occurrences == 1 ? "" : occurrences );
471            }
472            namesMap.clear();
473            return names;
474        }

476        private JavaCodeModel.JType[] toParameterTypes( org.apache.bcel.generic.Type[] types ) {
477            JavaCodeModel.JType[] result = new JavaCodeModel.JType[types.length];
478            for(int i = 0; i < types.length; i++) {
479                result[i] = bcelTypeToJType(types[i]);
480            }
481            return result;
482        }

484        private JavaCodeModel.ExceptionType[] toExceptionTypes(ExceptionTable exceptionTable) {
485            if(exceptionTable == null) {
486                return null;
487            }
488            String[] exceptionNamesStr = exceptionTable.getExceptionNames();
489            Identifier[] exceptionNames = new Identifier[exceptionNamesStr.length];
490            int i = 0;
491            for( String exceptionNameStr : exceptionNamesStr ) {
492                exceptionNames[i++] = IdentifierReader.readFullyQualifiedClass(exceptionNameStr);
493            }
494            JavaCodeModel.ExceptionType[] exceptions = new JavaCodeModel.ExceptionType[exceptionNames.length];
495            for(i = 0; i < exceptionNames.length; i++) {
496                exceptions[i] = new JavaCodeModel.ExceptionType(exceptionNames[i]);
497            }
498            return exceptions;
499        }

501        /**
502         * Extracts the list of modifiers associated to the given entity.
503         * @param accessFlags
504         * @return
505         */
506        private JavaCodeModel.JModifier[] extractModifiers(AccessFlags accessFlags) {
507            List<JavaCodeModel.JModifier> modifiers =
508                    new ArrayList<JavaCodeModel.JModifier>(JavaCodeModel.JModifier.values().length);
509            
510            if( accessFlags.isAbstract() ) {
511                modifiers.add(JavaCodeModel.JModifier.ABSTRACT);
512            }
513            if( accessFlags.isFinal() ) {
514                modifiers.add(JavaCodeModel.JModifier.FINAL);
515            }
516            if( accessFlags.isStatic() ) {
517                modifiers.add(JavaCodeModel.JModifier.STATIC);
518            }
519            if( accessFlags.isVolatile() ) {
520                modifiers.add(JavaCodeModel.JModifier.VOLATILE);
521            }
522            if( accessFlags.isNative() ) {
523                modifiers.add(JavaCodeModel.JModifier.NATIVE);
524            }
525            if( accessFlags.isTransient() ) {
526                modifiers.add(JavaCodeModel.JModifier.TRANSIENT);
527            }
528            if( accessFlags.isSynchronized() ) {
529                modifiers.add(JavaCodeModel.JModifier.SYNCHRONIZED);
530            }
531            return modifiers.toArray(new JavaCodeModel.JModifier[modifiers.size()]);
532        }
